namespace = sb_trek

# =============================================================================
# Spes Bona - Great Trek Events
# =============================================================================
# V2 trek structure:
#   - ORA is the only inland Boer tag at game start.
#   - MTB opens the inland frontier war against ORA.
#   - Blood River remains ORA-only.
#   - TRN only appears later, once ORA establishes a real foothold beyond the Vaal.
#
# This file therefore focuses on:
#   - the live ORA-first trek spine
#   - the delayed TRN split and immediate inland republican content
#   - a few legacy TRN-aids-ORA placeholders kept for save compatibility
#     rather than for new-game flow
#
# Later recognition / convention content now lives in
# `events/sb_trek_recognition_events.txt`.
#
# EVENT CATALOG:
#   sb_trek.001     MTB opens the inland native-uprising play against ORA
#   sb_trek.002     MTB defeat and frontier retreat split
#   sb_trek.004     Legacy TRN receives Mzilikazi's wrath and sends for help
#   sb_trek.005     Legacy ORA answers TRN's call for Potgieter
#   sb_trek.006     Founding of Potchefstroom / delayed Transvaal split
#   sb_trek.030     Stage 1 reward — ORA economy established
#   sb_trek.031     Stage 1 reward — TRN economy established
#   sb_trek.040     Stage 2 reward — ORA controls Vrystaat + mass migration
#   sb_trek.041     Stage 2 reward — TRN controls Transvaal + mass migration
#   sb_trek.100     JE completion (both stages done)
#   sb_trek.110-.131  post-trek republican forks, LYD, ZPB spawn hooks
#   sb_trek.220     Griqualand East formation
#   sb_trek.221     Griqualand East willing annexation into full-liberal CAP
#
# MASS MIGRATION PATTERN (Stage 2 events):
#   Uses create_mass_migration from V3's South America migration pattern.
#   Source: CAP (Cape Colony), Culture: Boer, Target: highest migration_pull state.
#   This creates a visible wave of Boer settlers arriving over several years.
#
# CROSS-REFERENCES:
#   JE: common/journal_entries/1-02_sb_great_trek.txt
#   Migration modifier: common/static_modifiers/sb_modifiers.txt (sb_trek_migration_pull)
#   Ndebele tensions: common/on_actions/sb_on_actions.txt (ORA-MTB tension = 90)
# =============================================================================

sb_trek.001 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:MTB
		NOT = { has_variable = sb_ndebele_opening_play_fired }
		is_at_war = no
		aggressive_diplomatic_plays_permitted = yes
		is_country_alive = yes
	}

	immediate = {
		set_variable = sb_ndebele_opening_play_fired
		create_diplomatic_play = {
			# Use the vanilla decentralized-native flow here. This play already
			# carries the built-in colonization-rights article, so avoid adding
			# custom opening articles on top or the presentation/order gets odd.
			type = dp_native_uprising
			target_country = c:ORA
		}
	}
}

# Legacy compatibility branch. New V2 starts no longer have a day-1 TRN, so
# fresh campaigns should never reach this event. It is left in place so older
# saves or partial content branches that still surface the call-for-help flow
# do not hard break.
sb_trek.004 = {
	type = country_event
	placement = root

	title = sb_trek.004.t
	desc = sb_trek.004.d
	flavor = sb_trek.004.f

	event_image = {
		video = "gfx/event_pictures/africa_soldiers_breaking_protest.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	duration = 3

	trigger = {
		country_definition = cd:ORA
		c:ORA ?= {
			is_country_alive = yes
		}
	}

	option = {
		name = sb_trek.004.a
		default_option = yes
		if = {
			limit = {
				c:ORA ?= {
					is_country_alive = yes
				}
			}
			c:ORA ?= {
				trigger_event = { id = sb_trek.005 days = 5 popup = yes }
			}
		}
	}
}

# Legacy partner to sb_trek.004. In new V2 flow ORA fights MTB directly and TRN
# does not exist yet, so this remains a dormant fallback rather than an active
# branch.
sb_trek.005 = {
	type = country_event
	placement = root

	title = sb_trek.005.t
	desc = sb_trek.005.d
	flavor = sb_trek.005.f

	event_image = {
		video = "gfx/event_pictures/africa_soldiers_breaking_protest.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	duration = 3

	trigger = {
		country_definition = cd:ORA
		c:TRN ?= {
			is_country_alive = yes
		}
	}

	option = {
		name = sb_trek.005.a
		default_option = yes
		# Explicitly add ORA to TRN's side on the already-running native uprising.
		# Only guard against duplicate joins to this specific play, not against
		# ORA being involved in some unrelated diplomatic play elsewhere.
			
		random_diplomatic_play = {
			limit = {
				AND = {
					target = c:TRN
					initiator = c:MTB
				}
			}
			add_target_backers = { c:ORA }
		}

		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_trek.005.b
		every_scope_treaty = {
			limit = {
				binds = c:TRN
				binds = c:ORA
				any_scope_article = {
					has_type = defensive_pact
				}
			}
			withdraw = {
				country = c:ORA
			}
		}
		c:TRN ?= {
			change_relations = {
				country = root
				value = -20
			}
		}
		ai_chance = {
			base = 0
		}
	}
}

# In Mzilikazi's Wake
# Keep this as a Spes Bona-owned copy of the vanilla Highveld withdrawal event
# so the retreat-to-Zimbabwe structure stays easy to tune without leaning on the
# disabled vanilla JE/event chain directly.
sb_trek.002 = {
	type = country_event
	placement = ROOT

	title = sb_trek.002.t
	desc = sb_trek.002.d
	flavor = sb_trek.002.f

	duration = 3

	event_image = {
		video = "middleeast_jungle_expedition"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
		country_definition = cd:ORA
	}

	immediate = {
		c:MTB ?= {
			save_scope_as = ndebele
			ruler = {
				save_scope_as = mzilikazi
			}
		}
		ROOT = {
			save_scope_as = transvaal
		}
		c:ZUL ?= {
			save_scope_as = zulu
		}
		s:STATE_TRANSVAAL = {
			save_scope_as = transvaal_region
		}
	}

	option = {
		name = sb_trek.002.a
		default_option = yes

		hidden_effect = {
			if = {
				limit = {
					exists = c:MSH
				}
				c:MTB ?= {
					annex = c:MSH
				}
			}
		}
			c:MTB ?= {
				s:STATE_VRYSTAAT = {
					set_owner_of_provinces = {
						country = c:ORA
						provinces = {
							xF17D3D xF075D7 x373E53 x9D6A16 xCEE3D2
						}
					}
					set_owner_of_provinces = {
						country = c:TSW
						provinces = {
							x83E266 x6253E3 x19AE59 xED7851 x75FA4B
						}
					}
				}
			every_scope_state = {
				limit = {
					OR = {
						state_region = s:STATE_TRANSVAAL
						state_region = s:STATE_NORTHERN_TRANSVAAL
					}
				}
				set_state_owner = c:TSW
			}
			every_scope_state = {
				limit = {
					state_region = s:STATE_EAST_TRANSVAAL
				}
				set_state_owner = c:PDI
			}
		}
		if = {
			limit = {
				is_ai = yes
			}
			c:MTB ?= {
				if = {
					limit = { has_modifier = sb_native_conscription_MTB }
					remove_modifier = sb_native_conscription_MTB
				}
				if = {
					limit = { NOT = { has_modifier = sb_native_conscription_MTB_player } }
					add_modifier = {
						name = sb_native_conscription_MTB_player
						months = -1
					}
				}
			}
		}

		ai_chance = {
			base = 100
		}
	}
}

sb_trek.006 = {
	type = country_event
	placement = root

	title = sb_trek.006.t
	desc = sb_trek.006.d
	flavor = sb_trek.006.f

	event_image = {
		video = "gfx/event_pictures/africa_construction_colony.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	duration = 3

	trigger = {
		# The Potchefstroom split can be surfaced either to ORA itself or to a
		# player-controlled Natalia, but the actual spawn work still runs from ORA
		# scope so the inland ownership handoff stays consistent.
		OR = {
			country_definition = cd:ORA
			country_definition = cd:NAL
		}
		NOT = { exists = c:TRN }
		c:ORA ?= {
			is_country_alive = yes
			has_variable = sb_trn_v2_spawn_pending_var
			NOT = { has_variable = sb_trn_v2_spawned_var }
			# Inline the west-Transvaal foothold audit directly here. This avoids a
			# boot-time dependency on a custom scripted trigger for the one place the
			# split event needs to know whether ORA has actually established itself
			# in the Potchefstroom corridor.
			OR = {
				p:xD0A010.state.owner = this
				p:x8BDC4D.state.owner = this
				p:xA22DC2.state.owner = this
				p:xD02190.state.owner = this
				p:x6C1786.state.owner = this
				p:x9A137B.state.owner = this
			}
		}
	}

	option = {
		name = sb_trek.006.a
		trigger = {
			is_player = yes
		}
		c:ORA ?= {
			sb_spawn_transvaal_republic_v2 = yes
		}
		play_as = c:TRN
		ai_chance = {
			base = 0
		}
	}

	option = {
		name = sb_trek.006.b
		default_option = yes
		c:ORA ?= {
			sb_spawn_transvaal_republic_v2 = yes
		}
		ai_chance = {
			base = 100
		}
	}
}
