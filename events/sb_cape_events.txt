# Cape Qualified Franchise demand — fired from the first advance checkpoint
# while CAP is enacting Cultural Exclusion. This keeps CQF tied to parliamentary
# reform politics rather than making it an automatic pragmatic-victory reward.
sb_cape.130 = {
	type = country_event
	placement = root

	title = sb_cape.130.t
	desc = sb_cape.130.d
	flavor = sb_cape.130.f

	category = enactment

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_sufferage.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/principles_icons/creative_legislature.dds"
	duration = 3

	trigger = {
		country_definition = cd:CAP
		has_variable = sb_cqf_first_checkpoint_pending_var
		is_enacting_law = law_type:law_cultural_exclusion
		currently_enacting_law ?= {
			NOT = { has_amendment = amendment_type:amendment_sb_cape_qualified_franchise }
		}
	}

	immediate = {
		remove_variable = sb_cqf_first_checkpoint_pending_var
		currently_enacting_law = {
			save_scope_as = current_law_scope
		}
		ig:ig_intelligentsia ?= {
			save_scope_as = cqf_sponsor
		}
	}

	cancellation_trigger = {
		NOT = { currently_enacting_law = scope:current_law_scope }
	}

	option = {
		name = sb_cape.130.a
		default_option = yes
		currently_enacting_law = {
			add_amendment = {
				type = amendment_sb_cape_qualified_franchise
				sponsor = scope:cqf_sponsor
				cooldown = 120
			}
		}
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_sb_cape_liberal
			}
			add_modifier = {
				name = sb_cape_liberal_cqf_satisfied
				months = 60
			}
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_cape.130.b
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_sb_cape_liberal
			}
			add_modifier = {
				name = sb_cape_liberal_cqf_refused
				months = 60
			}
		}
		ai_chance = {
			base = 0
		}
	}
}

	title = sb_cape.103.t
	desc = sb_cape.103.d

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/principles_icons/creative_legislature.dds"

	duration = 3

	trigger = {
		country_definition = cd:CAP
	}

	option = {
		name = sb_cape.103.a
		default_option = yes
		# CAP-side crisis event. This only opens the Albany question; the JE now
		# closes from London's answer 21 days later.
		set_variable = sb_albany_crisis_var
		if = {
			limit = {
				NOT = { has_law = law_type:law_cultural_exclusion }
			}
			activate_law = law_type:law_cultural_exclusion
		}
		if = {
			limit = {
				is_subject_of = c:GBR
			}
			c:GBR ?= {
				trigger_event = { id = sb_cape.200 popup = yes }
			}
		}
		else = {
			trigger_event = { id = sb_cape.202 days = 21 popup = yes }
		}
	}
}

# If the settlers dominate the Cape question without a liberal citizenship
# settlement in place, the Cape liberals should demand that the ministry turn
# principle into law instead of merely speaking in their name.
sb_cape.120 = {
	type = country_event
	placement = root

	title = sb_cape.120.t
	desc = sb_cape.120.d
	flavor = sb_cape.120.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_sufferage.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/principles_icons/creative_legislature.dds"
	duration = 3

	trigger = {
		country_definition = cd:CAP
		has_journal_entry = je_sb_cape_politics
	}

	option = {
		name = sb_cape.120.a
		default_option = yes
		random_law = {
			limit = {
				type = law_type:law_cultural_exclusion
			}
			save_scope_as = sb_cape_promised_law
		}
		if = {
			limit = {
				OR = {
					NOT = { has_journal_entry = je_government_petition_negotiation_version }
					NOT = { has_journal_entry = je_government_petition_negotiation_version_b }
				}
			}
			ig:ig_intelligentsia ?= {
				save_scope_as = ig
				set_variable = {
					name = desired_law_var
					value = scope:sb_cape_promised_law
				}
				set_variable = {
					name = promise_quest_degree
					value = 1
				}
				set_variable = {
					name = promise_quest_type
					value = 2
				}
			}
			promise_quest = yes
			if = {
				limit = { has_journal_entry = je_government_petition_negotiation_version }
				add_journal_entry = {
					type = je_government_petition_negotiation_version_b
				}
			}
			else = {
				add_journal_entry = {
					type = je_government_petition_negotiation_version
				}
			}
		}
		else = {
			add_modifier = {
				name = sb_cape_liberal_commitment
				months = 24
			}
		}
		if = {
			limit = {
				is_ai = yes
				NOT = { has_law = law_type:law_cultural_exclusion }
			}
			if = {
				limit = { enacting_any_law = yes }
				cancel_enactment = yes
			}
			start_enactment = law_type:law_cultural_exclusion
		}
		set_strategy = ai_strategy_progressive_agenda
		ai_chance = {
			base = 80
		}
	}

	option = {
		name = sb_cape.120.b
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_sb_cape_liberal
			}
			add_modifier = {
				name = sb_cape_liberal_refused
				months = 60
			}
		}
		ai_chance = {
			base = 20
		}
	}
}

# Albany Question — Britain can defuse a full ECSL secession crisis by hiving
# the Eastern Cape off into a separate colonial tag.
# =============================================================================
sb_cape.200 = {
	type = country_event
	placement = root

	title = sb_cape.200.t
	desc = sb_cape.200.d
	flavor = sb_cape.200.f

	event_image = {
		video = "gfx/event_pictures/europenorthamerica_political_extremism.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	duration = 3

	trigger = {
		country_definition = cd:GBR
		c:CAP ?= {
			has_variable = sb_albany_crisis_var
			NOT = { has_variable = sb_albany_sanctioned_var }
			NOT = { has_variable = sb_albany_refused_var }
		}
	}

	option = {
		name = sb_cape.200.a
		default_option = yes

		# London sanctions a separate Albany. If the secession tag does not exist
		# yet we create it peacefully out of the Eastern Cape; if it already exists
		# because the ECSL revolt has tipped over, we regularize that same tag
		# instead of creating a second country.
		if = {
			limit = {
				NOT = { exists = c:ABY }
			}
			random_country = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_EASTERN_CAPE
					}
				}
				save_scope_as = sb_albany_state_owner
			}
			scope:sb_albany_state_owner ?= {
				random_scope_state = {
					limit = {
						state_region = s:STATE_EASTERN_CAPE
					}
					save_scope_as = sb_albany_state_scope
				}
				create_country = {
					tag = ABY
					origin = c:CAP
					state = scope:sb_albany_state_scope
					on_created = {
						set_capital = STATE_EASTERN_CAPE
						activate_law = law_type:law_colonial_administration
						activate_law = law_type:law_appointed_bureaucrats
						activate_law = law_type:law_frontier_colonization
					}
				}
			}
		}

		if = {
			limit = {
				exists = c:ABY
			}
			c:ABY ?= {
				if = {
					limit = {
						NOT = { is_subject_of = c:GBR }
					}
					c:GBR ?= {
						create_diplomatic_pact = {
							country = c:ABY
							type = colony
						}
					}
				}
				hidden_effect = {
					activate_law = law_type:law_national_supremacy
					activate_law = law_type:law_national_militia
					# Give Albany an explicit frontier-settler cabinet before trying to
					# hand Robert Godlonton the reins. Without this, the country can keep
					# the auto-generated startup ruler instead of respecting the event handoff.
					every_interest_group = {
						remove_ruling_interest_group = yes
					}
					ig:ig_rural_folk ?= { add_ruling_interest_group = yes }
					ig:ig_landowners ?= { add_ruling_interest_group = yes }
					ig:ig_armed_forces ?= { add_ruling_interest_group = yes }
					ig:ig_devout ?= { add_ruling_interest_group = yes }
					active_law:lawgroup_army_model ?= {
						if = {
							limit = {
								NOT = { has_amendment = amendment_type:amendment_sb_commando_corps }
							}
							add_amendment = {
								type = amendment_sb_commando_corps
								sponsor = PREV.ig:ig_rural_folk
							}
						}
					}
					if = {
						limit = {
							NOT = { country_has_primary_culture = cu:boer }
						}
						add_primary_culture = cu:boer
					}
				
					change_relations = {
						country = c:CAP
						value = -40
					}
					change_tension = {
						country = c:XHR
						value = 100
					}
					change_tension = {
						country = c:XHG
						value = 100
					}
					change_tension = {
						country = c:XHT
						value = 100
					}
				}
			}

			s:STATE_EASTERN_CAPE = {
				remove_claim = c:CAP
				add_claim = c:ABY
				region_state:ABY = {
					set_state_type = incorporated
				}
			}

			c:CAP ?= {
				set_variable = sb_ecsl_resolved_var
			}
			c:CAP ?= {
				if = {
					limit = {
						any_scope_character = {
							has_variable = sb_robert_godlonton_var
						}
					}
					random_scope_character = {
						limit = {
							has_variable = sb_robert_godlonton_var
						}
						save_scope_as = sb_godlonton_scope
					}
				}
			}
			scope:sb_godlonton_scope ?= {
				# Mirror vanilla agitator promotion flow: move the character first,
				# convert him into a politician, then let the target country make
				# him the IG leader and ruler from its own scope.
				transfer_character = c:ABY
				add_character_role = politician
				if = {
					limit = {
						has_role = agitator
					}
					remove_character_role = agitator
				}
			}
			c:ABY ?= {
				scope:sb_godlonton_scope ?= {
					set_interest_group = c:ABY.ig:ig_rural_folk
					set_as_interest_group_leader = yes
					set_character_as_ruler = yes
				}
			}
			if = {
				limit = {
					NOT = { exists = scope:sb_godlonton_scope }
					NOT = {
						c:ABY ?= {
							any_scope_character = {
								has_variable = sb_robert_godlonton_var
							}
						}
					}
					}
					c:ABY ?= {
						create_character = {
							template = ABY_robert_godlonton
						}
					}
				}
		}

		c:CAP ?= {
			trigger_event = { id = sb_cape.201 days = 7 popup = yes }
		}

		ai_chance = {
			base = 80
		}
	}

	option = {
		name = sb_cape.200.b

		c:CAP ?= {
			trigger_event = { id = sb_cape.202 days = 21 popup = yes }
		}

		ai_chance = {
			base = 20
		}
	}
}
