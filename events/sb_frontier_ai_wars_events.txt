# Hidden dated AI frontier-war openers.
# Ownership:
#   - ORA/PHL/BST inland pacing wars
#   - ZUL follow-on pressure into Gaza
#   - Cape-Xhosa frontier war scheduling

sb_frontier_ai_wars.010 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:ORA
		is_ai = yes
		sb_frontier_ai_scripting_enabled = yes
	}

	immediate = {
		create_diplomatic_play = {
			type = dp_return_state
			target_country = c:PHL
			target_state = s:STATE_VRYSTAAT.region_state:PHL

			add_war_goal = {
				holder = root
				target_country = c:PHL
				type = humiliation
			}
		}
	}
}

sb_frontier_ai_wars.020 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:ORA
		is_ai = yes
		sb_frontier_ai_scripting_enabled = yes
	}

	immediate = {
		if = {
			limit = {
				sb_frontier_ai_behavior_dynamic_historical = yes
			}
			random_list = {
				20 = {
					c:BST ?= {
						add_modifier = {
							name = sb_bst_scripted_frontier_edge
							months = 36
						}
					}
				}
				80 = { }
			}
		}
		create_diplomatic_play = {
			type = dp_return_state
			target_country = c:BST
			target_state = s:STATE_VRYSTAAT.region_state:BST

			add_war_goal = {
				holder = c:BST
				target_state = s:STATE_VRYSTAAT.region_state:ORA
				type = return_state
			}
		}
		if = {
			limit = {
				c:TRN ?= {
					is_country_alive = yes
					is_ai = yes
				}
			}
			random_diplomatic_play = {
				limit = {
					initiator = root
					target = c:BST
					NOT = {
						any_scope_play_involved = {
							c:TRN ?= THIS
						}
					}
				}
				add_target_backers = { c:TRN }
			}
		}
	}
}
