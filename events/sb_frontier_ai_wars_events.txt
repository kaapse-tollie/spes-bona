namespace = sb_frontier_ai_wars

# Hidden dated AI frontier-war openers.
# Ownership:
#   - ORA/PHL/BST inland pacing wars
#   - ZUL follow-on pressure into Gaza
#   - Cape-Xhosa frontier war scheduling

sb_frontier_ai_wars.010 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:ORA
		is_ai = yes
		is_country_alive = yes
		is_subject = no
		is_at_war = no
		is_active_in_diplomatic_play = no
		aggressive_diplomatic_plays_permitted = yes
		sb_frontier_ai_scripting_enabled = yes
		c:PHL ?= {
			is_country_alive = yes
			is_ai = yes
			is_subject = no
			is_at_war = no
			is_active_in_diplomatic_play = no
			aggressive_diplomatic_plays_permitted = yes
		}
	}

	immediate = {
		set_variable = {
			name = sb_ora_phl_scripted_play_opening_var
			days = 30
		}
		create_diplomatic_play = {
			type = dp_return_state
			target_country = c:PHL
			target_state = s:STATE_VRYSTAAT.region_state:PHL

			add_war_goal = {
				holder = root
				target_country = c:PHL
				type = humiliation
			}
		}
		if = {
			limit = {
				c:TRN ?= {
					is_country_alive = yes
					is_ai = yes
				}
			}
			random_diplomatic_play = {
				limit = {
					is_diplomatic_play_type = dp_return_state
					initiator = { c:ORA ?= this }
					target = { c:PHL ?= this }
					NOT = {
						any_scope_play_involved = {
							c:TRN ?= THIS
						}
					}
				}
				add_initiator_backers = { c:TRN }
			}
		}
		trigger_event = { id = sb_frontier_ai_wars.011 days = 1 popup = no }
		trigger_event = { id = sb_frontier_ai_wars.011 days = 7 popup = no }
		set_variable = sb_ora_phl_scripted_war_resolved_var
	}
}

sb_frontier_ai_wars.100 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:XHT
		is_country_alive = yes
		c:CAP ?= {
			is_country_alive = yes
		}
	}

	immediate = {
		if = {
			limit = {
				game_date < 1848.1.1
				any_scope_state = {
					state_region = s:STATE_EASTERN_CAPE
					num_provinces >= 1
				}
				NOT = { sb_xht_owns_all_eastern_cape_starting_provinces = yes }
				c:CAP ?= {
					has_technology_researched = colonization
				}
			}
			create_diplomatic_play = {
				type = dp_sb_xhosa_war_7
				target_country = c:CAP
			}
			c:CAP ?= {
				set_variable = sb_xhosa_frontier_war_7_resolved_var
				if = {
					limit = { has_variable = sb_xhosa_frontier_war_7_scheduled_var }
					remove_variable = sb_xhosa_frontier_war_7_scheduled_var
				}
			}
		}
		else = {
			c:CAP ?= {
				if = {
					limit = { has_variable = sb_xhosa_frontier_war_7_scheduled_var }
					remove_variable = sb_xhosa_frontier_war_7_scheduled_var
				}
				if = {
					limit = { game_date >= 1848.1.1 }
					set_variable = sb_xhosa_frontier_war_7_resolved_var
				}
			}
		}
	}
}
