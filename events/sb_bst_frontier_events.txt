sb_bst_frontier.020 = {
	type = country_event
	placement = root

	title = sb_bst_frontier.020.t
	desc = sb_bst_frontier.020.d
	flavor = sb_bst_frontier.020.f

	event_image = {
		video = "gfx/event_pictures/africa_soldiers_breaking_protest.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"
	duration = 3

	trigger = {
		country_definition = cd:BST
		has_journal_entry = je_sb_bst_ora_frontier
		sb_bst_controls_only_drakensberg = yes
		NOT = { has_variable = sb_bst_into_the_mountains_var }
	}

	immediate = {
		set_variable = sb_bst_into_the_mountains_var
	}

	option = {
		name = sb_bst_frontier.020.a
		default_option = yes
		c:ORA ?= {
			if = {
				limit = { is_country_alive = yes }
				change_relations = {
					country = root
					value = -20
				}
				s:STATE_DRAKENSBERG = {
					add_claim = c:ORA
				}
			}
		}
		add_modifier = {
			name = sb_bst_biding_time
			months = 60
		}
	}
}

sb_bst_frontier.100 = {
	type = country_event
	placement = root

	title = sb_bst_frontier.100.t
	desc = sb_bst_frontier.100.d
	flavor = sb_bst_frontier.100.f

	event_image = {
		video = "gfx/event_pictures/africa_construction_colony.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:GBR
		c:BST ?= {
			is_country_alive = yes
			is_subject = no
			has_journal_entry = je_sb_bst_ora_frontier
			sb_has_adjacent_british_protection_route = yes
		}
	}

	option = {
		name = sb_bst_frontier.100.a
		default_option = yes
		c:GBR ?= {
			create_diplomatic_pact = {
				country = c:BST
				type = protectorate
			}
		}
		c:ORA ?= {
			if = {
				limit = { is_country_alive = yes }
				trigger_event = { id = sb_bst_frontier.110 days = 1 popup = yes }
			}
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_bst_frontier.100.b
		trigger = {
			is_player = yes
		}
		ai_chance = {
			base = 0
		}
	}
}

sb_bst_frontier.200 = {
	type = country_event
	placement = root

	title = sb_bst_frontier.200.t
	desc = sb_bst_frontier.200.d
	flavor = sb_bst_frontier.200.f

	event_image = {
		video = "gfx/event_pictures/africa_construction_colony.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:GBR
		c:BST ?= {
			has_variable = sb_bst_cap_transfer_pending_var
			is_country_alive = yes
			is_subject_of = c:GBR
			is_subject_type = subject_type_protectorate
		}
		c:CAP ?= {
			is_country_alive = yes
			is_subject_of = c:GBR
			OR = {
				is_subject_type = subject_type_dominion
				is_subject_type = subject_type_sb_dominion
				is_subject_type = subject_type_sb_responsible_colony
				is_subject_type = subject_type_sb_responsible_colony_monarchy
			}
			sb_cap_controls_bst_frontier = yes
		}
	}

	immediate = {
		c:BST ?= {
			remove_variable = sb_bst_cap_transfer_pending_var
		}
	}

	option = {
		name = sb_bst_frontier.200.a
		default_option = yes
		c:BST ?= {
			set_variable = sb_bst_cap_transfer_resolved_var
		}
		c:CAP ?= {
			transfer_subject = c:BST
		}
		c:BST ?= {
			if = {
				limit = {
					is_subject_of = c:CAP
					NOT = { is_subject_type = subject_type_puppet }
				}
				change_subject_type = subject_type_puppet
			}
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_bst_frontier.200.b
		trigger = {
			is_player = yes
		}
		c:BST ?= {
			set_variable = sb_bst_cap_transfer_resolved_var
		}
		ai_chance = {
			base = 0
		}
	}
}
