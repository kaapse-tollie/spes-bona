			limit = { exists = c:RHB }

				every_scope_pop = {
					limit = {

					}

			}
		}
	}

		name = sb_nam.020.b

		add_modifier = {
			name = sb_nam_rehoboth_discouraged
			months = 60
		}
		every_political_movement = {
			limit = {
				is_political_movement_type = movement_sb_cape_liberal

			add_modifier = {
