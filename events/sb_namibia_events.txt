sb_nam.010 = {
	type = country_event
	placement = root

	title = sb_nam.010.t
	desc = sb_nam.010.d
	flavor = sb_nam.010.f

	event_image = {
		video = "gfx/event_pictures/africa_diplomats_negotiating.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	duration = 3

	trigger = {
		country_definition = cd:GER
		game_date >= 1869.11.11
		game_date <= 1875.12.31
		NOT = { has_variable = sb_nam_colonial_prestige_resolved_var }
		NOT = { has_law = law_type:law_frontier_colonization }
		NOT = { has_law = law_type:law_colonial_resettlement }
		NOT = { has_law = law_type:law_colonial_exploitation }
		OR = {
			ig:ig_landowners   ?= { is_in_government = yes }
			ig:ig_armed_forces ?= { is_in_government = yes }
		}
		OR = {
			NOT = { has_journal_entry = je_government_petition_negotiation_version }
			NOT = { has_journal_entry = je_government_petition_negotiation_version_b }
		}
	}

	option = {
		name = sb_nam.010.a
		default_option = yes
		custom_tooltip = sb_nam_010_a_tt
		hidden_effect = {
			set_variable = sb_nam_colonial_prestige_resolved_var
			set_variable = sb_nam_angra_pequena_purchase_pending_var
			random_law = {
				limit = {
					type = law_type:law_colonial_exploitation
				}
				save_scope_as = sb_nam_promised_law
			}
			if = {
				limit = {
					ig:ig_landowners ?= { is_in_government = yes }
				}
				ig:ig_landowners ?= {
					save_scope_as = ig
					set_variable = {
						name = desired_law_var
						value = scope:sb_nam_promised_law
					}
					set_variable = {
						name = promise_quest_degree
						value = 1
					}
					set_variable = {
						name = promise_quest_type
						value = 2
					}
				}
			}
			else = {
				ig:ig_armed_forces ?= {
					save_scope_as = ig
					set_variable = {
						name = desired_law_var
						value = scope:sb_nam_promised_law
					}
					set_variable = {
						name = promise_quest_degree
						value = 1
					}
					set_variable = {
						name = promise_quest_type
						value = 2
					}
				}
			}
			promise_quest = yes
			if = {
				limit = { has_journal_entry = je_government_petition_negotiation_version }
				add_journal_entry = {
					type = je_government_petition_negotiation_version_b
				}
			}
			else = {
				add_journal_entry = {
					type = je_government_petition_negotiation_version
				}
			}
			if = {
				limit = {
					is_ai = yes
					NOT = { has_law = law_type:law_colonial_exploitation }
				}
				if = {
					limit = { enacting_any_law = yes }
					cancel_enactment = yes
				}
				start_enactment = law_type:law_colonial_exploitation
			}
			sb_nam_schedule_angra_pequena_purchase = yes
		}
	}

	option = {
		name = sb_nam.010.b
		custom_tooltip = sb_nam_010_b_tt
		hidden_effect = {
			set_variable = sb_nam_colonial_prestige_resolved_var
			if = {
				limit = {
					ig:ig_landowners ?= { is_in_government = yes }
				}
				ig:ig_landowners ?= {
					add_modifier = {
						name = sb_nam_german_colonial_rebuff
						months = 60
					}
				}
			}
			else = {
				ig:ig_armed_forces ?= {
					add_modifier = {
						name = sb_nam_german_colonial_rebuff
						months = 60
					}
				}
			}
		}
		ai_chance = { base = -1 }
	}
}

sb_nam.011 = {
	type = country_event
	placement = root

	title = sb_nam.011.t
	desc = sb_nam.011.d
	flavor = sb_nam.011.f

	event_image = {
		video = "gfx/event_pictures/africa_construction_colony.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_trade.dds"
	duration = 3

	trigger = {
		country_definition = cd:GER
		game_date >= 1883.1.1
		NOT = { has_variable = sb_nam_angra_pequena_purchased_var }
		NOT = { has_variable = sb_nam_angra_pequena_purchase_declined_var }
		OR = {
			has_law = law_type:law_colonial_resettlement
			has_law = law_type:law_colonial_exploitation
		}
		NOT = {
			any_scope_state = {
				state_region = s:STATE_NAMAQUALAND
				owner = root
				num_provinces >= 1
			}
		}
		p:xDEAA3C.state.owner = {
			is_country_type = decentralized
		}
		c:CAP ?= {
			NOT = {
				any_scope_state = {
					state_region = s:STATE_NAMAQUALAND
					owner = prev
					num_provinces >= 2
				}
			}
		}
	}

	option = {
		name = sb_nam.011.a
		default_option = yes
		custom_tooltip = sb_nam_011_a_tt
		hidden_effect = {
			add_treasury = -100000
			set_variable = sb_nam_angra_pequena_purchased_var
			if = {
				limit = { has_variable = sb_nam_angra_pequena_purchase_pending_var }
				remove_variable = sb_nam_angra_pequena_purchase_pending_var
			}
			s:STATE_NAMAQUALAND = {
				set_owner_of_provinces = {
					country = root
					provinces = { xDEAA3C }
				}
				add_claim = root
			}
			if = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_NAMAQUALAND
						NOT = { has_building = building_port }
					}
				}
				random_scope_state = {
					limit = {
						state_region = s:STATE_NAMAQUALAND
						NOT = { has_building = building_port }
					}
					create_building = {
						building = building_port
						level = 1
						reserves = 1
					}
				}
			}
			if = {
				limit = {
					NOT = { has_technology_researched = malaria_prevention }
					can_research = malaria_prevention
				}
				add_technology_progress = { progress = 7500 technology = malaria_prevention }
			}
		}
		ai_chance = { base = 1 }
	}

	option = {
		name = sb_nam.011.b
		custom_tooltip = sb_nam_011_b_tt
		hidden_effect = {
			set_variable = sb_nam_angra_pequena_purchase_declined_var
			if = {
				limit = { has_variable = sb_nam_angra_pequena_purchase_pending_var }
				remove_variable = sb_nam_angra_pequena_purchase_pending_var
			}
		}
		ai_chance = { base = 0 }
	}
}
