namespace = sb_boer_zulu

# =============================================================================
# Spes Bona - Oranje / Natal / Zulu Event Chain
# =============================================================================
# Current V2 structure:
#   - ORA is the only inland Boer participant in the Natal / Blood River lane.
#   - TRN is delayed until after Blood River and the west-Transvaal foothold.
#   - Natalia can still become the player branch after Boer victory.
#   - The later Zulu dynastic / Swazi follow-up now lives in
#     `events/sb_zulu_dynasty_events.txt` so this file stays focused on the
#     Natal crisis, the British ultimatum, and Boer coastal fallout.
#
# Chain summary:
#   010 ORA launches the Natal expedition
#   020 ZUL answers the Retief approach
#   021 Dingane faces unrest after conceding the southern strip
#   030 peaceful land grant -> independent Natalia formed
#   050/051 refusal and murder -> ORA backs down or starts Blood River
#   070 ORA victory in the Natal crisis -> Natalia formed after Boer victory
#   080 ZUL victory in the Natal crisis -> unlocks the Port Natal raid decision
#   085 GBR reaction to a Durban raid
#   100 GBR decides whether to issue the Natal ultimatum
#   110 NAL answers the ultimatum
#   111 ORA decides whether to support resisting Natalia
#   112 TRN decides whether to support resisting Natalia
#   114 GBR launches the protectorate war after resistance
#   115 NAL is reduced to a British crown colony after submission or defeat
#   116/117 ORA/TRN receive refugee fallout from Natalia's collapse
#
# Dormant compatibility branch:
#   056 / 057 / 071 remain defined for older TRN-assisted Blood River saves,
#   but the live V2 setup no longer routes new games into them.
# =============================================================================

sb_boer_zulu.010 = {
	type = country_event
	placement = root

	title = sb_boer_zulu.010.t
	desc = sb_boer_zulu.010.d
	flavor = sb_boer_zulu.010.f

	event_image = {
		video = "gfx/event_pictures/africa_diplomats_negotiating.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:ORA
		has_journal_entry = je_sb_great_trek
		has_variable = sb_natal_diplomacy_started_var
		NOT = { has_variable = sb_natal_path_resolved_var }
		NOT = { exists = c:NAL }
		c:ZUL ?= {
			is_country_alive = yes
		}
	}

	immediate = {
		c:ZUL ?= {
			save_scope_as = zulu_kingdom
		}
	}

	option = {
		name = sb_boer_zulu.010.a
		default_option = yes
		custom_tooltip = sb_boer_zulu_010_a_warning_tt
		# Launching the Retief embassy also secures the southeast trek corridor
		# ahead of the Natal split, so ORA keeps those frontier districts even
		# before the later Winburg/TSW post-MTB settlement is finalized.
		s:STATE_VRYSTAAT = {
			set_owner_of_provinces = {
				country = c:ORA
				provinces = { x6253E3 x19AE59 }
			}
		}
		set_variable = sb_retief_petition_var
		scope:zulu_kingdom = {
			trigger_event = { id = sb_boer_zulu.020 days = 21 }
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_boer_zulu.010.b
		set_variable = sb_natal_path_resolved_var
		remove_variable = sb_natal_diplomacy_started_var
		add_modifier = {
			name = sb_prestige_loss_small
			months = 120
		}
		ai_chance = {
			base = 0
		}
	}
}

sb_boer_zulu.019 = {
	type = country_event
	hidden = yes

	trigger = {
		country_definition = cd:ZUL
	}

	immediate = {
		# One-shot AI balance resolver for the Blood River diplomatic play.
		# It runs from the DP start hook, not from war start, so the support / no-
		# support branch is locked in before the fronts open.
		if = {
			limit = {
				NOT = { is_player = yes }
				NOT = { has_variable = sb_zul_blood_river_support_resolved_var }
			}
			set_variable = sb_zul_blood_river_support_resolved_var
			if = {
				limit = {
					OR = {
						c:ORA ?= {
							is_country_alive = yes
							is_player = yes
						}
						c:TRN ?= {
							is_country_alive = yes
							is_player = yes
						}
					}
				}
				if = {
					limit = {
						NOT = { has_variable = sb_zul_emergency_barracks_var }
					}
					sb_raise_zul_blood_river_relief_force = yes
				}
			}
			else = {
				random_list = {
					33 = {
						if = {
							limit = {
								NOT = { has_variable = sb_zul_emergency_barracks_var }
							}
							sb_raise_zul_blood_river_relief_force = yes
						}
					}
					67 = {
						add_modifier = {
							name = sb_blood_river_zulu_no_relief
							months = 12
						}
					}
				}
			}
		}
	}
}

	}

			}

					NOT = { has_trait = cruel }
				}
				add_trait = cruel
			}
		}

			set_owner_of_provinces = {
				country = c:ZUL

			}
		}

			trigger_event = { id = sb_boer_zulu.050 days = 21 }
		}

			base = 70
		}

sb_boer_zulu.056 = {
	type = country_event
	placement = root

	title = sb_boer_zulu.056.t
	desc = sb_boer_zulu.056.d
	flavor = sb_boer_zulu.056.f

	event_image = {
		video = "gfx/event_pictures/africa_diplomats_negotiating.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:TRN
	}

	immediate = {
		c:ORA ?= {
			save_scope_as = orangia
		}
		c:ZUL ?= {
			save_scope_as = zulu_kingdom
		}
	}

	# Join the British protectorate war on Natalia's side.
	# Legacy compatibility: this used to be the TRN joins ORA against ZUL branch
	# in the pre-V2 Blood River setup. It is kept defined for older saves, but
	# fresh campaigns should no longer trigger it.
	option = {
		name = sb_boer_zulu.056.a
		default_option = yes
		if = {
			limit = {
				NOT = {
					any_scope_treaty = {
						binds = scope:orangia
						any_scope_article = {
							OR = {
								has_type = defensive_pact
								has_type = alliance
							}
						}
					}
				}
			}
			create_treaty = {
				first_country = root
				second_country = scope:orangia
				is_draft = no
				binding_period = { years = 5 }
				articles_to_create = {
					{
						article = defensive_pact
					}
				}
			}
		}
		set_variable = sb_trn_joined_natal_war_var
		scope:orangia ?= {
			set_variable = sb_trn_joined_natal_war_var
			change_relations = {
				country = root
				value = 20
			}
		}
		scope:zulu_kingdom = {
			create_diplomatic_play = {
				type = dp_sb_natal_crisis
				target_country = scope:orangia
				add_war_goal = {
					holder = scope:orangia
					type = humiliation
					target_country = scope:zulu_kingdom
				}
			}
		}
		random_diplomatic_play = {
			limit = {
				initiator = scope:zulu_kingdom
				target = scope:orangia
				is_diplomatic_play_type = dp_sb_natal_crisis
			}
			add_target_backers = { root }
		}
		ai_chance = {
			base = 100
			modifier = {
				trigger = {
					OR = {
						scope:orangia ?= { is_ai = no }
						is_ai = no
					}
				}
				# Player support is already handled explicitly through option access,
				# so this modifier remains neutral for AI weighting.
				add = 0
			}
		}
	}

	# Decline to intervene.
	option = {
		name = sb_boer_zulu.056.b
		every_scope_treaty = {
			limit = {
				binds = scope:orangia
				any_scope_article = {
					has_type = defensive_pact
				}
			}
			withdraw = {
				country = root
			}
		}
		change_relations = {
			country = scope:orangia
			value = -20
		}
		scope:zulu_kingdom = {
			create_diplomatic_play = {
				type = dp_sb_natal_crisis
				target_country = scope:orangia
				add_war_goal = {
					holder = scope:orangia
					type = humiliation
					target_country = scope:zulu_kingdom
				}
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
