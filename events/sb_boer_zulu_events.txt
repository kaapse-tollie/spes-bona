namespace = sb_boer_zulu

# =============================================================================
# Spes Bona - Oranje / Natal / Zulu Event Chain
# =============================================================================
# Phase 1 design choice:
#   - ORA is the primary Natal actor.
#   - TRN owns the inland/Ndebele/Transvaal lane.
#   - The southern Natal strip begins under a small decentralized Qwabe polity,
#     reflecting the unstable frontier south of the Tugela without handing the
#     whole coast to either ZUL or ORA on day one.
#
# Chain summary:
#   010 ORA launches the Natal expedition
#   020 ZUL answers the Retief approach
#   021 Dingane is deposed after conceding the southern strip
#   030 peaceful land grant -> independent Natalia formed
#   050/056 refusal and murder -> ORA may ask TRN for help before war
#   070 ORA victory in the Natal crisis -> Natalia formed after Boer victory
#   071 TRN player follow-up after Blood River
#   080 ZUL victory in the Natal crisis -> optional Durban raid
#   085 GBR reaction to a Durban raid
#   090 Dingane deposed after defeat by the Boers
#   091 Dingane deposed after defeat by Britain
#   092 Dingane survives a British intervention
#   100 GBR decides whether to issue the Natal ultimatum
#   110 NAL answers the ultimatum
#   111 ORA decides whether to support resisting Natalia
#   112 TRN decides whether to support resisting Natalia
#   114 GBR launches the protectorate war after resistance
#   115 ORA receives refugee/migration fallout if Natalia submits
#   116/117 ORA/TRN receive refugee fallout if Natalia loses the war
# =============================================================================

sb_boer_zulu.010 = {
	type = country_event
	placement = root

	title = sb_boer_zulu.010.t
	desc = sb_boer_zulu.010.d
	flavor = sb_boer_zulu.010.f

	event_image = {
		video = "gfx/event_pictures/africa_diplomats_negotiating.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:ORA
		has_journal_entry = je_sb_great_trek
		has_variable = sb_natal_diplomacy_started_var
		NOT = { has_variable = sb_natal_path_resolved_var }
		NOT = { exists = c:NAL }
		c:ZUL ?= {
			is_country_alive = yes
		}
	}

	immediate = {
		c:ZUL ?= {
			save_scope_as = zulu_kingdom
		}
	}

	option = {
		name = sb_boer_zulu.010.a
		default_option = yes
		custom_tooltip = sb_boer_zulu_010_a_warning_tt
		set_variable = sb_retief_petition_var
		scope:zulu_kingdom = {
			trigger_event = { id = sb_boer_zulu.020 days = 21 }
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = sb_boer_zulu.010.b
		set_variable = sb_natal_path_resolved_var
		remove_variable = sb_natal_diplomacy_started_var
		add_modifier = {
			name = sb_prestige_loss_small
			months = 120
		}
		ai_chance = {
			base = 0

	}

sb_boer_zulu.056 = {
	type = country_event
	placement = root

	title = sb_boer_zulu.056.t
	desc = sb_boer_zulu.056.d
	flavor = sb_boer_zulu.056.f

	event_image = {
		video = "gfx/event_pictures/africa_diplomats_negotiating.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		country_definition = cd:TRN
	}

	immediate = {
		c:ORA ?= {
			save_scope_as = orangia
		}
		c:ZUL ?= {
			save_scope_as = zulu_kingdom
		}
	}

	option = {
		name = sb_boer_zulu.056.a
		default_option = yes
		if = {
			limit = {
				NOT = {
					any_scope_treaty = {
						binds = scope:orangia
						any_scope_article = {
							OR = {
								has_type = defensive_pact
								has_type = alliance
							}
						}
					}
				}
			}
			create_treaty = {
				first_country = root
				second_country = scope:orangia
				is_draft = no
				binding_period = { years = 5 }
				articles_to_create = {
					{
						article = defensive_pact
					}
				}
			}
		}
		set_variable = sb_trn_joined_natal_war_var
		scope:orangia ?= {
			set_variable = sb_trn_joined_natal_war_var
			change_relations = {
				country = root
				value = 20
			}
		}
		scope:zulu_kingdom = {
			create_diplomatic_play = {
				type = dp_sb_natal_crisis
				target_country = scope:orangia
				add_war_goal = {
					holder = scope:orangia
					type = humiliation
					target_country = scope:zulu_kingdom
				}
			}
		}
		ai_chance = {
			base = 100
			modifier = {
				trigger = {
					OR = {
						scope:orangia ?= { is_ai = no }
						is_ai = no
					}
				} 
				add = 0 # ? 
			}
		}
	}

	option = {
		name = sb_boer_zulu.056.b
		every_scope_treaty = {
			limit = {
				binds = scope:orangia
				any_scope_article = {
					has_type = defensive_pact
				}
			}
			withdraw = {
				country = scope:orangia
			}
		}
		change_relations = {
			country = scope:orangia
			value = -20
		}
		scope:zulu_kingdom = {
			create_diplomatic_play = {
				type = dp_sb_natal_crisis
				target_country = scope:orangia
				add_war_goal = {
					holder = scope:orangia
					type = humiliation
					target_country = scope:zulu_kingdom
				}
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
