	icon = "gfx/interface/icons/event_icons/event_default.dds"
	duration = 3

	trigger = {
		sb_martinus_ora_can_receive_cape_alignment = yes
	}

	option = {
		name = sb_martinus_confederation.001.a
		default_option = yes
		set_variable = sb_martinus_cape_alignment_resolved_var
		set_variable = sb_martinus_blocked_by_cape_alignment_var

			limit = { has_variable = sb_martinus_cape_alignment_pending_var }
			remove_variable = sb_martinus_cape_alignment_pending_var
		}
		c:CAP ?= {
			create_diplomatic_pact = {
				country = root
				type = sb_cape_confederal_dependency
			}
			change_relations = {
				country = root
				value = 25
			}
		}
		c:GBR ?= {
			change_relations = {
				country = root
				value = 20
			}
		}
		ai_chance = {
			base = 55
			modifier = {
				trigger = {
					c:GBR ?= {
						relations:root >= relations_threshold:amicable
					}
				}
				add = 25
			}
			modifier = {
				trigger = {
					has_variable = sb_bloemfontein_convention_accepted_var
				}
				add = 15
			}
		}
	}

	option = {
		name = sb_martinus_confederation.001.b
		set_variable = sb_martinus_cape_alignment_resolved_var
		if = {
			limit = { has_variable = sb_martinus_cape_alignment_pending_var }
			remove_variable = sb_martinus_cape_alignment_pending_var
		}
		add_modifier = {
			name = sb_ora_republican_defiance
			months = 60
		}
		change_relations = {
			country = c:CAP
			value = -15
		}
		change_relations = {
			country = c:GBR
			value = -15
		}
		ai_chance = {
			base = 45
			modifier = {
				trigger = {
					ig:ig_armed_forces ?= { is_in_government = yes }
				}
				add = 20
			}
		}
	}
}
