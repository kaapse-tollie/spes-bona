je_sb_zulu_enforce_dynastic_claims_button = {
	name = "je_sb_zulu_enforce_dynastic_claims_button"
	desc = "je_sb_zulu_enforce_dynastic_claims_button_desc"

	visible = {
		has_journal_entry = je_sb_zulu_kingdom
		NOT = { has_variable = sb_zulu_enforce_dynastic_claims_var }
	}

	possible = {
		custom_tooltip = {
			text = je_sb_zulu_enforce_dynastic_claims_button_tt
			authority >= 75
		}
	}

	effect = {
		sb_zulu_enable_dynastic_claims = yes
	}

	ai_chance = {
		if ={
			limit = {
				authority >= 75
			}
			value = 1000
		}
		value = 20
	}
}
