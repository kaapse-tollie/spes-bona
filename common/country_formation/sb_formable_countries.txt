# Spes Bona - Formable Countries

# Union of South Africa — requires controlling most of southern Africa
SAF = {
	states = {
		STATE_CAPE_COLONY
		STATE_EASTERN_CAPE
		STATE_NORTHERN_CAPE
		STATE_VRYSTAAT
		STATE_ZULULAND
		STATE_TRANSVAAL
		STATE_EAST_TRANSVAAL
		STATE_NORTHERN_TRANSVAAL
	}

	required_states_fraction = 0.70

	ai_will_do = { always = yes }

	possible = {
		has_technology_researched = nationalism
		country_rank >= rank_value:minor_power
		OR = {
			country_has_primary_culture = cu:british
			country_has_primary_culture = cu:boer
		}
	}
}

NGU = {
	states = {
		STATE_EASTERN_CAPE
		STATE_ZULULAND
		STATE_EAST_TRANSVAAL
		STATE_LOURENCO_MARQUES
	}

	# Keep the native formation UI on a 3-of-4 state-region threshold; the
	# supplemental possible trigger below enforces the 80% province share.
	required_states_fraction = 0.75

	ai_will_do = { always = yes }

	possible = {
		is_subject = no
		NOT = { is_country_type = decentralized }
		sb_nguni_controls_three_core_state_province_blocks = yes
		OR = {
			country_has_primary_culture = cu:zulu
			country_has_primary_culture = cu:nguni
		}
	}
}
