# Spes Bona - Formable Countries

# Union of South Africa — requires controlling most of southern Africa
SAF = {
	states = {
		STATE_CAPE_COLONY
		STATE_EASTERN_CAPE
		STATE_NORTHERN_CAPE
		STATE_VRYSTAAT
		STATE_DRAKENSBERG
		STATE_ZULULAND
		STATE_TRANSVAAL
		STATE_EAST_TRANSVAAL
		STATE_NORTHERN_TRANSVAAL
	}

	required_states_fraction = 0.70

	ai_will_do = { always = yes }

	possible = {
		has_technology_researched = nationalism
		country_rank >= rank_value:minor_power
		OR = {
			country_has_primary_culture = cu:british
			country_has_primary_culture = cu:anglo_african
			country_has_primary_culture = cu:boer
		}
	}
}

NGU = {
	states = {
		STATE_ZULULAND
		STATE_EAST_TRANSVAAL
		STATE_LOURENCO_MARQUES
	}

	# The real gate is handled by sb_nguni_conqueror_path_open: any foothold
	# in the eastern Nguni sphere plus the other founder tags being dead.
	required_states_fraction = 0.33

	ai_will_do = { always = yes }

	possible = {
		is_subject = no
		NOT = { is_country_type = decentralized }
		sb_nguni_conqueror_path_open = yes
		OR = {
			country_has_primary_culture = cu:zulu
			country_has_primary_culture = cu:nguni
		}
	}
}

STA = {
	states = {
		STATE_VRYSTAAT
		STATE_DRAKENSBERG
	}

	required_states_fraction = 1.0

	ai_will_do = { always = yes }

	possible = {
		country_definition = cd:BST
		is_subject = no
		has_variable = sb_sotho_tswana_nation_completed_var
		c:TSW ?= {
			is_country_alive = yes
			is_subject_of = root
			is_subject_type = subject_type_vassal
			liberty_desire <= 25
		}
	}
}
