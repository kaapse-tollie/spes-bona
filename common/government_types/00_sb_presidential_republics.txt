# Boer republic override for the presidential-republic family.
#
# Keep this as a custom Boer republic, but use the command-friendly
# dictatorial transfer path internally so the engine consistently treats the
# Commandant-General as eligible for field command. IG leadership is restored
# in runtime scripted effects to avoid brittle startup validation conflicts.

gov_boer_commandant_general_republic = {
	transfer_of_power = dictatorial
	new_leader_on_reform_government = no

	male_ruler = "RULER_TITLE_COMMANDANT_GENERAL"
	female_ruler = "RULER_TITLE_COMMANDANT_GENERAL"

	possible = {
		has_law = law_type:law_sb_commandant_general_republic
		country_has_primary_culture = cu:boer
	}

	on_government_type_change = {
		change_to_dictatorial = yes
	}

	on_post_government_type_change = {
		post_change_to_dictatorial = yes
	}
}
