# Settler Separatist — Ideology for the Eastern Cape Separatist League and Boer frontier leaders.
# Represents Afrikaner/British settler politics: republican governance, restricted franchise,
# racial hierarchy, aggressive frontier expansion, and citizen-militia defense.
# Historically reflects the mindset of Boer Voortrekker leaders (Pretorius, Potgieter) and
# British settlers in the Eastern Cape who wanted separation from the liberal Cape Town government.
ideology_sb_settler_separatist = {
	icon = "gfx/interface/icons/ideology_icons/frontier_expansionist.dds"

	# Governance: Eastern Cape separatists ultimately preferred staying within a
	# Crown framework if London would sponsor a harder frontier colony of their own.
	lawgroup_governance_principles = {
		law_presidential_republic = neutral
		law_parliamentary_republic = neutral
		law_monarchy  = approve
	}

	# Voting: Landed/wealth franchise reflects the Boer tradition of white male landholder suffrage.
	# Universal suffrage strongly opposed — settlers feared being outvoted by indigenous majorities.
	lawgroup_distribution_of_power = {
		law_landed_voting = strongly_approve
		law_wealth_voting = approve
		law_census_voting = neutral
		law_universal_suffrage = strongly_disapprove
	}

	# Citizenship: National supremacy = white settler dominance over indigenous populations.
	# Multicultural strongly disapproved — antithetical to the separatist worldview.
	lawgroup_citizenship = {
		law_ethnostate = strongly_approve
		law_national_supremacy = approve
		law_racial_segregation = neutral
		law_cultural_exclusion = disapprove
		law_multicultural = strongly_disapprove
	}

	# Colonization: Frontier expansion was central to settler identity (the Great Trek, etc.).
	# No colonial affairs = giving up expansion entirely, anathema to settlers.
	lawgroup_colonization = {
		law_frontier_colonization = strongly_approve
		law_colonial_resettlement = approve
		law_colonial_exploitation = neutral
		law_no_colonial_affairs = strongly_disapprove
	}

	# Land: Settler Farmers & Boers strongly attached to the land 
	lawgroup_land_reform = {
		law_serfdom = strongly_disapprove
		law_tenant_farmers = strongly_disapprove
		law_commercialized_agriculture = neutral
		law_collectivized_agriculture = disapprove
		law_homesteading = strongly_approve
	}	

	# Army: National militia reflects the Boer commando system (citizen-soldiers, no standing army).
	# Peasant levies disapproved — settlers saw themselves as free citizens, not feudal subjects.
	lawgroup_army_model = {
		law_national_militia = strongly_approve
		law_peasant_levies = strongly_disapprove
	}
}

# Cape Liberal — Ideology representing the progressive Cape liberal tradition.
# Historically embodied by figures like John Fairbairn, Saul Solomon, and later John X. Merriman.
# The Cape Colony was unusual in southern Africa for its non-racial qualified franchise (from 1853),
# which granted voting rights based on property ownership regardless of race.
# This ideology champions parliamentary democracy, press freedom, education, and racial equality.
ideology_sb_cape_liberal = {
	icon = "gfx/interface/icons/ideology_icons/liberal.dds"

	# Governance: Parliamentary republic preferred — reflects the Cape's strong parliamentary tradition.
	# The Cape had one of the most active colonial parliaments in the British Empire.
	lawgroup_governance_principles = {
		law_parliamentary_republic = strongly_approve
		law_presidential_republic = approve
		law_monarchy  = disapprove

	}

	# Voting: Census voting (property-based, race-blind) mirrors the historical Cape Qualified Franchise.
	# The 1853 Cape franchise set a low property threshold specifically to include non-white voters.
	# Landed voting neutral because it concentrates power with large estate holders (settlers).
	# Cross-ref: common/amendments/sb_amendments.txt (amendment_sb_cape_qualified_franchise)
	lawgroup_distribution_of_power = {
		law_census_voting = strongly_approve
		law_wealth_voting = strongly_approve
		law_universal_suffrage = approve
		law_landed_voting = neutral
		law_oligarchy = strongly_disapprove
		law_autocracy = strongly_disapprove
	}

	# Citizenship: Multicultural strongly approved — the core Cape liberal principle.
	# Cape liberals explicitly opposed racial barriers to citizenship and political participation.
	# National supremacy and ethnostate = anathema to the Cape liberal tradition.
	lawgroup_citizenship = {
		law_multicultural = strongly_approve
		law_cultural_exclusion = approve
		law_subjecthood = neutral
		law_racial_segregation = disapprove
		law_national_supremacy = strongly_disapprove
		law_ethnostate = strongly_disapprove
	}

	# Colonization: Cape liberals were generally anti-expansionist, preferring stable borders
	# and integration over costly frontier wars. This contrasts with settler separatists.
	lawgroup_colonization = {
		law_no_colonial_affairs = strongly_approve
		law_colonial_resettlement = disapprove
		law_frontier_colonization = disapprove
		law_colonial_exploitation = strongly_disapprove
	}

	# Free speech: Freedom of the press was a key Cape liberal cause.
	# John Fairbairn's fight for press freedom (1820s-1830s) was a landmark in Cape politics.
	lawgroup_free_speech = {
		law_protected_speech = strongly_approve
		law_right_of_assembly = approve
		law_censorship = disapprove
		law_outlawed_dissent = strongly_disapprove
	}

	# Education: Public schools strongly approved — Cape liberals championed education for all races.
	# Missionary schools were a key vehicle for non-racial education in the historical Cape.
	lawgroup_education_system = {
		law_public_schools = strongly_approve
		# Follow-up design note: a CAP-specific "Anglican Missionary" clergy split
		# is still a plausible later refinement if the Devout bloc feels too
		# generically conservative in playtests.
		law_religious_schools = strongly_approve
		law_private_schools = approve
		law_no_schools = strongly_disapprove
	}
}
