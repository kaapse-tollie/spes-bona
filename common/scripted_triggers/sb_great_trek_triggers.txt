# Shared trek / Transvaal ownership triggers.
#
# These triggers sit between the audited three-state Transvaal split and the
# higher-level JE/event logic so the content can ask for "west Transvaal core"
# or "Transvaal except the Swazi carve-out" without repeating province lists.

sb_can_use_frontier_colonization = {
	OR = {
		AND = {
			country_is_colonial_or_company = yes
			any_decentralized_country = {
				is_adjacent_to_country = root
			}
		}
		any_scope_state = {
			is_homeland_of_country_cultures = root
			any_neighbouring_state = {
				owner = {
					is_country_type = decentralized
				}
			}
		}
		AND = {
			country_has_primary_culture = cu:boer
			OR = {
				country_definition = cd:ORA
				country_definition = cd:TRN
				country_definition = cd:ZPB
				country_definition = cd:LYD
				country_definition = cd:NAL
			}
		}
	}
}

sb_controls_west_transvaal_core = {
	owns_entire_state_region = STATE_TRANSVAAL
}

sb_has_west_transvaal_foothold = {
	OR = {
		p:xD0A010.state.owner = this
		p:x8BDC4D.state.owner = this
		p:xA22DC2.state.owner = this
		p:xD02190.state.owner = this
		p:x6C1786.state.owner = this
		p:x9A137B.state.owner = this
	}
}

sb_any_boer_republic_player = {
	OR = {
		c:ORA ?= {
			is_country_alive = yes
			is_player = yes
		}
		c:TRN ?= {
			is_country_alive = yes
			is_player = yes
		}
		c:ZPB ?= {
			is_country_alive = yes
			is_player = yes
		}
		c:LYD ?= {
			is_country_alive = yes
			is_player = yes
		}
		c:NAL ?= {
			is_country_alive = yes
			is_player = yes
			country_has_primary_culture = cu:boer
		}
	}
}
