sb_has_any_slavery_law = {
	OR = {
		has_law = law_type:law_slave_trade
		has_law = law_type:law_legacy_slavery
		has_law = law_type:law_debt_slavery
		has_law = law_type:law_colonial_slavery
	}
}

sb_should_queue_post_recognition_slavery_response = {
	OR = {
		country_definition = cd:TRN
		country_definition = cd:ORA
	}
	is_subject = no
	sb_has_any_slavery_law = yes
	OR = {
		has_variable = sb_sand_river_convention_accepted_var
		has_variable = sb_bloemfontein_convention_accepted_var
	}
	NOT = { has_variable = sb_gbr_highveld_slavery_response_pending_var }
}
