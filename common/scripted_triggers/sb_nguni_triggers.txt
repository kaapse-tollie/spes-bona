# Spes Bona - Nguni formation triggers

sb_nguni_controls_80_percent_eastern_cape = {
	s:STATE_EASTERN_CAPE = {
		any_scope_state = {
			owner = ROOT
			num_provinces >= 31
		}
	}
}

sb_nguni_controls_80_percent_zululand = {
	s:STATE_ZULULAND = {
		any_scope_state = {
			owner = ROOT
			num_provinces >= 18
		}
	}
}

sb_nguni_controls_80_percent_east_transvaal = {
	s:STATE_EAST_TRANSVAAL = {
		any_scope_state = {
			owner = ROOT
			num_provinces >= 16
		}
	}
}

sb_nguni_controls_80_percent_lourenco_marques = {
	s:STATE_LOURENCO_MARQUES = {
		any_scope_state = {
			owner = ROOT
			num_provinces >= 34
		}
	}
}

sb_nguni_controls_three_core_state_province_blocks = {
	OR = {
		AND = {
			sb_nguni_controls_80_percent_eastern_cape = yes
			sb_nguni_controls_80_percent_zululand = yes
			sb_nguni_controls_80_percent_east_transvaal = yes
		}
		AND = {
			sb_nguni_controls_80_percent_eastern_cape = yes
			sb_nguni_controls_80_percent_zululand = yes
			sb_nguni_controls_80_percent_lourenco_marques = yes
		}
		AND = {
			sb_nguni_controls_80_percent_eastern_cape = yes
			sb_nguni_controls_80_percent_east_transvaal = yes
			sb_nguni_controls_80_percent_lourenco_marques = yes
		}
		AND = {
			sb_nguni_controls_80_percent_zululand = yes
			sb_nguni_controls_80_percent_east_transvaal = yes
			sb_nguni_controls_80_percent_lourenco_marques = yes
		}
	}
}
