# Spes Bona - frontier AI helper triggers

sb_ai_trn_available_for_ora_scripted_play = {
	exists = c:TRN
	c:TRN ?= {
		is_country_alive = yes
		is_ai = yes
		is_at_war = no
		is_active_in_diplomatic_play = no
	}
}

sb_ai_bst_available_for_ora_return_play = {
	NOT = { has_truce_with = c:BST }
	any_scope_state = {
		state_region = s:STATE_VRYSTAAT
	}
	c:BST ?= {
		is_country_alive = yes
		is_ai = yes
		is_subject = no
		is_at_war = no
		is_active_in_diplomatic_play = no
		relations:root < relations_threshold:cordial
		any_scope_state = {
			state_region = s:STATE_VRYSTAAT
		}
	}
}
