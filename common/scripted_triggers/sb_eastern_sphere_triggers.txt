sb_is_imperial_confederation_participant = {
	OR = {
		country_has_primary_culture = cu:boer
		country_has_primary_culture = cu:anglo_african
	}
	capital = {
		OR = {
			state_region = s:STATE_CAPE_COLONY
			state_region = s:STATE_EASTERN_CAPE
			state_region = s:STATE_NORTHERN_CAPE
			state_region = s:STATE_ZULULAND
			state_region = s:STATE_TRANSVAAL
			state_region = s:STATE_EAST_TRANSVAAL
			state_region = s:STATE_NORTHERN_TRANSVAAL
			state_region = s:STATE_VRYSTAAT
			state_region = s:STATE_DRAKENSBERG
			state_region = s:STATE_BOTSWANA
			state_region = s:STATE_LOURENCO_MARQUES
			state_region = s:STATE_ZAMBEZIA
			state_region = s:STATE_ZAMBEZI
			state_region = s:STATE_NAMAQUALAND
			state_region = s:STATE_HEREROLAND
		}
	}
	NOT = { is_country_type = decentralized }
	NOT = { country_definition = cd:SAF }
}

sb_is_autonomous_anglo_african_british_subject = {
	country_has_primary_culture = cu:anglo_african
	is_subject_of = c:GBR
	NOT = { is_subject_type = subject_type_puppet }
	NOT = { is_subject_type = subject_type_tributary }
	NOT = { is_subject_type = subject_type_vassal }
	NOT = { is_subject_type = subject_type_personal_union }
	NOT = { is_subject_type = subject_type_chartered_company }
}

sb_is_imperial_confederation_owner = {
	OR = {
		AND = {
			country_definition = cd:GBR
			NOT = {
				any_country = {
					sb_is_autonomous_anglo_african_british_subject = yes
				}
			}
		}
		AND = {
			sb_is_autonomous_anglo_african_british_subject = yes
			NOT = {
				any_country = {
					sb_is_autonomous_anglo_african_british_subject = yes
					prestige > root.prestige
				}
			}
		}
	}
}

		sb_is_imperial_confederation_participant = yes
		sb_controls_any_complete_southern_africa_state = yes

		}
	}

	NOT = {
		any_country = {
			sb_is_imperial_confederation_participant = yes
			NOT = { this = root }
			NOR = {
				is_subject_of = root
				AND = {
					exists = top_overlord
					top_overlord = c:GBR
				}
			}
		}
	}
}

		NOT = { has_variable = sb_imperial_trade_binding_var }
	}
}

sb_delagoa_owner_outside_gbr_bloc = {
	OR = {
		c:POR ?= {
			is_country_alive = yes
			OR = {
				any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:POR }
				owns_entire_state_region = STATE_LOURENCO_MARQUES
			}
			NOT = { is_in_same_power_bloc = c:GBR }
		}
		c:MZQ ?= {
			is_country_alive = yes
			OR = {
				any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:MZQ }
				owns_entire_state_region = STATE_LOURENCO_MARQUES
			}
			NOT = { is_in_same_power_bloc = c:GBR }
		}
	}
}
