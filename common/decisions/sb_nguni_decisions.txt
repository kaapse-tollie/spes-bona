# Spes Bona - Nguni decisions

sb_proclaim_nguni_nation = {
	is_shown = {
		OR = {
			country_definition = cd:ZUL
			country_definition = cd:SWZ
			country_definition = cd:GZA
		}
		is_subject = no
	}

	possible = {
		has_technology_researched = nationalism
		sb_nguni_conqueror_path_open = yes
	}

	when_taken = {
		if = {
			limit = { country_definition = cd:ZUL }
			set_variable = sb_ngu_inherited_zul_content_var
		}
		else_if = {
			limit = { country_definition = cd:GZA }
			set_variable = sb_ngu_inherited_gza_content_var
		}
		change_tag = NGU
		c:NGU ?= {
			sb_nguni_apply_formation_inheritance = yes
		}
	}

	ai_chance = {
		value = 100
	}
}
