# Generic colony-side constitutional autonomy actions for Spes Bona.

sb_grant_responsible_government = {
	groups = {
		overlord
	}
	requires_approval = no
	show_effect_in_tooltip = yes

	texture = "gfx/interface/icons/diplomatic_action_icons/responsible_government.dds"
	confirmation_sound = "event:/SFX/UI/SubjectInteractions/subject_button_small_autonomy_increased"

	potential = {
		scope:target_country = {
			is_subject_of = root
			is_subject_type = subject_type_colony
			is_country_type = colonial
		}
	}

	possible = {
		scope:target_country = {
			is_subject_of = root
			is_subject_type = subject_type_colony
			is_country_type = colonial
			has_law = law_type:law_colonial_administration
			NOT = { is_diplomatic_play_enemy_of = root }
		}
		custom_tooltip = {
			text = "SB_GRANT_RESPONSIBLE_GOVERNMENT_ON_COOLDOWN_OVERLORD"
			NOT = {
				scope:target_country = {
					any_diplomatic_catalyst = {
						is_diplomatic_catalyst_type = sb_catalyst_responsible_government_imposed
						target = root
					}
				}
			}
		}
	}

	accept_effect = {
		scope:target_country = {
			# sb_grant_responsible_government_effect = yes # moved effects into here directly

			add_liberty_desire = -30
			change_relations = {
				country = scope:actor
				value = 20
			}
			sb_apply_responsible_colony_subject_type_from_overlord = yes
			sb_apply_responsible_colony_governance_from_overlord = yes
			hidden_effect = {
				if = {
					limit = {
						country_definition = cd:CAP
					}
					trigger_event = { id = sb_cape.020 popup = yes }
				}
				create_diplomatic_catalyst = {
					type = sb_catalyst_responsible_government_imposed
					target = root
				}
			}
		}
	}

	# values tweaked from autonomy (softend)
	ai = {
		evaluation_chance = {
			value = 0.25
		}

		propose_score = {
			value = 10
		}

		will_propose = {
			scope:target_country = {
				liberty_desire >= 60 # down from 75
			}
			OR = {
				AND = {
					"relations(scope:target_country)" >= relations_threshold:amicable # down from friendly
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}
					OR = {
						has_strategy = ai_strategy_anti_imperialism
						has_strategy = ai_strategy_maintain_power_balance
					}
				}
				scope:target_country = { "potential_diplomatic_play_power_ratio(root)" >= 0.50 } # down from 0.75
			}
		}

	}
}


sb_ask_responsible_government = {
	groups = {
		subject
	}
	requires_approval = yes
	show_effect_in_tooltip = yes
	can_use_obligations = yes
	uses_random_approval = yes

	texture = "gfx/interface/icons/diplomatic_action_icons/responsible_government.dds"

	potential = {
		is_subject = yes
		is_subject_type = subject_type_colony
		is_country_type = colonial
	}

	possible = {
		scope:target_country = {
			NOT = {
				is_diplomatic_play_enemy_of = root
			}
		}

		# Might need more checks see vanilla 01_colonial_administrations.txt
		is_subject_type = subject_type_colony
		is_country_type = colonial
		has_law = law_type:law_colonial_administration # not sure if narrow enough!

		# These cooldown tooltips mirror vanilla autonomy, but keep distinct keys
		# so the responsible-government interaction can be adjusted independently.
		custom_tooltip = {
			text = "SB_REQUEST_RESPONSIBLE_GOVERNMENT_ON_COOLDOWN"
			NOT = {
				scope:target_country = {
					any_diplomatic_catalyst = {
						is_diplomatic_catalyst_type = sb_catalyst_responsible_government_requested
						target = root
					}
				}
			}
		}
		custom_tooltip = {
			text = "SB_GRANT_RESPONSIBLE_GOVERNMENT_ON_COOLDOWN_SUBJECT"
			NOT = {
				scope:target_country = {
					any_diplomatic_catalyst = {
						is_diplomatic_catalyst_type = sb_catalyst_responsible_government_imposed
						target = scope:target_country
					}
				}
			}
		}
	}

	accept_effect = {
		# sb_grant_responsible_government_effect = yes # Moved directly here

		add_liberty_desire = -30
		change_relations = {
			country = scope:actor
			value = 20
		}
		sb_apply_responsible_colony_subject_type_from_overlord = yes
		sb_apply_responsible_colony_governance_from_overlord = yes

		if = {
			limit = {
				country_definition = cd:CAP
			}
			trigger_event = { id = sb_cape.020 popup = yes }
		}




		hidden_effect = {
			scope:target_country = {
				create_diplomatic_catalyst = {
					type = sb_catalyst_responsible_government_requested
					target = root
				}
			}
			create_diplomatic_catalyst = {
				type = sb_catalyst_responsible_government_imposed
				target = scope:target_country
			}
		}
	}

	decline_effect = {
		add_liberty_desire = 10
		scope:target_country = { change_relations = { country = root value = -20 } } # +10 from autonomy decline
		hidden_effect = {
			scope:target_country = {
				create_diplomatic_catalyst = {
					type = sb_catalyst_responsible_government_requested
					target = root
				}
			}
			create_diplomatic_catalyst = {
				type = sb_catalyst_responsible_government_denied
				target = scope:target_country
			}
		}
	}

	# Values taken from 38_subjects_increase_autonomy.txt and softend
	ai = {
		evaluation_chance = {
			value = 0.25
		}

		accept_score = {
			value = 0

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = -100
			}

			if = {
				limit = {
					"scope:actor.ai_ideological_opinion(root)" > 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION"
					value = "scope:actor.ai_ideological_opinion(root)"
					multiply = 0.5
					max = 25
				}
			}
			else_if = {
				limit = {
					"scope:actor.ai_ideological_opinion(root)" < 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_IDEOLOGICAL_OPINION_NEGATIVE"
					value = "scope:actor.ai_ideological_opinion(root)"
					multiply = 0.5
					min = -25
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
				if = {
					limit = {
						has_attitude = {
							who = scope:actor
							attitude = protective
						}
					}
					value = 50
				}
				else_if = {
					limit = {
						has_attitude = {
							who = scope:actor
							attitude = domineering
						}
					}
					value = -50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_CHANCE_TO_WIN_DIPLOMATIC_PLAY"
				value = "scope:actor.potential_diplomatic_play_power_ratio(root)"
				subtract = 0.25 # down from 0.5
				multiply = 50
				max = 100
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ACTOR_LIBERTY_DESIRE"
				value = "scope:actor.liberty_desire"
				subtract = 60 # down from 75
				multiply = 2
				min = -50
				max = 50
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 50
				}
				else_if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
					}
					value = 25
				}
				else_if = {
					limit = {
						"relations(scope:actor)" >=  relations_threshold:cordial
					}
					value = 10
				}
				else_if = {
					limit = {
						"relations(scope:actor)" > relations_threshold:poor
					}
					value = -25
				}
				else = {
					value = -50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -10
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -25
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -50
				}
			}
		}

		propose_score = {
			value = 100
		}

		will_propose = {
			OR = {
				has_strategy = ai_strategy_subject_strengthen_self
				has_strategy = ai_strategy_subject_break_free
			}
		}

		will_propose_even_if_not_accepted = {
			always = no # otherwise will push in autonomy region
		}
	}
}
