# Spes Bona - Phase 3 branch snapshot and Wave 1 helpers

# Apply the inherited Phase 1-2 branch interface onto the current country.
# This is intentionally snapshot-based: later Phase 3 content should read these
# fixed route markers rather than re-deciding branch outcomes every year.
sb_phase3_initialize_branch_snapshot = {
	# CAP branch family
	if = {
		limit = { c:CAP ?= { has_variable = sb_cape_liberal_resolved_var } }
		set_variable = sb_phase3_cape_liberal_path_var
	}
	if = {
		limit = { c:CAP ?= { has_modifier = sb_settler_victory } }
		set_variable = sb_phase3_cape_settler_path_var
	}
	if = {
		limit = { c:CAP ?= { has_modifier = sb_cape_frontier_compromise } }
		set_variable = sb_phase3_cape_compromise_path_var
	}
	# Shell-only late-century snapshot rails are intentionally commented out
	# until their downstream Phase 3 waves land. Keeping them live now only
	# creates startup "set but never used" noise.
}

# Seed the player-facing Wave 1 interface on whichever country should carry the
# opening late-century content in the current run.
sb_phase3_seed_public_wave1_interface = {
	set_variable = sb_phase3_mineral_revolution_available_var
}

# Resolve the initial Griqualand West route. The heavier territorial/dynamic-
# naming work is a later Wave 1 expansion; for now we land the political
# direction cleanly so later content has a stable branch to build on.
sb_phase3_apply_glw_arbitration_path = {
	set_variable = sb_phase3_glw_question_resolved_var
	set_variable = sb_phase3_glw_arbitration_path_var

	if = {
		limit = {
			c:WBL ?= {
				is_country_alive = yes
			}
		}
		change_relations = {
			country = c:WBL
			value = 20
		}
	}
	if = {
		limit = {
			c:TRN ?= {
				is_country_alive = yes
				has_variable = sb_sand_river_convention_accepted_var
			}
		}
		c:TRN ?= {
			change_relations = {
				country = ROOT
				value = 5
			}
		}
	}
}

sb_phase3_apply_glw_hardline_path = {
	set_variable = sb_phase3_glw_question_resolved_var
	set_variable = sb_phase3_glw_hardline_path_var

	if = {
		limit = {
			c:WBL ?= {
				is_country_alive = yes
			}
		}
		change_relations = {
			country = c:WBL
			value = -20
		}
	}
	if = {
		limit = {
			c:CAP ?= {
				is_country_alive = yes
			}
		}
		s:STATE_NORTHERN_CAPE = {
			add_claim = c:CAP
		}
	}
	else = {
		s:STATE_NORTHERN_CAPE = {
			add_claim = ROOT
		}
	}
	if = {
		limit = {
			c:TRN ?= {
				is_country_alive = yes
				has_variable = sb_sand_river_convention_accepted_var
			}
		}
		c:TRN ?= {
			change_relations = {
				country = ROOT
				value = -10
			}
		}
	}
}

sb_phase3_apply_kimberley_london_expedition_path = {
	set_variable = sb_phase3_glw_question_resolved_var
	set_variable = sb_phase3_glw_hardline_path_var

	p:x68B5E8.state.owner = {
		save_scope_as = kimberley_holder
	}

	s:STATE_NORTHERN_CAPE = {
		add_claim = root
		if = {
			limit = {
				c:GBR ?= {
					is_country_alive = yes
				}
			}
			add_claim = c:GBR
		}
	}
	scope:kimberley_holder ?= {
		change_relations = {
			country = root
			value = -30
		}
		c:GBR ?= {
			change_relations = {
				country = scope:kimberley_holder
				value = -20
			}
		}
	}
	c:GBR ?= {
		change_relations = {
			country = root
			value = 10
		}
	}
}

sb_phase3_annul_wbl_guarantee_frame = {
	c:GBR ?= {
		every_scope_treaty = {
			limit = {
				binds = c:WBL
				any_scope_article = {
					has_type = guarantee_independence
					target_country = c:WBL
				}
			}
			withdraw = {
				country = c:GBR
			}
		}
	}
	every_scope_treaty = {
		limit = {
			binds = c:WBL
			any_scope_article = {
				has_type = guarantee_independence
				target_country = c:WBL
			}
		}
		withdraw = {
			country = root
		}
	}
}

sb_phase3_apply_wbl_protection_path = {
	set_variable = sb_phase3_glw_question_resolved_var
	set_variable = sb_phase3_glw_arbitration_path_var
	sb_phase3_annul_wbl_guarantee_frame = yes

	if = {
		limit = {
			sb_cap_is_full_liberal_griqua_route = yes
		}
		annex = c:WBL
	}
	else = {
		if = {
			limit = {
				c:WBL ?= {
					is_country_alive = yes
					NOT = { is_subject_of = root }
				}
			}
			create_diplomatic_pact = {
				country = c:WBL
				type = puppet
			}
		}
		else_if = {
			limit = {
				c:WBL ?= {
					is_country_alive = yes
					is_subject_of = root
					NOT = { is_subject_type = subject_type_puppet }
				}
			}
			c:WBL ?= {
				change_subject_type = subject_type_puppet
			}
		}
		c:WBL ?= {
			change_relations = {
				country = root
				value = 30
			}
		}
	}
}

sb_phase3_create_kimberley_goldfield_building_if_missing = {
	random_scope_state = {
		limit = {
			state_region = s:STATE_NORTHERN_CAPE
		}
		if = {
			limit = {
				has_potential_resource = building_gold_field
				NOT = { has_building = building_gold_field }
			}
			force_resource_discovery = building_gold_field
		}
	}
}

sb_phase3_create_gle_route = {
	set_variable = sb_griqualand_east_resolved_var

	if = {
		limit = {
			NOT = { exists = c:GLE }
		}
		create_country = {
			tag = GLE
			origin = root
			province = p:xE60DC5
		}
	}

	c:GLE ?= {
		set_country_type = unrecognized
		if = {
			limit = {
				NOT = { is_subject_of = root }
			}
			root = {
				create_diplomatic_pact = {
					country = c:GLE
					type = puppet
				}
			}
		}
		if = {
			limit = {
				c:XHG ?= { is_country_alive = yes }
			}
			annex = c:XHG
		}
		if = {
			limit = {
				c:XHR ?= { is_country_alive = yes }
			}
			annex = c:XHR
		}
		if = {
			limit = {
				c:XHT ?= { is_country_alive = yes }
			}
			annex = c:XHT
		}
		change_relations = {
			country = root
			value = 30
		}
	}

	p:xC5B1AF.state.owner = {
		save_scope_as = griqua_migration_source
	}
	cu:griqua = {
		save_scope_as = griqua_migration_culture
	}
	c:GLE ?= {
		random_scope_state = {
			limit = {
				state_region = s:STATE_EASTERN_CAPE
			}
			save_scope_as = griqua_migration_target
		}
	}
	scope:griqua_migration_target ?= {
		create_mass_migration = {
			origin = scope:griqua_migration_source
			culture = scope:griqua_migration_culture
		}
	}
	scope:griqua_migration_source ?= {
		random_scope_state = {
			limit = {
				state_region = s:STATE_VRYSTAAT
			}
			every_scope_pop = {
				limit = {
					culture = cu:griqua
				}
				move_partial_pop = {
					state = scope:griqua_migration_target
					population_ratio = 0.20
				}
			}
		}
	}

	if = {
		limit = {
			sb_cap_is_full_liberal_griqua_route = yes
			c:GLE ?= {
				is_country_alive = yes
			}
		}
		c:GLE ?= {
			set_variable = sb_gle_willing_annexation_pending_var
			trigger_event = { id = sb_griqualand_east.221 days = 1825 popup = yes }
		}
	}
}
