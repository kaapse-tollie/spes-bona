# Spes Bona - Natalia British Colony and Refugee Effects
#
# This file owns the later Natalia branch where the Boer republic is either
# forced into submission or reorganized by Britain as a crown colony. Keeping
# the colony/refugee work separate from `sb_natalia_effects.txt` leaves the
# Blood River and republican-foundation path much easier to navigate.

sb_assign_british_natal_colony_territory = {
	s:STATE_ZULULAND = {
		set_owner_of_provinces = {
			country = c:NAL
			provinces = {
				x279045
				xBE6FEE x1A084B xBFA16B x9E9742 x88FAD4 x904EBE x41C070 xE882CE xE1E455 xBBCA32
				x5B124F xFF0EF1 x552449 xE0EB02 x85695F xDE0EDE x7ACC38 xB1F868 x3CED3D x11A090 xCD31DB
			}
		}
		add_claim = c:NAL
	}
}

# Britain can respond to Dingane's coastal raid by annexing the Zulu-held Natal
# littoral outright and reorganizing it as a crown colony.
sb_place_natalia_under_british_protectorate = {
	change_relations = {
		country = c:GBR
		value = 50
	}
	if = {
		limit = {
			NOT = { is_subject_of = c:GBR }
		}
		c:GBR ?= {
			create_diplomatic_pact = {
				country = root
				type = protectorate
			}
		}
	}
	else_if = {
		limit = {
			is_subject_of = c:GBR
			NOT = { is_subject_type = subject_type_protectorate }
			NOT = { is_subject_type = subject_type_colony }
		}
		change_subject_type = subject_type_protectorate
	}
}

sb_convert_natalia_to_british_crown_colony = {
	if = {
		limit = {
			NOT = { has_variable = sb_natalia_british_colony_resolved_var }
		}
		set_variable = sb_natalia_british_colony_resolved_var
	}
	change_relations = {
		country = c:GBR
		value = 50
	}
	# Vanilla colony subject types only remain valid on countries with the
	# colonial country type. Without this, Natalia can briefly become a colony
	# and then immediately drop back out of subjecthood on the next tick.
	if = {
		limit = {
			NOT = { is_country_type = colonial }
		}
		set_country_type = colonial
	}
	if = {
		limit = {
			NOT = { is_subject_of = c:GBR }
		}
		c:GBR ?= {
			create_diplomatic_pact = {
				country = root
				type = colony
			}
		}
	}
	else_if = {
		limit = {
			is_subject_of = c:GBR
			NOT = { is_subject_type = subject_type_colony }
		}
		change_subject_type = subject_type_colony
	}
	sb_trigger_british_natal_colony_migration_effect = yes
}

sb_apply_british_natal_colony_setup = {
	sb_convert_natalia_to_british_crown_colony = yes
	if = {
		limit = {
			has_variable = sb_commandant_general_visible_var
		}
		remove_variable = sb_commandant_general_visible_var
	}
	if = {
		limit = {
			has_variable = sb_early_republic_var
		}
		remove_variable = sb_early_republic_var
	}
	if = {
		limit = { has_modifier = sb_trek_frontier_drive_natalia }
		remove_modifier = sb_trek_frontier_drive_natalia
	}
	if = {
		limit = {
			NOT = { has_variable = southern_africa_subject_var }
		}
		set_variable = southern_africa_subject_var
	}
	activate_law = law_type:law_colonial_administration
	activate_law = law_type:law_professional_army
	activate_law = law_type:law_slavery_banned
	if = {
		limit = {
			c:ZUL ?= { is_country_alive = yes }
			NOT = { has_truce_with = c:ZUL }
		}
		create_bidirectional_truce = {
			country = c:ZUL
			months = 120
		}
	}
	active_law:lawgroup_army_model ?= {
		if = {
			limit = {
				has_amendment = amendment_type:amendment_sb_commando_corps
			}
			every_scope_amendment = {
				limit = {
					type = amendment_type:amendment_sb_commando_corps
				}
				remove_amendment = yes
			}
		}
	}
	if = {
		limit = {
			NOT = { country_has_primary_culture = cu:british }
		}
		add_primary_culture = cu:british
	}
	if = {
		limit = {
			country_has_primary_culture = cu:boer
		}
		remove_primary_culture = cu:boer
	}
	sb_apply_generic_british_colony_interest_group_identity = yes
	s:STATE_ZULULAND = {
		set_owner_of_provinces = {
			country = root
			provinces = { x279045 }
		}
		region_state:NAL ?= {
			set_state_type = unincorporated
		}
	}
	sb_install_british_natal_governor = yes
}

sb_select_natalia_refugee_receiver = {
	random_country = {
		limit = {
			is_country_alive = yes
			OR = {
				country_definition = cd:ORA
				country_definition = cd:TRN
				country_definition = cd:ZPB
				country_definition = cd:LYD
			}
		}
		save_scope_as = natal_refugee_receiver
	}
}

sb_annul_natalia_boer_treaties = {
	if = {
		limit = {
			any_scope_treaty = {
				binds = c:ORA
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:ORA
			}
			withdraw = {
				country = root
			}
		}
	}
	if = {
		limit = {
			any_scope_treaty = {
				binds = c:TRN
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:TRN
			}
			withdraw = {
				country = root
			}
		}
	}
	if = {
		limit = {
			exists = c:ZPB
			any_scope_treaty = {
				binds = c:ZPB
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:ZPB
			}
			withdraw = {
				country = root
			}
		}
	}
	if = {
		limit = {
			exists = c:LYD
			any_scope_treaty = {
				binds = c:LYD
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:LYD
			}
			withdraw = {
				country = root
			}
		}
	}
}

sb_exile_natalia_boer_leadership = {
	if = {
		limit = {
			NOT = { exists = scope:natal_refugee_receiver }
		}
		sb_select_natalia_refugee_receiver = yes
	}
	random_scope_character = {
		limit = {
			has_template = TRN_andries_pretorius
		}
		save_scope_as = natal_exiled_leader
	}
	if = {
		limit = {
			NOT = { exists = scope:natal_exiled_leader }
			exists = ruler
			ruler = {
				culture = cu:boer
			}
		}
		ruler = {
			save_scope_as = natal_exiled_leader
		}
	}
	scope:natal_exiled_leader ?= {
		if = {
			limit = {
				NOT = { has_role = politician }
			}
			add_character_role = politician
		}
		transfer_character = scope:natal_refugee_receiver
		set_interest_group = scope:natal_refugee_receiver.ig:ig_landowners
	}
}

sb_install_british_natal_governor = {
	random_scope_character = {
		limit = {
			has_variable = sb_martin_west_var
		}
		save_scope_as = natal_colonial_governor
	}
	scope:natal_colonial_governor ?= {
		set_interest_group = c:NAL.ig:ig_armed_forces
		if = {
			limit = {
				NOT = { has_role = politician }
			}
			add_character_role = politician
		}
		set_as_interest_group_leader = yes
		set_character_as_ruler = yes
	}
	if = {
		limit = {
			NOT = { exists = scope:natal_colonial_governor }
		}
		create_character = {
			first_name = Martin
			last_name = West
			historical = yes
			ruler = yes
			ig_leader = yes
			birth_date = 1804.1.1
			culture = cu:british
			religion = rel:protestant
			interest_group = ig_armed_forces
			ideology = ideology_moderate
			traits = {
				tactful
				master_bureaucrat
				persistent
			}
			on_created = {
				set_variable = sb_martin_west_var
			}
		}
	}
}

sb_finalize_british_natal_colony = {
	sb_select_natalia_refugee_receiver = yes
	sb_exile_natalia_boer_leadership = yes
	sb_apply_british_natal_colony_setup = yes
	sb_annul_natalia_boer_treaties = yes
	sb_trigger_selected_natal_refugee_receiver_followup = yes
}

	sb_assign_british_natal_colony_territory = yes
	c:NAL ?= {
		sb_seed_british_natal_colony_shell = yes
		sb_apply_british_natal_colony_setup = yes
	}
}

sb_trigger_random_natal_refugee_receiver_event = {
	sb_select_natalia_refugee_receiver = yes
	sb_trigger_selected_natal_refugee_receiver_followup = yes
}

sb_trigger_selected_natal_refugee_receiver_followup = {
	if = {
		limit = {
			exists = scope:natal_refugee_receiver
			scope:natal_refugee_receiver ?= {
				country_definition = cd:ORA
			}
		}
		scope:natal_refugee_receiver = {
			trigger_event = { id = sb_boer_zulu.116 days = 1 popup = yes }
		}
	}
	else_if = {
		limit = {
			exists = scope:natal_refugee_receiver
			scope:natal_refugee_receiver ?= {
				country_definition = cd:TRN
			}
		}
		scope:natal_refugee_receiver = {
			trigger_event = { id = sb_boer_zulu.117 days = 1 popup = yes }
		}
	}
	else_if = {
		limit = {
			exists = scope:natal_refugee_receiver
		}
		scope:natal_refugee_receiver ?= {
			sb_trigger_natal_refugee_migration_effect = yes
		}
	}
}

# Add a Boer ally directly to Natalia's side once Britain turns the ultimatum
# into a live diplomatic play.
sb_join_natalia_against_britain = {
	save_scope_as = natalia_supporter
	random_diplomatic_play = {
		limit = {
			is_diplomatic_play_type = dp_make_protectorate
			any_scope_play_involved = { c:GBR ?= THIS }
			any_scope_play_involved = { c:NAL ?= THIS }
		}
		add_target_backers = { scope:natalia_supporter }
		add_war_goal = {
			holder = c:GBR
			target_country = scope:natalia_supporter
			type = make_protectorate
		}
		if = {
			limit = {
				scope:natalia_supporter ?= {
					country_definition = cd:ORA
				}
			}
			add_war_goal = {
				holder = scope:natalia_supporter
				target_country = c:CAP
				type = liberate_country
			}
		}
	}
}

# Support Natalia against Britain with both a live war join and a short-term
# defensive pact, so the diplomatic relationship survives beyond the first war.
sb_support_natalia_against_britain = {
	c:NAL ?= {
		save_scope_as = natalia
	}
	if = {
		limit = {
			NOT = {
				any_scope_treaty = {
					binds = root
					binds = scope:natalia
					any_scope_article = {
						OR = {
							has_type = defensive_pact
							has_type = alliance
						}
					}
				}
			}
		}
		create_treaty = {
			first_country = root
			second_country = scope:natalia
			is_draft = no
			binding_period = { years = 10 }
			articles_to_create = {
				{
					article = defensive_pact
				}
			}
		}
	}
	sb_join_natalia_against_britain = yes
}

# Boer refugees from Natal move inland toward whichever republic receives them.
sb_trigger_natal_refugee_migration_effect = {
	c:NAL ?= {
		save_scope_as = natalia_source
	}
	cu:boer = {
		save_scope_as = migration_culture
	}
	ordered_scope_state = {
		limit = {
			is_incorporated = yes
		}
		order_by = migration_pull
		position = 0
		check_range_bounds = no
		save_scope_as = migration_target
	}
	if = {
		limit = {
			exists = scope:natalia_source
			exists = scope:migration_target
		}
		scope:migration_target = {
			create_mass_migration = {
				origin = scope:natalia_source
				culture = scope:migration_culture
			}
		}
		c:NAL.capital ?= {
			every_scope_pop = {
				limit = {
					culture = cu:boer
				}
				move_partial_pop = {
					state = scope:migration_target
					population_ratio = 0.20
				}
			}
		}
	}
}

# Natalia needs its own trekker migration helper because the generic trek effect
# targets the calling country, which is ORA in the formation chain.
sb_trigger_natalia_boer_migration_effect = {
	random_country = {
		limit = {
			country_definition = cd:CAP
			is_country_alive = yes
		}
		save_scope_as = boer_migration_origin
	}
	c:NAL ?= {
		ordered_scope_state = {
			order_by = migration_pull
			position = 0
			check_range_bounds = no
			save_scope_as = boer_migration_target
		}
	}
	if = {
		limit = {
			exists = scope:boer_migration_origin
			exists = scope:boer_migration_target
		}
		scope:boer_migration_target = {
			create_mass_migration = {
				origin = scope:boer_migration_origin
				culture = cu:boer
			}
		}
		scope:boer_migration_origin = {
			add_modifier = {
				name = sb_trek_migration_cost
				months = 24
			}
		}
	}
}

# Submission to Britain should encourage an English inflow toward the port
# colony, underscoring that the coast has been pulled back into imperial orbit.
sb_trigger_british_natal_colony_migration_effect = {
	random_country = {
		limit = {
			country_definition = cd:GBR
			is_country_alive = yes
			any_primary_culture = {
				this = cu:british
			}
		}
		save_scope_as = british_migration_origin
	}
	c:NAL ?= {
		ordered_scope_state = {
			order_by = migration_pull
			position = 0
			check_range_bounds = no
			save_scope_as = british_migration_target
		}
	}
	if = {
		limit = {
			exists = scope:british_migration_origin
			exists = scope:british_migration_target
		}
		scope:british_migration_target = {
			create_mass_migration = {
				origin = scope:british_migration_origin
				culture = cu:british
			}
		}
	}
}
