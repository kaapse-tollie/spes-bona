sb_gaza_clear_runtime = {
	remove_variable = sb_gaza_event_cooldown_var
	remove_variable = sb_gaza_010_fired_var
	remove_variable = sb_gaza_020_fired_var
	remove_variable = sb_gaza_030_fired_var
	remove_variable = sb_gaza_040_fired_var
}

sb_gaza_monthly_random_event = {
	if = {
		limit = {
			has_journal_entry = je_sb_consolidating_gaza
			NOT = { has_variable = sb_gaza_event_cooldown_var }
		}
		random_list = {
			8 = {
				trigger = { NOT = { has_variable = sb_gaza_010_fired_var } }
				set_variable = { name = sb_gaza_event_cooldown_var months = 1 }
				trigger_event = { id = sb_gaza.010 days = 7 popup = yes }
			}
			5 = {
				trigger = { NOT = { has_variable = sb_gaza_020_fired_var } }
				set_variable = { name = sb_gaza_event_cooldown_var months = 1 }
				trigger_event = { id = sb_gaza.020 days = 7 popup = yes }
			}
			5 = {
				trigger = { NOT = { has_variable = sb_gaza_030_fired_var } }
				set_variable = { name = sb_gaza_event_cooldown_var months = 1 }
				trigger_event = { id = sb_gaza.030 days = 7 popup = yes }
			}
			5 = {
				trigger = {
					NOT = { has_variable = sb_gaza_040_fired_var }
					c:POR ?= {
						OR = {
							any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:POR }
							any_scope_state = { state_region = s:STATE_ZAMBEZIA owner = c:POR }
						}
					}
				}
				set_variable = { name = sb_gaza_event_cooldown_var months = 1 }
				trigger_event = { id = sb_gaza.040 days = 7 popup = yes }
			}
			77 = { }
		}
	}
}

sb_imperial_confederation_bind_trade = {
	save_scope_as = sb_confed_owner
	every_country = {
		limit = {
			sb_is_imperial_confederation_participant = yes
			is_subject = no
			NOT = { this = scope:sb_confed_owner }
			NOT = { has_variable = sb_imperial_trade_binding_var }
		}
		save_temporary_scope_as = sb_bound_participant
		set_variable = {
			name = sb_imperial_trade_binding_var
			months = 60
		}
		scope:sb_confed_owner = {
			create_treaty = {
				first_country = scope:sb_confed_owner
				second_country = scope:sb_bound_participant
				is_draft = no
				binding_period = { years = 5 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = scope:sb_confed_owner
						target_country = scope:sb_bound_participant
					}
					{
						article = goods_transfer
						source_country = scope:sb_confed_owner
						target_country = scope:sb_bound_participant
						inputs = {
							{ goods = g:merchant_marine }
							{ quantity = 10 }
						}
					}
				}
			}
		}
		if = {
			limit = {
				c:GBR ?= { is_country_alive = yes }
			}
			sb_ensure_gbr_power_bloc_embassy_with_root = yes
		}
	}
}

sb_imperial_confederation_form_saf = {
	if = {
		limit = {
			country_definition = cd:GBR
			c:CAP ?= { is_country_alive = yes }
		}
		c:CAP ?= {
			save_scope_as = sb_saf_founder
			change_tag = SAF
		}
	}
	else = {
		save_scope_as = sb_saf_founder
		change_tag = SAF
	}
c:SAF ?= {
		while = {
			limit = {
				any_country = {
					sb_is_imperial_confederation_participant = yes
					is_ai = yes
					NOT = { this = c:SAF }
					OR = {
						is_subject_of = c:SAF
						AND = {
							c:SAF = { is_subject = yes }
							is_subject_of = c:SAF.top_overlord
						}
					}
				}
			}
			random_country = {
				limit = {
					sb_is_imperial_confederation_participant = yes
					is_ai = yes
					NOT = { this = c:SAF }
					OR = {
						is_subject_of = c:SAF
						AND = {
							c:SAF = { is_subject = yes }
							is_subject_of = c:SAF.top_overlord
						}
					}
				}
				save_temporary_scope_as = sb_saf_participant_to_annex
			}
			annex_with_incorporation = scope:sb_saf_participant_to_annex
		}
	}
}

sb_eastern_sphere_monthly_housekeeping = {
	if = {
		limit = {
			OR = {
				country_definition = cd:GZA
				AND = {
					country_definition = cd:NGU
					has_variable = sb_ngu_inherited_gza_content_var
				}
			}
			is_subject = no
			NOT = { has_variable = sb_gaza_consolidated_var }
			NOT = { has_variable = sb_gaza_consolidation_failed_var }
			NOT = { has_journal_entry = je_sb_consolidating_gaza }
			sb_gaza_has_target_holdings = yes
		}
		add_journal_entry = { type = je_sb_consolidating_gaza }
	}
	if = {
		limit = {
			country_has_primary_culture = cu:portuguese
			is_subject = no
			sb_portuguese_controls_mozambique_admin_area = yes
			NOT = { exists = c:MZQ }
		}
		trigger_event = { id = sb_mozambique_company.010 days = 1 popup = yes }
	}
	if = {
		limit = {
			country_has_primary_culture = cu:portuguese
			is_subject = no
			exists = c:MZQ
		}
		sb_portugal_redirect_southern_africa_claims_to_mzq = yes
	}
	if = {
		limit = {
			country_definition = cd:TRN
			NOT = { has_journal_entry = je_sb_delagoa_route_to_sea }
			exists = c:POR
			sb_delagoa_owner_outside_gbr_bloc = yes
			has_journal_entry = je_sb_imperial_confederation_scheme
			any_scope_state = {
				state_region = s:STATE_EAST_TRANSVAAL
				owner = root
				is_incorporated = yes
			}
		}
		add_journal_entry = { type = je_sb_delagoa_route_to_sea }
	}
	if = {
		limit = {
			country_definition = cd:TRN
			is_ai = yes
			has_journal_entry = je_sb_delagoa_route_to_sea
			is_subject = no
			NOT = { has_variable = sb_delagoa_treaty_refused_var }
			sb_has_railway_in_transvaal = yes
			sb_has_railway_in_east_transvaal = yes
			NOT = { sb_trn_has_delagoa_trade_through = yes }
		}
		if = {
			limit = {
				c:MZQ ?= {
					is_country_alive = yes
					any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:MZQ }
				}
			}
			c:MZQ ?= { trigger_event = { id = sb_delagoa.010 days = 1 popup = yes } }
		}
		else_if = {
			limit = {
				c:POR ?= {
					is_country_alive = yes
					any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:POR }
				}
			}
			c:POR ?= { trigger_event = { id = sb_delagoa.010 days = 1 popup = yes } }
		}
	}
	if = {
		limit = {
			country_definition = cd:GBR
			is_ai = yes
			has_journal_entry = je_sb_imperial_confederation_scheme
		}
		if = {
			limit = {
				c:TRN ?= {
					is_country_alive = yes
					NOT = { is_in_same_power_bloc = c:GBR }
					OR = {
						any_scope_state = { state_region = s:STATE_LOURENCO_MARQUES owner = c:TRN }
						owns_entire_state_region = STATE_LOURENCO_MARQUES
					}
					any_scope_state = {
						state_region = s:STATE_LOURENCO_MARQUES
						owner = c:TRN
						has_active_building = building_port
						has_active_building = building_trade_center
					}
					NOT = { has_variable = sb_imperial_confederation_delagoa_response_var }
				}
			}
			trigger_event = { id = sb_imperial_confederation.030 days = 1 popup = yes }
		}
		if = {
			limit = {
				NOT = { has_variable = sb_nal_anglo_zulu_accelerated_var }
				c:NAL ?= {
					is_country_alive = yes
					is_ai = yes
					is_subject = yes
				}
				c:ZUL ?= {
					is_country_alive = yes
					NOT = { is_in_same_power_bloc = c:GBR }
				}
				any_country = {
					sb_is_imperial_confederation_participant = yes
					country_has_primary_culture = cu:boer
					any_scope_treaty = {
						binds = c:ZUL
						any_scope_article = { has_type = transit_rights }
					}
				}
			}
			set_variable = sb_nal_anglo_zulu_accelerated_var
			trigger_event = { id = sb_imperial_confederation.040 days = 1 popup = yes }
		}
	}
}
