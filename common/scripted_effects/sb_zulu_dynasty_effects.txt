# Zulu dynastic stability helpers
#
# Structure:
#   - small additive/removal helpers for the 0-100 dynastic bar
#   - monthly housekeeping that updates malus tiers and opens follow-up content
#   - collapse / success resolution effects
#
# The bar is intentionally reused by later content:
#   Blood River, Port Natal, and the Swazi question all feed back into the same
#   dynastic-stability story rather than opening disconnected minigames.

sb_boost_firearms_progress_minor = {
	if = {
		limit = {
			has_journal_entry = je_sb_adopt_firearms
			NOT = { has_variable = sb_firearms_modernized_var }
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_treaty_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_treaty_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_industry_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_industry_progress_months_var
				value = 0
			}
		}
		change_variable = {
			name = sb_firearms_treaty_progress_months_var
			add = 2.4
		}
		change_variable = {
			name = sb_firearms_industry_progress_months_var
			add = 2.4
		}
		sb_update_firearms_progress = yes
		sb_sync_iron_age_weapons_penalty = yes
	}
}

sb_zulu_maybe_open_swazi_question = {
	if = {
		limit = {
			sb_is_zulu_dynasty_country = yes
			has_journal_entry = je_sb_zulu_kingdom
			NOT = { has_variable = sb_zulu_swazi_question_opened_var }
			NOT = { has_variable = sb_zulu_swazi_war_active_var }
			OR = {
				has_variable = sb_zulu_swazi_ai_honour_deal_var
				has_variable = sb_zulu_swazi_ai_guns_south_var
				has_variable = sb_zulu_swazi_ai_lost_blood_river_var
				has_variable = sb_zulu_swazi_ai_won_blood_river_var
				has_variable = sb_zulu_blood_river_resolved_var
				has_variable = sb_zulu_port_natal_raid_available_var
				c:ORA ?= {
					has_variable = sb_natal_path_resolved_var
				}
			}
			is_at_war = no
			c:SWZ ?= {
				is_country_alive = yes
				is_subject = no
				any_scope_state = {
					state_region = s:STATE_EAST_TRANSVAAL
				}
			}
			ruler = {
				has_template = ZUL_dingane
			}
		}
		set_variable = sb_zulu_swazi_question_opened_var
		trigger_event = { id = sb_boer_zulu.094 days = 21 popup = yes }
	}
}

sb_zulu_install_summoned_claimant_as_ruler = {
	scope:summoned_character_scope ?= {
		if = {
			limit = {
				is_in_void = yes
			}
			hidden_effect = {
				free_character_from_void = yes
				transfer_character = prev
			}
		}
		if = {
			limit = {
				is_in_exile_pool = yes
			}
			transfer_character = prev
			set_variable = exile_to_ruler_var
			set_variable = {
				name = memento_mori_immunity
				months = 60
			}
			hidden_effect = {
				free_from_exile_effect = yes
			}
		}
		if = {
			limit = {
				NOT = { has_role = politician }
			}
			add_character_role = politician
		}
		set_interest_group = c:ZUL.ig:ig_armed_forces
		hidden_effect = {
			cleanup_ruler_character_roles = yes
		}
		set_character_as_ruler = yes
	}
}

sb_resolve_zulu_dynastic_collapse = {
	if = {
		limit = {
			has_variable = sb_zulu_dynastic_collapse_pending_var
		}
		remove_variable = sb_zulu_dynastic_collapse_pending_var
	}
	set_variable = sb_zulu_dynastic_collapse_resolved_var
	sb_zulu_disable_dynastic_claims = yes
	sb_zulu_clear_dynastic_stability_modifiers = yes
	capital = {
		add_radicals_in_state = {
			value = large_radicals
		}
	}
	if = {
		limit = {
			NOT = { has_modifier = sb_zulu_dynastic_crisis }
		}
		add_modifier = {
			name = sb_zulu_dynastic_crisis
			months = 120
		}
	}
	if = {
		limit = {
			ruler = {
				has_template = ZUL_dingane
			}
		}
		sb_zulu_kill_current_heir = yes
		sb_zulu_prepare_mpande_claimant = yes
		sb_zulu_prepare_cetshwayo_heir = yes
		random_scope_character = {
			limit = {
				has_template = ZUL_dingane
				is_character_alive = yes
			}
			add_modifier = {
				name = sb_dingane_humbled
				months = 120
			}
			if = {
				limit = {
					NOT = { is_ruler = yes }
				}
				exile_character = yes
			}
		}
	}
	else_if = {
		limit = {
			ruler = {
				has_template = ZUL_mpande_zulu
			}
		}
		sb_zulu_kill_current_heir = yes
		sb_zulu_prepare_cetshwayo_claimant = yes
		sb_zulu_prepare_dinuzulu_heir = yes
	}
	else = {
		sb_zulu_kill_current_heir = yes
		sb_zulu_prepare_dinuzulu_claimant = yes
		sb_zulu_prepare_solomon_heir = yes
	}
}
