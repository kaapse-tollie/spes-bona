# Zulu dynastic stability helpers
#
# Structure:
#   - small additive/removal helpers for the 0-100 dynastic bar
#   - monthly housekeeping that updates malus tiers and opens follow-up content
#   - succession / collapse resolution now lives in
#     `sb_zulu_dynasty_succession_effects.txt`
#
# The bar is intentionally reused by later content:
#   Blood River, Port Natal, and the Swazi question all feed back into the same
#   dynastic-stability story rather than opening disconnected minigames.

sb_zulu_add_dynastic_stability_5 = {
	if = {
		limit = {
			has_journal_entry = je_sb_zulu_kingdom
		}
		je:je_sb_zulu_kingdom ?= {
			add_progress = {
				value = 5
				name = sb_zulu_dynastic_stability_bar
			}
		}
	}
}

sb_boost_firearms_progress_minor = {
	if = {
		limit = {
			has_journal_entry = je_sb_adopt_firearms
			NOT = { has_variable = sb_firearms_modernized_var }
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_treaty_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_treaty_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_industry_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_industry_progress_months_var
				value = 0
			}
		}
		change_variable = {
			name = sb_firearms_treaty_progress_months_var
			add = 2.4
		}
		change_variable = {
			name = sb_firearms_industry_progress_months_var
			add = 2.4
		}
		sb_update_firearms_progress = yes
		sb_sync_iron_age_weapons_penalty = yes
	}
}

sb_zulu_clear_swazi_question_ai_context = {
	if = { limit = { has_variable = sb_zulu_swazi_ai_honour_deal_var } remove_variable = sb_zulu_swazi_ai_honour_deal_var }
	if = { limit = { has_variable = sb_zulu_swazi_ai_guns_south_var } remove_variable = sb_zulu_swazi_ai_guns_south_var }
	if = { limit = { has_variable = sb_zulu_swazi_ai_lost_blood_river_var } remove_variable = sb_zulu_swazi_ai_lost_blood_river_var }
	if = { limit = { has_variable = sb_zulu_swazi_ai_won_blood_river_var } remove_variable = sb_zulu_swazi_ai_won_blood_river_var }
}

sb_zulu_maybe_open_swazi_question = {
	if = {
		limit = {
			sb_is_zulu_dynasty_country = yes
			has_journal_entry = je_sb_zulu_kingdom
			NOT = { has_variable = sb_zulu_swazi_question_opened_var }
			NOT = { has_variable = sb_zulu_swazi_war_active_var }
			OR = {
				has_variable = sb_zulu_swazi_ai_honour_deal_var
				has_variable = sb_zulu_swazi_ai_guns_south_var
				has_variable = sb_zulu_swazi_ai_lost_blood_river_var
				has_variable = sb_zulu_swazi_ai_won_blood_river_var
				has_variable = sb_zulu_blood_river_resolved_var
				has_variable = sb_zulu_port_natal_raid_available_var
				c:ORA ?= {
					has_variable = sb_natal_path_resolved_var
				}
			}
			is_at_war = no
			c:SWZ ?= {
				is_country_alive = yes
				is_subject = no
				any_scope_state = {
					state_region = s:STATE_EAST_TRANSVAAL
				}
			}
			ruler = {
				has_template = ZUL_dingane
			}
		}
		set_variable = sb_zulu_swazi_question_opened_var
		trigger_event = { id = sb_boer_zulu.094 days = 21 popup = yes }
	}
}

sb_zulu_finalize_port_natal_survival = {
	if = {
		limit = {
			has_variable = sb_port_natal_raid_var
		}
		remove_variable = sb_port_natal_raid_var
	}
	if = {
		limit = {
			has_variable = sb_port_natal_british_war_var
		}
		remove_variable = sb_port_natal_british_war_var
	}
	sb_zulu_add_dynastic_stability_20 = yes
}

sb_zulu_enable_dynastic_claims = {
	set_variable = sb_zulu_enforce_dynastic_claims_var
	if = {
		limit = {
			NOT = { has_modifier = sb_zulu_enforcing_dynastic_claims }
		}
		add_modifier = {
			name = sb_zulu_enforcing_dynastic_claims
		}
	}
}

sb_zulu_disable_dynastic_claims = {
	if = {
		limit = { has_variable = sb_zulu_enforce_dynastic_claims_var }
		remove_variable = sb_zulu_enforce_dynastic_claims_var
	}
	if = {
		limit = { has_modifier = sb_zulu_enforcing_dynastic_claims }
		remove_modifier = sb_zulu_enforcing_dynastic_claims
	}
}
