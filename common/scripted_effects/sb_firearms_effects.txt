sb_update_firearms_progress = {
	if = {
		limit = {
			NOT = { has_variable = sb_firearms_treaty_progress_months_var }
		}
		set_variable = {
			name = sb_firearms_treaty_progress_months_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = sb_firearms_industry_progress_months_var }
		}
		set_variable = {
			name = sb_firearms_industry_progress_months_var
			value = 0
		}
	}
	clamp_variable = {
		name = sb_firearms_treaty_progress_months_var
		min = 0
		max = 12
	}
	clamp_variable = {
		name = sb_firearms_industry_progress_months_var
		min = 0
		max = 12
	}

	set_variable = {
		name = sb_firearms_progress_months_var
		value = 0
	}

	if = {
		limit = {
			var:sb_firearms_treaty_progress_months_var >= 12
			var:sb_firearms_progress_months_var < 60
		}
		set_variable = { name = sb_firearms_progress_months_var value = 60 }
	}
	else_if = {
		limit = {
			var:sb_firearms_treaty_progress_months_var >= 11
			var:sb_firearms_progress_months_var < 55
		}
		set_variable = { name = sb_firearms_progress_months_var value = 55 }
	}
	else_if = {
		limit = {
			var:sb_firearms_treaty_progress_months_var >= 10
			var:sb_firearms_progress_months_var < 50
		}
		set_variable = { name = sb_firearms_progress_months_var value = 50 }
	}
	else_if = {
		limit = {
			var:sb_firearms_treaty_progress_months_var >= 9
			var:sb_firearms_progress_months_var < 45
		}
		set_variable = { name = sb_firearms_progress_months_var value = 45 }
