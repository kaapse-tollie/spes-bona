# Shared helper effects for the staged firearms-acquisition JE.
#
# Bar model:
#   - the visible bar runs from 0 to 60
#   - each progress lane tracks months up to 24
#   - the best lane is copied into the master value, then multiplied by 2.5
#     so 24 months becomes 60 visible bar progress
#
# This keeps seeded starts like SWZ/GZA readable while still allowing either
# treaty imports or domestic arms production to complete the JE.

sb_seed_initial_firearms_progress = {
	# Balance seeding:
	#   SWZ starts close to completion for gameplay viability.
	#   GZA starts part-way there.
	if = {
		limit = { country_definition = cd:SWZ }
		set_variable = {
			name = sb_firearms_treaty_progress_months_var
			value = 19.2
		}
		set_variable = {
			name = sb_firearms_industry_progress_months_var
			value = 19.2
		}
	}
	else_if = {
		limit = { country_definition = cd:GZA }
		set_variable = {
			name = sb_firearms_treaty_progress_months_var
			value = 7.2
		}
		set_variable = {
			name = sb_firearms_industry_progress_months_var
			value = 7.2
		}
	}
}

sb_apply_port_natal_raid_firearms_windfall = {
	if = {
		limit = {
			has_journal_entry = je_sb_adopt_firearms
			NOT = { has_variable = sb_firearms_modernized_var }
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_treaty_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_treaty_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_industry_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_industry_progress_months_var
				value = 0
			}
		}
		change_variable = {
			name = sb_firearms_treaty_progress_months_var
			add = 2.4
		}
		change_variable = {
			name = sb_firearms_industry_progress_months_var
			add = 2.4
		}
		sb_update_firearms_progress = yes
		sb_sync_iron_age_weapons_penalty = yes
	}
}

sb_sync_firearms_progress_bar = {
	if = {
		limit = { has_journal_entry = je_sb_adopt_firearms }
		# Preserve country scope explicitly. This helper is called from several
		# event/effect paths, and inside a JE scope `ROOT` is not reliably the
		# owning country.
		save_temporary_scope_as = sb_firearms_country_scope
		je:je_sb_adopt_firearms ?= {
			save_temporary_scope_as = sb_firearms_je_scope
		}
		if = {
			limit = { has_variable = sb_firearms_progress_months_var }
			scope:sb_firearms_je_scope = {
				set_bar_progress = {
					value = scope:sb_firearms_country_scope.var:sb_firearms_progress_months_var
					name = sb_firearms_adoption_bar
				}
			}
		}
		else = {
			scope:sb_firearms_je_scope = {
				set_bar_progress = {
					value = 0
					name = sb_firearms_adoption_bar
				}
			}
		}
	}
}

sb_update_firearms_progress = {
	if = {
		limit = {
			NOT = { has_variable = sb_firearms_treaty_progress_months_var }
		}
		set_variable = {
			name = sb_firearms_treaty_progress_months_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = { has_variable = sb_firearms_industry_progress_months_var }
		}
		set_variable = {
			name = sb_firearms_industry_progress_months_var
			value = 0
		}
	}
	clamp_variable = {
		name = sb_firearms_treaty_progress_months_var
		min = 0
		max = 24
	}
	clamp_variable = {
		name = sb_firearms_industry_progress_months_var
		min = 0
		max = 24
	}

	set_variable = {
		name = sb_firearms_progress_months_var
		value = 0
	}

	if = {
		limit = {
			var:sb_firearms_treaty_progress_months_var >= var:sb_firearms_industry_progress_months_var
		}
		set_variable = {
			name = sb_firearms_progress_months_var
			value = var:sb_firearms_treaty_progress_months_var
		}
	}
	else_if = {
		limit = {
			var:sb_firearms_industry_progress_months_var > var:sb_firearms_treaty_progress_months_var
		}
		set_variable = {
			name = sb_firearms_progress_months_var
			value = var:sb_firearms_industry_progress_months_var
		}
	}

	change_variable = {
		name = sb_firearms_progress_months_var
		multiply = 2.5
	}
	clamp_variable = {
		name = sb_firearms_progress_months_var
		min = 0
		max = 60
	}
	sb_sync_firearms_progress_bar = yes
}
