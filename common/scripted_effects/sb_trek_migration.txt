# Spes Bona - Trek Migration Scripted Effects

# Creates mass migration of Boer pops from Cape Colony to the calling country
sb_trigger_boer_migration_effect = {
	# Find the Cape Colony as migration origin
	random_country = {
		limit = {
			country_definition = cd:CAP
			is_country_alive = yes
		}
		save_scope_as = boer_migration_origin
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_culture = {
					this = cu:boer
				}
			}
			save_scope_as = boer_origin_state
		}
	}
	# Find best target state in the Boer republic
	ordered_scope_state = {
		limit = {
			is_incorporated = yes
		}
		order_by = migration_pull
		position = 0
		check_range_bounds = no
		save_scope_as = boer_migration_target
	}
	# Create the mass migration
	if = {
		limit = {
			exists = scope:boer_migration_origin
			exists = scope:boer_migration_target
		}
		scope:boer_migration_target = {
			create_mass_migration = {
				origin = scope:boer_migration_origin
				culture = cu:boer
			}
		}
		scope:boer_migration_origin = {
			add_modifier = {
				name = sb_trek_migration_cost
				months = 24
			}
		}
	}
}

# Shared Zoutpansberg split setup so event branches can offer a play-as choice
# without duplicating the full spawn / migration / government wiring.
sb_found_zoutpansberg_republic = {
	if = {
		limit = {
			NOT = { exists = c:ZPB }
		}
		create_country = {
			tag = ZPB
			origin = root
			province = p:x5C4C78
		}
	}
	s:STATE_NORTHERN_TRANSVAAL = {
		set_owner_of_provinces = {
			country = c:ZPB
			provinces = {
				x24A528 x2917E2 x336285 x3DA2B5 x3E338E x4760DD x5078AD x5C4C78 x775CAE x7E986B x8CD439 x8DF37C x9EC141
				xB19AFE xC74943 xC9D165 xCAEAEB xD022FF xD34CB1 xD9521C xDABE56 xDB7D32 xF14024 xFF97A1
			}
		}
		add_claim = root
	}
	c:ZPB ?= {
		set_country_type = unrecognized
		sb_apply_boer_republic_spawn_setup = yes
		activate_law = law_type:law_slave_trade
		if = {
			limit = {
				exists = scope:zpb_ruler_scope
			}
			scope:zpb_ruler_scope ?= {
				transfer_character = c:ZPB
				set_interest_group = c:ZPB.ig:ig_armed_forces
				set_as_interest_group_leader = yes
				set_character_as_ruler = yes
				if = {
					limit = {
						NOT = { has_role = general }
					}
					add_character_role = general
				}
				if = {
					limit = {
						NOT = { has_role = politician }
					}
					add_character_role = politician
				}
			}
		}
		else = {
			create_character = {
				template = TRN_stephanus_schoeman
				ruler = yes
				is_general = yes
				ig_leader = yes
				interest_group = ig_armed_forces
			}
		}
		change_relations = {
			country = root
			value = -40
		}
		add_modifier = {
			name = sb_trek_migration_cost
			months = 24
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_NORTHERN_TRANSVAAL
			}
			add_modifier = {
				name = sb_trek_migration_pull
				months = 24
			}
			create_mass_migration = {
				origin = scope:migration_source
				culture = scope:migration_culture
			}
		}
	}
	change_relations = {
		country = c:ZPB
		value = -40
	}
}

sb_found_lydenburg_republic = {
	if = {
		limit = {
			NOT = { exists = c:LYD }
		}
		create_country = {
			tag = LYD
			origin = root
			province = p:x91CA59
		}
	}
	s:STATE_EAST_TRANSVAAL = {
		set_owner_of_provinces = {
			country = c:LYD
			provinces = {
				x50A011 x0591EA x927F5A xDF169D x452304 x972FB2 x73CDE8 x7470A2 xBD0AEB x9037B0 xDC7726
				x91CA59 xB98B42 x6E419C xD02011 xD72273 x25A652
			}
		}
		add_claim = root
	}
	c:LYD ?= {
		capital = {
			save_scope_as = lyd_settler_target
		}
	}
	root.capital ?= {
		every_scope_pop = {
			limit = {
				culture = cu:boer
			}
			move_partial_pop = {
				state = scope:lyd_settler_target
				population_ratio = 0.12
			}
		}
	}
	c:LYD ?= {
		set_country_type = unrecognized
		sb_apply_boer_republic_spawn_setup = yes
		create_character = {
			template = LYD_cornelis_potgieter
			ruler = yes
			is_general = yes
			ig_leader = yes
			interest_group = ig_armed_forces
		}
		change_relations = {
			country = root
			value = -25
		}
		add_modifier = {
			name = sb_trek_migration_cost
			months = 24
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_EAST_TRANSVAAL
			}
			add_modifier = {
				name = sb_trek_migration_pull
				months = 24
			}
			create_mass_migration = {
				origin = scope:migration_source
				culture = scope:migration_culture
			}
		}
	}
	if = {
		limit = {
			country_definition = cd:TRN
			NOT = { has_journal_entry = je_sb_transvaal_unity }
		}
		add_journal_entry = { type = je_sb_transvaal_unity }
	}
	change_relations = {
		country = c:LYD
		value = -25
	}
}
