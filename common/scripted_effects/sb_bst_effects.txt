# Spes Bona - BST frontier, protection, and STA effects

sb_bst_execute_oranje_raid = {
	hidden_effect = {
		set_variable = {
			name = sb_bst_oranje_raid_cooldown_var
			days = 365
		}
	}

	c:ORA ?= {
		random_scope_state = {
			limit = {
				state_region = s:STATE_VRYSTAAT
			}
			kill_population_in_state = {
				value = 10
			}
		}
		add_treasury = -1000
		trigger_event = { id = sb_bst_frontier.010 days = 1 popup = yes }
	}

	add_treasury = 1000
	random_scope_state = {
		limit = {
			OR = {
				state_region = s:STATE_VRYSTAAT
				state_region = s:STATE_DRAKENSBERG
			}
		}
		kill_population_in_state = {
			value = 50
		}
	}
	if = {
		limit = {
			has_journal_entry = je_sb_adopt_firearms
			NOT = { has_variable = sb_firearms_modernized_var }
		}
		custom_tooltip = sb_bst_raid_firearms_progress_tt
		hidden_effect = {
			sb_boost_firearms_progress_raid = yes
		}
	}
}

sb_bst_restore_gbr_protectorate = {
	if = {
		limit = { has_variable = sb_bst_cap_annexation_crisis_var }
		remove_variable = sb_bst_cap_annexation_crisis_var
	}

	c:CAP ?= {
		if = {
			limit = { has_variable = sb_bst_cap_annexation_crisis_var }
			remove_variable = sb_bst_cap_annexation_crisis_var
		}
	}

	if = {
		limit = {
			is_subject_of = c:CAP
		}
		c:GBR ?= {
			transfer_subject = root
		}
	}

	if = {
		limit = {
			NOT = { is_subject_of = c:GBR }
		}
		c:GBR ?= {
			create_diplomatic_pact = {
				country = root
				type = protectorate
			}
		}
	}
	else_if = {
		limit = {
			is_subject_of = c:GBR
			NOT = { is_subject_type = subject_type_protectorate }
		}
		change_subject_type = subject_type_protectorate
	}
}

sb_bst_centralize_tswana_client = {
	c:TSW ?= {
		if = {
			limit = {
				is_country_type = decentralized
			}
			set_country_type = unrecognized
		}
		if = {
			limit = {
				NOT = { is_subject_of = root }
			}
			root = {
				create_diplomatic_pact = {
					country = c:TSW
					type = tributary
				}
			}
		}
		else_if = {
			limit = {
				is_subject_of = root
				NOT = { is_subject_type = subject_type_tributary }
			}
			change_subject_type = subject_type_tributary
		}
		change_relations = {
			country = root
			value = 50
		}
		add_liberty_desire = -50
	}
}
