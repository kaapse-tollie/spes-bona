sb_martinus_create_presidential_union_with_ora = {
	set_variable = sb_martinus_ambition_resolved_var
	set_variable = sb_martinus_presidential_union_active_var
	if = {
		limit = { has_variable = sb_martinus_coercive_chain_active_var }
		remove_variable = sb_martinus_coercive_chain_active_var
	}
	if = {
		limit = { has_variable = sb_martinus_coercive_chain_pending_var }
		remove_variable = sb_martinus_coercive_chain_pending_var
	}
	if = {
		limit = {
			c:ORA ?= {
				is_country_alive = yes
				is_subject = no
			}
		}
		create_diplomatic_pact = {
			country = c:ORA
			type = sb_boer_presidential_union
		}
	}
	c:ORA ?= {
		set_variable = sb_martinus_presidential_union_junior_var
		if = {
			limit = { has_variable = sb_martinus_legal_union_pending_var }
			remove_variable = sb_martinus_legal_union_pending_var
		}
		change_relations = {
			country = c:TRN
			value = 30
		}
	}
}

sb_martinus_ensure_ora_pretorius_candidate = {
	if = {
		limit = {
			NOT = {
				any_scope_character = {
					has_template = ORA_marthinus_wessel_pretorius
				}
			}
		}
		create_character = {
			template = ORA_marthinus_wessel_pretorius
			ig_leader = yes
			interest_group = ig_landowners
			ideology = ideology_sb_volksraad_elitism_leader
		}
	}
	random_scope_character = {
		limit = {
			has_template = ORA_marthinus_wessel_pretorius
			is_character_alive = yes
		}
		if = {
			limit = {
				NOT = { has_role = politician }
			}
			add_character_role = politician
		}
		set_interest_group = c:ORA.ig:ig_landowners
		set_as_interest_group_leader = yes
	}
}

sb_martinus_prepare_ora_pretorius_candidate = {
	sb_martinus_ensure_ora_pretorius_candidate = yes
	add_modifier = {
		name = sb_ora_pretorius_groot_boere_campaign
		months = 36
	}
}

sb_martinus_resolve_legal_union_failure = {
	sb_martinus_restore_ora_historical_groot_boere_leader = yes
	set_variable = sb_martinus_ora_separate_republic_path_var
	if = {
		limit = { has_variable = sb_martinus_legal_union_pending_var }
		remove_variable = sb_martinus_legal_union_pending_var
	}
	add_modifier = {
		name = sb_ora_separate_republic
		months = 120
	}
	c:TRN ?= {
		if = {
			limit = { has_variable = sb_martinus_legal_union_pending_var }
			remove_variable = sb_martinus_legal_union_pending_var
		}
		set_variable = sb_martinus_legal_union_resolved_var
	}
}

sb_martinus_resolve_legal_union_success = {
	if = {
		limit = { has_variable = sb_martinus_legal_union_pending_var }
		remove_variable = sb_martinus_legal_union_pending_var
	}
	c:TRN ?= {
		if = {
			limit = { has_variable = sb_martinus_legal_union_pending_var }
			remove_variable = sb_martinus_legal_union_pending_var
		}
		set_variable = sb_martinus_legal_union_resolved_var
		sb_martinus_create_presidential_union_with_ora = yes
	}
}

sb_martinus_break_presidential_union_with_ora = {
	if = {
		limit = {
			c:ORA ?= {
				is_country_alive = yes
				is_direct_subject_of = root
				is_subject_type = subject_type_sb_boer_presidential_union
			}
		}
		c:ORA ?= {
			make_independent = yes
		}
	}
	if = {
		limit = { has_variable = sb_martinus_presidential_union_active_var }
		remove_variable = sb_martinus_presidential_union_active_var
	}
	c:ORA ?= {
		if = {
			limit = { has_variable = sb_martinus_presidential_union_junior_var }
			remove_variable = sb_martinus_presidential_union_junior_var
		}
	}
}

sb_martinus_create_confederation_with_ora = {
	set_variable = sb_martinus_confederation_active_var
	if = {
		limit = { has_variable = sb_martinus_presidential_union_active_var }
		remove_variable = sb_martinus_presidential_union_active_var
	}
	c:ORA ?= {
		if = {
			limit = {
				is_country_alive = yes
				is_direct_subject_of = c:TRN
				is_subject_type = subject_type_sb_boer_presidential_union
			}
			change_subject_type = subject_type_sb_boer_confederal_partner
		}
		set_variable = sb_martinus_confederal_partner_var
		if = {
			limit = { has_variable = sb_martinus_presidential_union_junior_var }
			remove_variable = sb_martinus_presidential_union_junior_var
		}
	}
	sb_martinus_sync_ora_laws_to_trn = yes
	c:GBR ?= {
		if = {
			limit = {
				sb_martinus_gbr_can_respond_to_confederation = yes
			}
			trigger_event = { id = sb_martinus_confederation.110 days = 14 popup = yes }
		}
	}
}

sb_martinus_revert_confederation_to_presidential_union = {
	if = {
		limit = {
			c:ORA ?= {
				is_country_alive = yes
				is_direct_subject_of = root
				is_subject_type = subject_type_sb_boer_confederal_partner
			}
		}
		c:ORA ?= {
			change_subject_type = subject_type_sb_boer_presidential_union
			if = {
				limit = { has_variable = sb_martinus_confederal_partner_var }
				remove_variable = sb_martinus_confederal_partner_var
			}
			set_variable = sb_martinus_presidential_union_junior_var
		}
	}
	if = {
		limit = { has_variable = sb_martinus_confederation_active_var }
		remove_variable = sb_martinus_confederation_active_var
	}
	set_variable = sb_martinus_presidential_union_active_var
}

sb_martinus_break_gbr_trn_trade_through = {
	c:GBR ?= {
		every_scope_treaty = {
			limit = {
				binds = c:TRN
				any_scope_article = {
					has_type = transit_rights
					source_country = c:GBR
					target_country = c:TRN
				}
			}
			withdraw = {
				country = c:GBR
			}
		}
	}
}

sb_martinus_sync_ora_laws_to_trn = {
	if = {
		limit = {
			c:ORA ?= {
				is_country_alive = yes
				is_direct_subject_of = root
				is_subject_type = subject_type_sb_boer_confederal_partner
			}
		}
		if = {
			limit = { has_law = law_type:law_monarchy }
			c:ORA ?= { activate_law = law_type:law_monarchy }
		}
		if = {
			limit = { has_law = law_type:law_sb_commandant_general_republic }
			c:ORA ?= {
				set_variable = sb_commandant_general_visible_var
				activate_law = law_type:law_sb_commandant_general_republic
			}
		}
		else_if = {
			limit = { has_law = law_type:law_presidential_republic }
			c:ORA ?= { activate_law = law_type:law_presidential_republic }
		}
		if = {
			limit = { has_law = law_type:law_parliamentary_republic }
			c:ORA ?= { activate_law = law_type:law_parliamentary_republic }
		}
		if = {
			limit = { has_law = law_type:law_theocracy }
			c:ORA ?= { activate_law = law_type:law_theocracy }
		}
		if = {
			limit = { has_law = law_type:law_council_republic }
			c:ORA ?= { activate_law = law_type:law_council_republic }
		}
		if = {
			limit = { has_law = law_type:law_ethnostate }
			c:ORA ?= { activate_law = law_type:law_ethnostate }
		}
		if = {
			limit = { has_law = law_type:law_national_supremacy }
			c:ORA ?= { activate_law = law_type:law_national_supremacy }
		}
		if = {
			limit = { has_law = law_type:law_racial_segregation }
			c:ORA ?= { activate_law = law_type:law_racial_segregation }
		}
		if = {
			limit = { has_law = law_type:law_cultural_exclusion }
			c:ORA ?= { activate_law = law_type:law_cultural_exclusion }
		}
		if = {
			limit = { has_law_or_variant = law_type:law_multicultural }
			c:ORA ?= { activate_law = law_type:law_multicultural }
		}
		if = {
			limit = { has_law = law_type:law_slavery_banned }
			c:ORA ?= { activate_law = law_type:law_slavery_banned }
		}
		if = {
			limit = { has_law = law_type:law_debt_slavery }
			c:ORA ?= { activate_law = law_type:law_debt_slavery }
		}
		if = {
			limit = { has_law = law_type:law_slave_trade }
			c:ORA ?= { activate_law = law_type:law_slave_trade }
		}
		if = {
			limit = { has_law = law_type:law_legacy_slavery }
			c:ORA ?= { activate_law = law_type:law_legacy_slavery }
		}
		if = {
			limit = { has_law = law_type:law_peasant_levies }
			c:ORA ?= { activate_law = law_type:law_peasant_levies }
		}
		if = {
			limit = { has_law = law_type:law_professional_army }
			c:ORA ?= { activate_law = law_type:law_professional_army }
		}
		if = {
			limit = { has_law = law_type:law_national_militia }
			c:ORA ?= { activate_law = law_type:law_national_militia }
		}
		if = {
