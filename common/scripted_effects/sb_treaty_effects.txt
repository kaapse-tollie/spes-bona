# Startup treaty helpers for Southern Africa. We keep treaty history additive
# and normalize the relevant vanilla South African bundles at runtime so other
# mods can still extend 00_historical_treaties.txt safely.

sb_normalize_boer_trade_treaties = {
	# GBR-ORA: remove the vanilla guarantee bundle, keep only 25-year transit rights.
	if = {
		limit = {
			c:GBR ?= {
				any_scope_treaty = {
					binds = c:ORA
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:ORA
					}
				}
			}
		}
		c:GBR ?= {
			every_scope_treaty = {
				limit = {
					binds = c:ORA
					any_scope_article = {

						source_country = c:GBR
						target_country = c:ORA
					}
				}
				withdraw = {
					country = c:GBR
				}
			}
			change_infamy = -10
			change_relations = {
				country = c:ORA
				value = 20
			}
		}
	}

	# GBR-TRN: remove the vanilla guarantee bundle, keep only 25-year transit rights,
	# and preserve the intended 20-year truce with Britain.

# Post-trek republican recognition and Zoutpansberg crisis helpers.
sb_ensure_gbr_transit_rights_with_root = {
	save_scope_as = convention_target

	if = {
		limit = {
			NOT = {
				c:GBR ?= {
					any_scope_treaty = {
						binds = scope:convention_target
						any_scope_article = {
							has_type = transit_rights
							source_country = c:GBR
							target_country = scope:convention_target
						}
					}
				}
			}
		}
		c:GBR ?= {
			create_treaty = {
				first_country = c:GBR
				second_country = scope:convention_target
				is_draft = no
					binding_period = { years = 25 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR
						target_country = scope:convention_target
					}
					{
						article = goods_transfer
						source_country = c:GBR
						target_country = scope:convention_target
						inputs = {
							{ goods = g:merchant_marine }
							{ quantity = 10 }
						}
					}
					{
						article = host_power_bloc_embassy
						source_country = scope:convention_target
						target_country = c:GBR
					}
				}
			}
		}
	}
}

sb_ensure_convention_trade_through_with_root = {
	if = {
		limit = {
			c:CAP ?= {
				is_country_alive = yes
				sb_cap_is_self_governing_colony = yes
			}
		}
		sb_ensure_cap_transit_rights_with_root = yes
		sb_ensure_gbr_power_bloc_embassy_with_root = yes
	}
	else = {
		sb_ensure_gbr_transit_rights_with_root = yes
	}
}

sb_ensure_gbr_power_bloc_embassy_with_root = {
	save_scope_as = embassy_target

	if = {

sb_remove_cap_boer_defensive_compact_with_root = {
	save_scope_as = cap_boer_compact_target

	c:CAP ?= {
		every_scope_treaty = {
			limit = {
				binds = scope:cap_boer_compact_target

					OR = {
						has_type = defensive_pact
						has_type = alliance
					}
				}
			}
			withdraw = { country = c:CAP }
