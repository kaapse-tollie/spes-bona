# Startup treaty helpers for Southern Africa. We keep treaty history additive
# and normalize the relevant vanilla South African bundles at runtime so other
# mods can still extend 00_historical_treaties.txt safely.

sb_normalize_boer_trade_treaties = {
	# GBR-ORA: remove the vanilla guarantee bundle, keep only 25-year transit rights.
	if = {
		limit = {
			c:GBR ?= {
				any_scope_treaty = {
					binds = c:ORA
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:ORA
					}
				}
			}
		}
		c:GBR ?= {
			every_scope_treaty = {
				limit = {
					binds = c:ORA
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:ORA
					}
				}
				withdraw = {
					country = c:GBR
				}
			}
			change_infamy = -10
			change_relations = {
				country = c:ORA
				value = 20
			}
		}
	}

	# GBR-TRN: remove the vanilla guarantee bundle, keep only 25-year transit rights,
	# and preserve the intended 20-year truce with Britain.
	if = {
		limit = {
			c:GBR ?= {
				any_scope_treaty = {
					binds = c:TRN
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:TRN
					}
				}
			}
		}
		c:GBR ?= {
			every_scope_treaty = {
				limit = {
					binds = c:TRN
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:TRN
					}
				}
				withdraw = {
					country = c:GBR
				}
			}
			change_infamy = -10
			change_relations = {
				country = c:TRN
				value = 20
			}
		}
	}

	c:ORA ?= {
		create_bidirectional_truce = {
			country = c:GBR
			months = 240
		}
	}

	c:TRN ?= {
		create_bidirectional_truce = {
			country = c:GBR
			months = 240
		}
	}

	# GBR-PHL: remove the vanilla guarantee bundle, keep only 25-year transit rights.
	if = {
		limit = {
			c:GBR ?= {
				any_scope_treaty = {
					binds = c:PHL

				limit = {
					binds = c:PHL
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:PHL
					}
				}
				withdraw = {
					country = c:GBR
				}
			}
			change_infamy = -10
			change_relations = {
				country = c:PHL
				value = 20
			}
		}
		c:PHL ?= {
			end_truce = c:GBR
		}
	}

	# GBR-WBL: replace the vanilla 10-year bundle with a 25-year bundle.
	if = {
		limit = {
			c:GBR ?= {
				any_scope_treaty = {
					binds = c:WBL
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:WBL
					}
				}
			}
		}
		c:GBR ?= {
			every_scope_treaty = {
				limit = {
					binds = c:WBL
					any_scope_article = {
						has_type = guarantee_independence
						source_country = c:GBR
						target_country = c:WBL
					}
				}
				withdraw = {
					country = c:GBR
				}
			}
			change_infamy = -10
			change_relations = {
				country = c:WBL
				value = 20
			}
		}
		c:WBL ?= {
			end_truce = c:GBR
		}
	}

	# ORA-TRN: replace the vanilla 10-year transit rights with a 25-year corridor.
	c:ORA ?= {
		if = {
			limit = {
				any_scope_treaty = {
					binds = c:TRN

	}

	c:GBR ?= {
		if = {
			limit = {
				c:ORA ?= { is_country_alive = yes }
				NOT = {

						binds = c:ORA

				first_country = c:GBR

				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR

					}

		}

			limit = {

				binding_period = { years = 25 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR
						target_country = c:TRN
					}
				}
			}
		}

		if = {
			limit = {
				c:PHL ?= { is_country_alive = yes }
				NOT = {
					any_scope_treaty = {
						binds = c:PHL
						any_scope_article = {
							has_type = transit_rights
							source_country = c:GBR
							target_country = c:PHL
						}
					}
				}
			}
			create_treaty = {
				first_country = c:GBR
				second_country = c:PHL
				is_draft = no
				binding_period = { years = 25 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR
						target_country = c:PHL
					}
				}
			}
		}

		if = {
			limit = {
				c:WBL ?= { is_country_alive = yes }
				NOT = {
					any_scope_treaty = {
						binds = c:WBL
						any_scope_article = {
							has_type = transit_rights
							source_country = c:GBR
							target_country = c:WBL
						}
					}
				}
			}
			create_treaty = {
				first_country = c:GBR
				second_country = c:WBL
				is_draft = no
				binding_period = { years = 25 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR
						target_country = c:WBL
					}
					{
						article = guarantee_independence
						source_country = c:GBR
						target_country = c:WBL
					}
				}
			}
		}
	}

	c:ORA ?= {
		if = {
			limit = {
				c:TRN ?= { is_country_alive = yes }
				NOT = {
					any_scope_treaty = {
						binds = c:TRN
						any_scope_article = {
							has_type = transit_rights
							source_country = c:ORA

			}
			create_treaty = {
				first_country = c:ORA
				second_country = c:TRN

				binding_period = { years = 25 }
				articles_to_create = {

		if = {

		}

		remove_modifier = modifier_untrustworthy_country

# Post-trek republican recognition and Zoutpansberg crisis helpers.
sb_ensure_gbr_transit_rights_with_root = {
	save_scope_as = convention_target

	if = {
		limit = {
			NOT = {
				c:GBR ?= {
					any_scope_treaty = {
						binds = scope:convention_target
						any_scope_article = {
							has_type = transit_rights
							source_country = c:GBR
							target_country = scope:convention_target
						}
					}
				}
			}
		}
		c:GBR ?= {
			create_treaty = {
				first_country = c:GBR
				second_country = scope:convention_target
				is_draft = no
				binding_period = { years = 10 }
				articles_to_create = {
					{
						article = transit_rights
						source_country = c:GBR
						target_country = scope:convention_target
					}
				}
			}
		}
	}
}
