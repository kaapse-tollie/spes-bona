sb_apply_cap_interest_group_identity = {
	ig:ig_petty_bourgeoisie ?= {
		set_interest_group_name = ig_burghers
		if = {
			limit = { has_ideology = ideology:ideology_reactionary }
			remove_ideology = ideology_reactionary
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_burgher_civic } }
			add_ideology = ideology_sb_burgher_civic
		}
		set_ig_trait = ig_trait:sb_ig_trait_civic_grievances
		set_ig_trait = ig_trait:ig_trait_civil_service
		set_ig_trait = ig_trait:ig_trait_treasury_bonds
	}
	ig:ig_devout ?= {
		set_interest_group_name = ig_mission_churches
		if = {
			limit = { has_ideology = ideology:ideology_patriarchal }
			remove_ideology = ideology_patriarchal
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_mission_churches } }
			add_ideology = ideology_sb_mission_churches
		}
		set_ig_trait = ig_trait:sb_ig_trait_mission_disquiet
		set_ig_trait = ig_trait:ig_trait_work_ethic
		set_ig_trait = ig_trait:sb_ig_trait_mission_schools
	}
	ig:ig_rural_folk ?= {
		set_interest_group_name = ig_frontier_farmers
		if = {
			limit = { has_ideology = ideology:ideology_isolationist }
			remove_ideology = ideology_isolationist
		}
		if = {
			limit = { has_ideology = ideology:ideology_agrarian }
			remove_ideology = ideology_agrarian
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_kleinboer_agrarianism } }
			add_ideology = ideology_sb_kleinboer_agrarianism
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_settler_colonialist } }
			add_ideology = ideology_sb_settler_colonialist
		}
		set_ig_trait = ig_trait:ig_trait_born_particularists
		set_ig_trait = ig_trait:ig_trait_honest_work
		set_ig_trait = ig_trait:ig_trait_farmers_markets
	}
	ig:ig_intelligentsia ?= {
		if = {
			limit = { has_ideology = ideology:ideology_liberal }
			remove_ideology = ideology_liberal
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_cape_ig_liberal } }
			add_ideology = ideology_sb_cape_ig_liberal
		}
		set_ig_trait = ig_trait:sb_ig_trait_stirring_the_masses
		set_ig_trait = ig_trait:ig_trait_avant_garde
		set_ig_trait = ig_trait:ig_trait_propagandists
	}
}

sb_apply_boer_republic_interest_group_identity = {
	ig:ig_landowners ?= {
		set_interest_group_name = ig_groot_boere
		if = {
			limit = { has_ideology = ideology:ideology_paternalistic }
			remove_ideology = ideology_paternalistic
		}
		if = {
			limit = { has_ideology = ideology:ideology_republican_paternalistic }
			remove_ideology = ideology_republican_paternalistic
		}
		if = {
			limit = { has_ideology = ideology:ideology_hierarchic }
			remove_ideology = ideology_hierarchic
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_volksraad_elitism } }
			add_ideology = ideology_sb_volksraad_elitism
		}
		if = {
			limit = { has_ideology = ideology:ideology_agrarian }
			remove_ideology = ideology_agrarian
		}

		set_ig_trait = ig_trait:sb_ig_trait_laager_politics
		set_ig_trait = ig_trait:ig_trait_family_ties
		set_ig_trait = ig_trait:sb_ig_trait_republican_notables
	}
	ig:ig_armed_forces ?= {
		set_interest_group_name = ig_kommandante
		if = {
			limit = { has_ideology = ideology:ideology_loyalist }
			remove_ideology = ideology_loyalist
		}
		if = {
			limit = { has_ideology = ideology:ideology_jingoist }
			remove_ideology = ideology_jingoist
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_boer_commando_republicanism } }
			add_ideology = ideology_sb_boer_commando_republicanism
		}
		set_ig_trait = ig_trait:ig_trait_parochial_leadership
		set_ig_trait = ig_trait:ig_trait_veteran_consultation
		set_ig_trait = ig_trait:ig_trait_patriotic_fervor

	ig:ig_devout ?= {
		set_interest_group_name = ig_ng_kerk
		if = {
			limit = { has_ideology = ideology:ideology_moralist }
			remove_ideology = ideology_moralist
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_ng_kerk } }
			add_ideology = ideology_sb_ng_kerk
		}
		set_ig_trait = ig_trait:ig_trait_pious_fiction
		# Override the vanilla Devout happy-tier bonus with a fertility-oriented
		# Boer-specific trait so the NG Kerk reads differently from standard
		# Protestant devout and better matches the desired demographic flavor.
		set_ig_trait = ig_trait:sb_ig_trait_be_fruitful_and_multiply
		set_ig_trait = ig_trait:sb_ig_trait_god_and_volk
	}
	ig:ig_rural_folk ?= {
		set_interest_group_name = ig_klein_boere
		if = {
			limit = { has_ideology = ideology:ideology_isolationist }
			remove_ideology = ideology_isolationist
		}
		if = {
			limit = { has_ideology = ideology:ideology_agrarian }
			remove_ideology = ideology_agrarian
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_kleinboer_agrarianism } }
			add_ideology = ideology_sb_kleinboer_agrarianism
		}
		if = {
			limit = { NOT = { has_ideology = ideology:ideology_sb_settler_colonialist } }
			add_ideology = ideology_sb_settler_colonialist
		}
		set_ig_trait = ig_trait:ig_trait_born_particularists
		set_ig_trait = ig_trait:ig_trait_honest_work
		set_ig_trait = ig_trait:ig_trait_farmers_markets
	}
}
