sb_effect_native_conscription_MTB = {
	if = {
		limit = { has_modifier = sb_native_conscription_MTB }
		remove_modifier = sb_native_conscription_MTB
	}
	if = {
		limit = { has_modifier = sb_native_conscription_MTB_player }
		remove_modifier = sb_native_conscription_MTB_player
	}

	if = {
		limit = {
			is_country_type = decentralized
			c:TRN ?= {
				is_country_alive = yes
				is_player = yes
			}
		}

		add_modifier = {
			name = sb_native_conscription_MTB_player
			months = -1
		}
	}
	else_if = {
		limit = {
			is_country_type = decentralized
		}

		add_modifier = {
			name = sb_native_conscription_MTB
			months = -1
		}
	}
}
