# Spes Bona - Swazi frontier and accommodation helpers
#
# Chain order:
#   1. TRN completes Transvaal Unity
#   2. TRN chooses whether to press the western border claims
#   3. SWZ answers Boer consolidation pressure (cede land or resist)
#   4. only after that, and only if Natal is British, SWZ can choose whether to
#      seek British shelter or stand apart

sb_swazi_raise_zulu_campaign_muster = {
	if = {
		limit = {
			NOT = { has_variable = sb_swazi_zulu_campaign_muster_var }
		}
		set_variable = {
			name = sb_swazi_zulu_campaign_muster_var
			months = 24
		}
		every_scope_state = {
			limit = {
				state_region = s:STATE_EAST_TRANSVAAL
			}
			create_pop = {
				pop_type = officers
				culture = nguni
				size = 600
			}
			create_pop = {
				pop_type = soldiers
				culture = nguni
				size = 6000
			}
			create_building = {
				building = building_barrack
				level = 4
				reserves = 1
			}
		}
		create_military_formation = {
			type = army
			hq_region = sr:region_southern_africa
			name = swazi_emabutfo

			combat_unit = {
				type = unit_type:combat_unit_type_irregular_infantry
				state_region = s:STATE_EAST_TRANSVAAL
				count = 6
			}
		}
	}
}

sb_swazi_apply_guns_for_land_bonus = {
	add_modifier = {
		name = sb_swazi_frontier_rearmed
		months = 60
	}

	if = {
		limit = {
			has_journal_entry = je_sb_adopt_firearms
			NOT = { has_variable = sb_firearms_modernized_var }
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_treaty_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_treaty_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_industry_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_industry_progress_months_var
				value = 0
			}
		}
		if = {
			limit = { var:sb_firearms_treaty_progress_months_var < 24 }
			change_variable = {
				name = sb_firearms_treaty_progress_months_var
				add = 9.6
			}
		}
		if = {
			limit = { var:sb_firearms_industry_progress_months_var < 24 }
			change_variable = {
				name = sb_firearms_industry_progress_months_var
				add = 9.6
			}
		}
		sb_update_firearms_progress = yes
		sb_sync_iron_age_weapons_penalty = yes
	}
}

sb_swazi_apply_boer_buy_time_settlement = {
	# "Buy Time" is deliberately narrow: only the disputed frontier pocket changes
	# hands, and the payoff is security + firearms progress rather than a full
	# subject relationship.
	set_variable = sb_swazi_boer_pressure_resolved_var
	set_variable = sb_swazi_boer_land_sold_var
	if = {
		limit = { has_variable = sb_swazi_boer_resisted_var }
		remove_variable = sb_swazi_boer_resisted_var
	}

	s:STATE_EAST_TRANSVAAL = {
		set_owner_of_provinces = {
			country = c:TRN
			provinces = {
				x73CDE8 x25A652 x50A011 x452304
			}
		}
		remove_claim = c:SWZ
	}

	change_relations = {
		country = c:TRN
		value = 25
	}
	hidden_effect = {
		create_bidirectional_truce = {
			country = c:TRN
			months = 300
		}
	}
	custom_description = {
		text = sb_swzai_frontier_concede_truce
	}
	hidden_effect = {
		sb_swazi_apply_guns_for_land_bonus = yes
	}
	custom_description = {
		text = sb_swzai_frontier_concede_guns
	}



}

sb_trn_apply_swazi_reject_posture = {
	# Attitudes are derived, so make the inputs unambiguous: a rejected frontier
	# settlement leaves TRN hostile and, for AI TRN, actively claim-focused.
	if = {
		limit = {
			c:SWZ ?= { is_country_alive = yes }
		}
		change_relations = {
			country = c:SWZ
			value = -100
		}
	}
	if = {
		limit = { is_ai = yes }
		set_strategy = ai_strategy_sb_boer_frontier_republic
	}
}

sb_swazi_apply_boer_resistance = {
	# Resistance does not immediately force a war; it creates hostility and a
	# grievance window for later pressure instead.
	set_variable = sb_swazi_boer_pressure_resolved_var
	set_variable = sb_swazi_boer_resisted_var
	if = {
		limit = { has_variable = sb_swazi_boer_land_sold_var }
		remove_variable = sb_swazi_boer_land_sold_var
	}

	c:TRN ?= {
		sb_trn_apply_swazi_reject_posture = yes
	}

}

sb_swazi_apply_stand_apart = {
	change_relations = {
		country = c:GBR
		value = -10
	}
	add_modifier = {
		name = sb_swazi_stand_apart
		months = 120
	}
}
