# Shared scripted effects for the Natalia chain.
#
# NAL can now appear either as a Boer republic or as a British crown colony.
# Keep the base tag neutral and apply the correct one-time setup here.

sb_create_natalia_republic_if_missing = {
	if = {
		limit = {
			NOT = { exists = c:NAL }
		}
		create_country = {
			tag = NAL
			origin = root
			province = p:x5B124F
		}
	}
}

sb_assign_natalia_republic_territory = {
	s:STATE_ZULULAND = {
		set_owner_of_provinces = {
			country = c:NAL
			provinces = {
				x5B124F xFF0EF1 x552449 xE0EB02 x85695F xDE0EDE
				x7ACC38 xB1F868 x3CED3D x11A090 xCD31DB
			}
		}
		add_claim = c:NAL
	}
}

sb_apply_natalia_boer_republic_setup = {
	if = {
		limit = { has_variable = southern_africa_subject_var }
		remove_variable = southern_africa_subject_var
	}
	if = {
		limit = { has_modifier = sb_trek_frontier_drive_natalia }
		remove_modifier = sb_trek_frontier_drive_natalia
	}
	effect_starting_technology_tier_3_tech = yes
	effect_starting_politics_conservative = yes
	add_technology_researched = napoleonic_warfare
	set_country_type = unrecognized
	set_next_election_date = 1840.1.1
	if = {
		limit = {
			NOT = { country_has_primary_culture = cu:boer }
		}
		add_primary_culture = cu:boer
	}
	if = {
		limit = {
			country_has_primary_culture = cu:british
		}
		remove_primary_culture = cu:british
	}
	sb_apply_boer_commandant_general_baseline = yes
	activate_law = law_type:law_national_supremacy
	activate_law = law_type:law_elected_bureaucrats
	activate_law = law_type:law_agrarianism
	activate_law = law_type:law_land_based_taxation
	activate_law = law_type:law_homesteading
	activate_law = law_type:law_national_militia
	activate_law = law_type:law_legacy_slavery
	activate_law = law_type:law_no_colonial_affairs
	sb_apply_boer_republic_interest_group_identity = yes
	add_modifier = {
		name = sb_trek_frontier_drive_natalia
		months = 120
	}
	active_law:lawgroup_army_model ?= {
		if = {
			limit = {
				NOT = { has_amendment = amendment_type:amendment_sb_commando_corps }
			}
			add_amendment = {
				type = amendment_sb_commando_corps
				sponsor = PREV.ig:ig_rural_folk
			}
		}
	}
	every_interest_group = {
		remove_ruling_interest_group = yes
	}
	ig:ig_armed_forces ?= { add_ruling_interest_group = yes }
	ig:ig_landowners ?= { add_ruling_interest_group = yes }
	ig:ig_rural_folk ?= { add_ruling_interest_group = yes }
	ig:ig_devout ?= { add_ruling_interest_group = yes }
	sb_refresh_boer_commandant_general_government = yes
	sb_ensure_commandant_general_ruler_command = yes
}

sb_found_natalia_peacefully = {
	random_scope_character = {
		limit = {
			has_template = ORA_piet_retief
		}
		save_scope_as = retief_scope
	}
	sb_create_natalia_republic_if_missing = yes
	sb_assign_natalia_republic_territory = yes
	c:NAL ?= {
		sb_apply_natalia_boer_republic_setup = yes
		change_relations = {
			country = root
			value = 50
		}
		scope:retief_scope ?= {
			transfer_character = c:NAL
			set_character_as_ruler = yes
			add_character_role = general
			add_character_role = politician
			set_interest_group = c:NAL.ig:ig_landowners
		}
		if = {
			limit = {
				NOT = { any_scope_character = { has_template = ORA_piet_retief } }
			}
			create_character = {
				template = ORA_piet_retief
				ruler = yes
				is_general = yes
				interest_group = ig_landowners
			}
		}
		random_scope_character = {
			limit = {
				has_template = ORA_piet_retief
			}
			if = {
				limit = {
					NOT = { has_role = general }
				}
				add_character_role = general
			}
			if = {
				limit = {
					NOT = { has_role = politician }
				}
				add_character_role = politician
			}
			set_interest_group = c:NAL.ig:ig_landowners
		}
		if = {
			limit = {
				NOT = { has_truce_with = c:ZUL }
			}
			create_bidirectional_truce = {
				country = c:ZUL
				months = 60
			}
		}
		create_military_formation = {
			type = army
			hq_region = sr:region_southern_africa
			name = natalia_commando
			combat_unit = {
				type = unit_type:combat_unit_type_line_infantry
				state_region = s:STATE_ZULULAND
				count = 2
			}
			combat_unit = {
				type = unit_type:combat_unit_type_dragoons
				state_region = s:STATE_ZULULAND
				count = 1
			}
		}
	}
	if = {
		limit = {
			NOT = {
				any_scope_treaty = {
					binds = c:NAL
					any_scope_article = {
						has_type = transit_rights
						source_country = c:NAL
						target_country = root
					}
				}
			}
		}
		create_treaty = {
			first_country = c:NAL
			second_country = root
			is_draft = no
			binding_period = { years = 25 }
			articles_to_create = {
				{
					article = transit_rights
					source_country = c:NAL
					target_country = root
				}
			}
		}
	}
	c:NAL ?= {
		sb_trigger_natalia_boer_migration_effect = yes
	}
	set_variable = sb_natal_path_resolved_var
	remove_variable = sb_natal_diplomacy_started_var
}

# Britain can respond to Dingane's coastal raid by annexing the Zulu-held Natal
# littoral outright and reorganizing it as a crown colony.
sb_place_natalia_under_british_protectorate = {
	change_relations = {
		country = c:GBR
		value = 50
	}
	if = {
		limit = {
			NOT = { is_subject_of = c:GBR }
		}
		c:GBR ?= {
			create_diplomatic_pact = {
				country = root
				type = protectorate
			}
		}
	}
	else_if = {
		limit = {
			is_subject_of = c:GBR
			NOT = { is_subject_type = subject_type_protectorate }
			NOT = { is_subject_type = subject_type_colony }
		}
		change_subject_type = subject_type_protectorate
	}
}

sb_convert_natalia_to_british_crown_colony = {
	if = {
		limit = {
			NOT = { has_variable = sb_natalia_british_colony_resolved_var }
		}
		set_variable = sb_natalia_british_colony_resolved_var
	}
	change_relations = {
		country = c:GBR
		value = 50
	}
	# Vanilla colony subject types only remain valid on countries with the
	# colonial country type. Without this, Natalia can briefly become a colony
	# and then immediately drop back out of subjecthood on the next tick.
	if = {
		limit = {
			NOT = { is_country_type = colonial }
		}
		set_country_type = colonial
	}
	if = {
		limit = {
			NOT = { is_subject_of = c:GBR }
		}
		c:GBR ?= {
			create_diplomatic_pact = {
				country = root
				type = colony
			}
		}
	}
	else_if = {
		limit = {
			is_subject_of = c:GBR
			NOT = { is_subject_type = subject_type_colony }
		}
		change_subject_type = subject_type_colony
	}
	sb_trigger_british_natal_colony_migration_effect = yes
}

sb_apply_british_natal_colony_setup = {
	sb_convert_natalia_to_british_crown_colony = yes
	if = {
		limit = {
			has_variable = sb_commandant_general_visible_var
		}
		remove_variable = sb_commandant_general_visible_var
	}
	if = {
		limit = { has_modifier = sb_trek_frontier_drive_natalia }
		remove_modifier = sb_trek_frontier_drive_natalia
	}
	if = {
		limit = {
			NOT = { has_variable = southern_africa_subject_var }
		}
		set_variable = southern_africa_subject_var
	}
	activate_law = law_type:law_colonial_administration
	activate_law = law_type:law_professional_army
	activate_law = law_type:law_slavery_banned
	active_law:lawgroup_army_model ?= {
		if = {
			limit = {
				has_amendment = amendment_type:amendment_sb_commando_corps
			}
			every_scope_amendment = {
				limit = {
					type = amendment_type:amendment_sb_commando_corps
				}
				remove_amendment = yes
			}
		}
	}
	if = {
		limit = {
			NOT = { country_has_primary_culture = cu:british }
		}
		add_primary_culture = cu:british
	}
	if = {
		limit = {
			country_has_primary_culture = cu:boer
		}
		remove_primary_culture = cu:boer
	}
	s:STATE_ZULULAND = {
		region_state:NAL ?= {
			set_state_type = unincorporated
		}
	}
	sb_install_british_natal_governor = yes
}

sb_select_natalia_refugee_receiver = {
	random_country = {
		limit = {
			is_country_alive = yes
			OR = {
				country_definition = cd:ORA
				country_definition = cd:TRN
				country_definition = cd:ZPB
				country_definition = cd:LYD
			}
		}
		save_scope_as = natal_refugee_receiver
	}
}

			}
		}

			limit = {

			}

		limit = {
			any_scope_treaty = {
				binds = c:TRN
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:TRN
			}
			withdraw = {
				country = root
			}
		}
	}
	if = {
		limit = {
			exists = c:ZPB
			any_scope_treaty = {
				binds = c:ZPB
			}
		}
		every_scope_treaty = {
			limit = {
				binds = c:ZPB
			}

	if = {

sb_exile_natalia_boer_leadership = {
	if = {
		limit = {
			NOT = { exists = scope:natal_refugee_receiver }
		}
		sb_select_natalia_refugee_receiver = yes
	}
	random_scope_character = {
		limit = {
			has_template = TRN_andries_pretorius
		}
		save_scope_as = natal_exiled_leader
	}
	if = {
		limit = {
			NOT = { exists = scope:natal_exiled_leader }
			exists = ruler
			ruler = {
				culture = cu:boer
			}
		}
		ruler = {
			save_scope_as = natal_exiled_leader
		}
	}
	scope:natal_exiled_leader ?= {
		if = {
			limit = {
				NOT = { has_role = politician }
			}
			add_character_role = politician
		}
		transfer_character = scope:natal_refugee_receiver
		set_interest_group = scope:natal_refugee_receiver.ig:ig_landowners
	}
}

# Add a Boer ally directly to Natalia's side once Britain turns the ultimatum
# into a live diplomatic play.
sb_join_natalia_against_britain = {
	random_diplomatic_play = {
		limit = {
			is_diplomatic_play_type = dp_make_protectorate
			any_scope_play_involved = { c:GBR ?= THIS }
			any_scope_play_involved = { c:NAL ?= THIS }
		}
		add_target_backers = { root }
	}
}

# Support Natalia against Britain with both a live war join and a short-term
# defensive pact, so the diplomatic relationship survives beyond the first war.
sb_support_natalia_against_britain = {
	c:NAL ?= {
		save_scope_as = natalia
	}
	if = {
		limit = {
			NOT = {
				any_scope_treaty = {
					binds = root
					binds = scope:natalia
					any_scope_article = {
						OR = {
							has_type = defensive_pact
							has_type = alliance
						}
					}
				}
			}
		}
		create_treaty = {
			first_country = root
			second_country = scope:natalia
			is_draft = no
			binding_period = { years = 5 }
			articles_to_create = {
				{
					article = defensive_pact
				}
			}
		}
	}
	sb_join_natalia_against_britain = yes
}

# Boer refugees from Natal move inland toward whichever republic receives them.
sb_trigger_natal_refugee_migration_effect = {
	c:NAL ?= {
		save_scope_as = natalia_source
	}
	cu:boer = {
		save_scope_as = migration_culture
	}
	ordered_scope_state = {
		limit = {
			is_incorporated = yes
		}
		order_by = migration_pull
		position = 0
		check_range_bounds = no
		save_scope_as = migration_target
	}
	if = {
		limit = {
			exists = scope:natalia_source
			exists = scope:migration_target
		}
		scope:migration_target = {
			create_mass_migration = {
				origin = scope:natalia_source
				culture = scope:migration_culture
			}
		}
		c:NAL.capital ?= {
			every_scope_pop = {
				limit = {
					culture = cu:boer
				}
				move_partial_pop = {
					state = scope:migration_target
					population_ratio = 0.20
				}
			}
		}
	}
}

# Natalia needs its own trekker migration helper because the generic trek effect
# targets the calling country, which is ORA in the formation chain.
sb_trigger_natalia_boer_migration_effect = {
	random_country = {
		limit = {
			country_definition = cd:CAP
			is_country_alive = yes
		}
		save_scope_as = boer_migration_origin
	}
	c:NAL ?= {
		ordered_scope_state = {
			order_by = migration_pull
			position = 0
			check_range_bounds = no
			save_scope_as = boer_migration_target
		}
	}
	if = {
		limit = {
			exists = scope:boer_migration_origin
			exists = scope:boer_migration_target
		}
		scope:boer_migration_target = {
			create_mass_migration = {
				origin = scope:boer_migration_origin
				culture = cu:boer
			}
		}
		scope:boer_migration_origin = {
			add_modifier = {
				name = sb_trek_migration_cost
				months = 24
			}
		}
	}
}
