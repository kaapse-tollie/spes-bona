# Shared scripted effects for the Natalia republican / Blood River chain.
#
# NAL can now appear either as:
#   - a Boer republic created from ORA's Blood River victory
#   - a British-organized crown colony after the later coast crisis
#
# V2 note:
#   Blood River itself is ORA-only in fresh starts. Some TRN-related helpers are
#   intentionally retained below as compatibility fallbacks for older saves or
#   partial branches that still have a live TRN participant.
#
# File split:
#   - this file owns republic creation, Blood River support, and Natalia's Boer
#     founding path
#   - British submission / colony conversion / refugee fallout lives in
#     `sb_natalia_colony_effects.txt`

sb_create_natalia_republic_if_missing = {
	if = {
		limit = {
			NOT = { exists = c:NAL }
		}
		create_country = {
			tag = NAL
			origin = root
			province = p:x5B124F
		}
	}
}

sb_assign_natalia_republic_territory = {
	s:STATE_ZULULAND = {
		set_owner_of_provinces = {
			country = c:NAL
			provinces = {
				x5B124F xFF0EF1 x552449 xE0EB02 x85695F xDE0EDE
				x7ACC38 xB1F868 x3CED3D x11A090 xCD31DB
			}
		}
		add_claim = c:NAL
	}
}

sb_apply_natalia_boer_republic_setup = {
	c:ORA ?= {
		if = {
			limit = { has_variable = sb_voortrekker_start_var }
			remove_variable = sb_voortrekker_start_var
		}
	}
	if = {
		limit = { has_variable = southern_africa_subject_var }
		remove_variable = southern_africa_subject_var
	}
	if = {
		limit = { has_variable = sb_natalia_british_colony_resolved_var }
		remove_variable = sb_natalia_british_colony_resolved_var
	}
	if = {
		limit = { has_modifier = sb_trek_frontier_drive_natalia }
		remove_modifier = sb_trek_frontier_drive_natalia
	}
	effect_starting_technology_tier_3_tech = yes
	effect_starting_politics_conservative = yes
	add_technology_researched = napoleonic_warfare
	set_country_type = unrecognized
	set_next_election_date = 1840.1.1
	if = {
		limit = {
			NOT = { country_has_primary_culture = cu:boer }
		}
		add_primary_culture = cu:boer
	}
	if = {
		limit = {
			country_has_primary_culture = cu:british
		}
		remove_primary_culture = cu:british
	}
	sb_apply_boer_commandant_general_baseline = yes
	activate_law = law_type:law_national_supremacy
	activate_law = law_type:law_elected_bureaucrats
	activate_law = law_type:law_agrarianism
	activate_law = law_type:law_land_based_taxation
	activate_law = law_type:law_homesteading
	activate_law = law_type:law_national_militia
	activate_law = law_type:law_legacy_slavery
	activate_law = law_type:law_no_colonial_affairs
	sb_apply_boer_republic_interest_group_identity = yes
	if = {
		limit = {
			country_has_primary_culture = cu:boer
			NOT = { has_journal_entry = je_sb_great_trek }
		}
		add_journal_entry = { type = je_sb_great_trek }
	}
	active_law:lawgroup_army_model ?= {
		if = {
			limit = {
				NOT = { has_amendment = amendment_type:amendment_sb_commando_corps }
			}
			add_amendment = {
				type = amendment_sb_commando_corps
				sponsor = PREV.ig:ig_rural_folk
			}
		}
	}
	every_interest_group = {
		remove_ruling_interest_group = yes
	}
	ig:ig_armed_forces ?= { add_ruling_interest_group = yes }
	ig:ig_landowners ?= { add_ruling_interest_group = yes }
	ig:ig_rural_folk ?= { add_ruling_interest_group = yes }
	ig:ig_devout ?= { add_ruling_interest_group = yes }
	sb_refresh_boer_commandant_general_government = yes
	sb_ensure_commandant_general_ruler_command = yes
}

sb_ensure_trn_ai_natal_front_commitment = {
	# Legacy compatibility path for pre-V2 saves where TRN still joined the Natal
	# war. Fresh campaigns should normally never hit this.
	if = {
		limit = {
			country_definition = cd:TRN
			is_ai = yes
			has_variable = sb_trn_joined_natal_war_var
			is_at_war = yes
		}
		if = {
			limit = {
				NOT = { has_variable = sb_trn_natal_relief_force_recent_var }
				NOT = {
					c:ORA ?= {
						is_country_alive = yes
						is_ai = yes
					}
				}
			}
			set_variable = {
				name = sb_trn_natal_relief_force_recent_var
				months = 12
			}
			create_military_formation = {
				type = army
				hq_region = sr:region_southern_africa
				name = pretorius_relief_commando

				combat_unit = {
					type = unit_type:combat_unit_type_dragoons
					state_region = s:STATE_TRANSVAAL
					count = 2
				}
				combat_unit = {
					type = unit_type:combat_unit_type_line_infantry
					state_region = s:STATE_TRANSVAAL
					count = 1
				}
			}
		}
		every_military_formation = {
			limit = {
				is_army = yes
			}
			fully_mobilize_army = yes
		}
		if = {
			limit = {
				any_scope_war = {
					any_scope_front = {
						always = yes
					}
				}
			}
			random_scope_war = {
				random_scope_front = {
					save_temporary_scope_as = sb_natal_front_scope
				}
			}
			every_military_formation = {
				limit = {
					is_army = yes
					num_units >= 1
				}
				teleport_to_front = {
					front = scope:sb_natal_front_scope
					base_camp = p:x04FA29
				}
			}
		}
	}
}

sb_raise_ora_blood_river_relief_force = {
	# Standard AI-vs-AI ORA relief package for Blood River.
	if = {
		limit = {
			country_definition = cd:ORA
			NOT = { has_variable = sb_ora_blood_river_relief_force_var }
		}
		set_variable = {
			name = sb_ora_blood_river_relief_force_var
			months = 24
		}
		add_modifier = {
			name = sb_blood_river_emergency_mobilization
			months = 12
		}
		sb_seed_ora_blood_river_relief_manpower = yes
		every_scope_state = {
			limit = {
				state_region = s:STATE_VRYSTAAT
			}
			if = {
				limit = {
					NOT = { has_building = building_barrack }
				}
				create_building = {
					building = building_barrack
					level = 5
					reserves = 1
				}
			}
			# Avoid an unbounded while-loop here. The play-started relief code runs
			# under diplomatic-play root, and if building scope resolution gets odd
			# the loop can keep re-evaluating forever. A fixed top-up ladder is more
			# verbose but far safer.
			else_if = {
				limit = { b:building_barrack.level == 1 }
				create_building = {
					building = building_barrack
					level = 4
					reserves = 1
				}
			}
			else_if = {
				limit = { b:building_barrack.level == 2 }
				create_building = {
					building = building_barrack
					level = 3
					reserves = 1
				}
			}
			else_if = {
				limit = { b:building_barrack.level == 3 }
				create_building = {
					building = building_barrack
					level = 2
					reserves = 1
				}
			}
			else_if = {
				limit = { b:building_barrack.level == 4 }
				create_building = {
					building = building_barrack
					level = 1
					reserves = 1
				}
			}
		}
		if = {
			limit = { army_size <= 0 }
			create_military_formation = {
				type = army
				hq_region = sr:region_southern_africa
				name = potgieter_relief_commando

				combat_unit = {
					type = unit_type:combat_unit_type_line_infantry
					state_region = s:STATE_VRYSTAAT
					count = 3
				}
				combat_unit = {
					type = unit_type:combat_unit_type_dragoons
					state_region = s:STATE_VRYSTAAT
					count = 2
				}
			}
		}
		else_if = {
			limit = { army_size == 1 }
			create_military_formation = {
				type = army
				hq_region = sr:region_southern_africa
				name = potgieter_relief_commando

				combat_unit = {
					type = unit_type:combat_unit_type_line_infantry
					state_region = s:STATE_VRYSTAAT
					count = 3
				}
				combat_unit = {
					type = unit_type:combat_unit_type_dragoons
					state_region = s:STATE_VRYSTAAT
					count = 1
				}
			}
		}
	}
}

			first_country = root

sb_join_trn_to_active_natal_war = {
	if = {
		limit = {
			c:TRN ?= {
				is_country_alive = yes
				is_ai = yes
				NOT = { has_variable = sb_trn_joined_natal_war_var }
			}
			c:ORA ?= {
				is_country_alive = yes
				has_variable = sb_natal_war_active_var
			}
			c:ZUL ?= {
				is_country_alive = yes
			}
		}
		if = {
			limit = {
				NOT = {
					c:TRN ?= {
						any_scope_treaty = {
							binds = c:ORA
							any_scope_article = {
								OR = {
									has_type = defensive_pact
									has_type = alliance
								}
							}
						}
					}
				}
			}
			c:TRN ?= {
				create_treaty = {
					first_country = c:TRN
					second_country = c:ORA
					is_draft = no
					binding_period = { years = 5 }
					articles_to_create = {
						{
							article = defensive_pact
						}
					}
				}
			}
		}
		c:TRN ?= {
			set_variable = sb_trn_joined_natal_war_var
			change_relations = {
				country = c:ORA
				value = 20
			}
		}
		c:ORA ?= {
			set_variable = sb_trn_joined_natal_war_var
			change_relations = {
				country = c:TRN
				value = 20
			}
		}
		random_diplomatic_play = {
			limit = {
				initiator = c:ZUL
				target = c:ORA
				is_diplomatic_play_type = dp_sb_natal_crisis
				NOT = {
					any_scope_play_involved = {
						c:TRN ?= THIS
					}
				}
			}
			add_target_backers = { c:TRN }
		}
	}
}
