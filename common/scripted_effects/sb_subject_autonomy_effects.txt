# Shared subject-autonomy helpers for mid-century constitutional content.

sb_grant_responsible_government_effect = {
	if = {
		limit = {
			is_subject_type = subject_type_colony
		}
		change_subject_type = subject_type_sb_responsible_colony
	}
	if = {
		limit = {
			NOT = { has_modifier = sb_responsible_government }
		}
		add_modifier = {
			name = sb_responsible_government
			months = -1
		}
	}
	add_liberty_desire = -15
	change_relations = {
		country = scope:actor
		value = 15
	}
	set_variable = sb_responsible_government_granted_var
}
