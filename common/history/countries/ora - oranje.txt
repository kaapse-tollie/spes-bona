# Orange Free State (ORA) — Boer republic between the Orange and Vaal rivers.
# Historically, the Orange River Sovereignty was a British protectorate
# (1848-1854) before London withdrew and the Orange Free State
# (Oranje-Vrystaat) was proclaimed. At the 1836 start, Voortrekker settlement
# in the region is only beginning.
# ORA shares most of TRN's setup as a Boer frontier republic. The key
# difference is that ORA does not begin with frontier colonization, since its
# geography is more constrained by the Cape, Basutoland, and the Griqua polities.
COUNTRIES = {
	c:ORA ?= {
		set_variable = sb_commandant_general_visible_var
		set_variable = sb_early_republic_var

		# Election date: 1839, mirroring the early Volksraad rhythm used for TRN.
		set_next_election_date = 1839.10.12

		# Tier 3 tech: same rationale as TRN, frontier knowledge without an
		# industrial or bureaucratic base on the Cape's level.
		effect_starting_technology_tier_3_tech = yes
		# Conservative politics: same Calvinist Boer social structure as TRN.
		effect_starting_politics_conservative = yes
		# Commandant-General Republic: Boer frontier variant of Presidential
		# Republic that keeps the elected Volksraad structure while allowing the
		# ruler to serve as field commandant.
		activate_law = law_type:law_sb_commandant_general_republic
		# Oligarchy: the early trekker Volksraad is still a narrow burgher elite
		# rather than a settled constitutional electorate.
		activate_law = law_type:law_oligarchy
		# National Supremacy: racially exclusive polity, same as TRN.
		activate_law = law_type:law_national_supremacy
		# Elected Bureaucrats: local officials elected by burghers, same as TRN.
		activate_law = law_type:law_elected_bureaucrats
		# Agrarianism: pastoral economy on the highveld grasslands.
		activate_law = law_type:law_agrarianism
		# Land-Based Taxation: farm quitrents as primary revenue.
		activate_law = law_type:law_land_based_taxation
		# Homesteading: settler land claims, same pattern as TRN.
		activate_law = law_type:law_homesteading
		# National Militia: Boer commando system.
			activate_law = law_type:law_national_militia
			active_law:lawgroup_army_model ?= {
				if = {
					limit = {
						NOT = { has_amendment = amendment_type:amendment_sb_commando_corps }
					}
					add_amendment = {
						type = amendment_sb_commando_corps
						sponsor = PREV.ig:ig_rural_folk
					}
				}
			}
		# Legacy Slavery: same indentured-labour practices as TRN.
		activate_law = law_type:law_legacy_slavery
		# No frontier colonization at start: ORA is boxed in far more tightly than
		# TRN and should not begin with the same open-ended colonial horizon.
		# TRN has it because of open expansion into the northern highveld.

		# Broad-but-coherent start coalition: enough of the trekker camp must sit
		# in government to avoid the ahistorical zero-legitimacy open, but the
		# republic should still read as a commandant's laager rather than a full
		# national front.
		ig:ig_landowners ?= {
			add_ruling_interest_group = yes
		}
		ig:ig_rural_folk ?= {
			add_ruling_interest_group = yes
		}
		ig:ig_armed_forces ?= {
			add_ruling_interest_group = yes
		}
		ig:ig_devout ?= {
			add_ruling_interest_group = yes
		}

		# Replace the vanilla Highveld JE with the mod's Great Trek content.
		add_journal_entry = { type = je_sb_great_trek }
	}
}
