# Thembu (XHT) — Xhosa-adjacent chiefdom in the eastern Cape hinterland
# Historically: The Thembu (abaThembu) were a Xhosa-speaking polity east of the Gcaleka,
# in what is now the interior Transkei. Though culturally Xhosa, the Thembu maintained a
# distinct political identity under their own paramount chiefs. They were less directly
# affected by the Frontier Wars than the Rharhabe but were drawn into Cape politics
# through missionary contacts and trade. The Thembu royal house later produced Nelson Mandela.
# Split from vanilla XHO. Identical institutional setup to XHG and XHR.
# See also: xhg - gcaleka xhosa.txt, xhr - rharhabe xhosa.txt
COUNTRIES = {
	c:XHT ?= {
		# Tier 6 tech: pre-industrial pastoral society, same as XHG and XHR.
		effect_starting_technology_tier_6_tech = yes
		# Native conscription 4: Xhosa-style warrior militia system.
		effect_native_conscription_4 = yes
		# Traditional politics: chieftaincy under the Thembu paramount.
		effect_starting_politics_traditional = yes
		# Debt Slavery: same regional practice as other Xhosa polities.
		activate_law = law_type:law_debt_slavery
	}
}
