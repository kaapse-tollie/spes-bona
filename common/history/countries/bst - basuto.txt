# Basutoland (BST) — Kingdom of the Basotho under Moshoeshoe I
# Historically: Moshoeshoe I (r. c.1822-1870) unified Sotho-speaking refugees of the Mfecane
# into the Basotho nation, centred on the mountain fortress of Thaba Bosiu. A brilliant diplomat,
# he played Boers, British, and neighbouring polities against each other to preserve Basotho
# independence. Basutoland became a British protectorate in 1868 (modern Lesotho).
# This file modifies vanilla BST by removing je_struggle_for_the_highveld.
COUNTRIES = {
	c:BST ?= {
		# Tier 6 tech: pre-industrial society, but Moshoeshoe was open to external knowledge.
		effect_starting_technology_tier_6_tech = yes
		effect_starting_politics_traditional = yes
		# International Trade: Moshoeshoe traded with Cape merchants and missionaries from the 1830s.
		add_technology_researched = international_trade
		# Rationalism: Paris Evangelical Missionary Society (PEMS) influence from 1833.
		# Moshoeshoe invited missionaries (Casalis, Arbousset) for diplomatic and educational purposes.
		add_technology_researched = rationalism

		# Subjecthood: subjects of the king (morena e moholo), not citizens.
		activate_law = law_type:law_subjecthood
		# Traditionalism: economy based on cattle and agriculture under chiefly redistribution.
		activate_law = law_type:law_traditionalism
		# Freedom of Conscience: Moshoeshoe was religiously tolerant — welcomed missionaries
		# without mandating Christianity. Unusual among indigenous rulers of the period.
		activate_law = law_type:law_freedom_of_conscience
		# Land-Based Taxation: tribute from sub-chiefs based on harvests and cattle.
		activate_law = law_type:law_land_based_taxation
		# Slave Trade: vanilla law — reflects regional practice of captive incorporation.
		activate_law = law_type:law_slave_trade

		# Vanilla Highveld JE replaced — BST has no P1 JE.
		# BST's story is driven by regional events and interactions with ORA/CAP.
	}
}
