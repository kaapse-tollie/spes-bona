# Rharhabe Xhosa (XHR) — junior house of the Xhosa, on the Cape frontier
# Historically: The Rharhabe were the western branch of the Xhosa, living between the Fish and
# Kei rivers (the Ciskei). They bore the brunt of nine Frontier Wars against the Cape Colony
# (1779-1879). Under chief Maqoma and regent Sandile at game start. The Rharhabe were the
# most directly threatened by Cape expansion — their territory was progressively annexed
# into "British Kaffraria" (1847) and the Cape Colony.
# Split from vanilla XHO. Identical institutional setup to XHG and XHT.
# See also: xhg - gcaleka xhosa.txt, xht - thembu.txt
COUNTRIES = {
	c:XHR ?= {
		# Tier 6 tech: pre-industrial pastoral society, same as XHG.
		effect_starting_technology_tier_6_tech = yes
		# Native conscription 4: Xhosa warrior militia system.
		effect_native_conscription_4 = yes
		# Traditional politics: chieftaincy system, though increasingly disrupted by Cape interference.
		effect_starting_politics_traditional = yes
		# Debt Slavery: same regional practice of captive incorporation as XHG.
		activate_law = law_type:law_debt_slavery
	}
}
