# Philippolis (PHL) — Griqua captaincy of Adam Kok III
# Historically: Philippolis was the seat of the Griqua captaincy under Adam Kok II (to 1835)
# and then Adam Kok III (r. 1837-1861). The Griqua were mixed Khoikhoi-European descendants
# who established semi-autonomous polities along the Orange River. They were Afrikaans-speaking,
# Christian (London Missionary Society influence), and organised under elected captains.
# Philippolis sat on the route between the Cape Colony and the interior, giving it trade access.
# This file modifies vanilla PHL by removing je_struggle_for_the_highveld.
COUNTRIES = {
	c:PHL ?= {
		# Tier 6 tech: baseline for indigenous polities. Additional techs below bring PHL above baseline.
		effect_starting_technology_tier_6_tech = yes
		effect_starting_politics_traditional = yes
		# International Trade: Griqua were active middlemen in Cape-interior trade (ivory, cattle, guns).
		add_technology_researched = international_trade
		# Rationalism: missionary influence brought literacy and some Enlightenment ideas.
		add_technology_researched = rationalism
		# Bureaucracy: the Griqua captaincy had written laws and a rudimentary civil administration.
		add_technology_researched = tech_bureaucracy

		# Oligarchy: Griqua governance was semi-democratic — captain elected by council of elders.
		# More structured than pure chieftaincy but not a full republic.
		activate_law = law_type:law_oligarchy
		# National Supremacy: Griqua identity was exclusive; non-Griqua had limited political rights.
		activate_law = law_type:law_national_supremacy
		# Slavery Banned: the LMS missionaries opposed slavery; Griqua laws prohibited it.
		activate_law = law_type:law_slavery_banned
		# Homesteading: Griqua allocated farmland to members on a first-come basis.
		activate_law = law_type:law_homesteading
		# State Religion: London Missionary Society (Protestant Christianity) deeply embedded in governance.
		activate_law = law_type:law_state_religion
		# Local Police: Griqua maintained mounted patrols to enforce order.
		activate_law = law_type:law_local_police
		# Religious Schools: mission schools were the primary education system.
		activate_law = law_type:law_religious_schools

		# Schools lvl 1: LMS mission schools provided basic literacy.
		set_institution_investment_level = {
			institution = institution_schools
			level = 1
		}

		# Police lvl 1: Griqua commando patrols served as law enforcement.
		set_institution_investment_level = {
			institution = institution_police
			level = 1
		}

		# Spes Bona: Removed je_struggle_for_the_highveld (disabled by mod).
	}
}
