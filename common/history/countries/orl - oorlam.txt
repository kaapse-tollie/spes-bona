# Oorlam (ORL) — Jonker Afrikaner's Oorlam polity in the Windhoek area
# Historically: The Oorlam were Khoi/mixed-heritage groups who migrated north from the Cape Colony
# in the late 18th/early 19th century. Jonker Afrikaner (c.1790-1861) established dominance over
# the Windhoek basin by the 1830s, raiding Herero and Nama groups with firearms acquired from
# Cape traders. Culturally Griqua — Afrikaans-speaking, partially Christianised, but with
# indigenous social structures.
# The Oorlam are defined with Griqua culture in the country definition files.
COUNTRIES = {
	c:ORL = {
		# Tier 6 tech: decentralized frontier raider polity with strong commando
		# organization, but limited formal state institutions.
		effect_starting_technology_tier_6_tech = yes
		# International Trade: Oorlam firearms, horses, and ammunition moved
		# through Cape-linked exchange networks.
		add_technology_researched = international_trade
		# Native conscription 4: reflects the Oorlam commando system — mounted raiders with firearms.
		# Shared setup with XHG, XHR, XHT. See also: common/history/countries/xhg, xhr, xht files.
		effect_native_conscription_4 = yes
		# Traditional politics: chieftaincy-based governance under Jonker Afrikaner.
		effect_starting_politics_traditional = yes
		# Debt Slavery: reflects the Oorlam practice of incorporating captives from raids as bonded labour.
		activate_law = law_type:law_debt_slavery
		# Isolationism: trade remained opportunistic and frontier-bound rather than
		# channeled through a durable state trade regime.
		activate_law = law_type:law_isolationism
	}
