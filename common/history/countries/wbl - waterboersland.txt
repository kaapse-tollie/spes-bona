# Waterboersland (WBL) — Griqua captaincy of the Waterboer family at Griquatown.
# Historically: Andries Waterboer (r. 1820-1852) led the western Griqua from Griquatown
# (modern Griekwastad). Like Philippolis, a LMS-influenced Griqua polity, but located further
# west along the middle Orange River. Waterboer was known for maintaining order and cooperating
# with the Cape Colony — later, the diamond discoveries at Kimberley (1867) fell within
# Waterboer's claimed territory, making WBL strategically significant.
# Similar setup to PHL but with fewer institutional investments — Waterboersland was
# smaller and less developed than Philippolis.
# This override removes the vanilla Highveld JE and preserves WBL as the
# western Griqua pole for later diamond and northern-Cape content.
COUNTRIES = {
	c:WBL ?= {
		# Tier 6 tech: baseline for indigenous polities. Additional techs below bring WBL above baseline.
		effect_starting_technology_tier_6_tech = yes
		effect_starting_politics_traditional = yes
		# International Trade: Griqua trade networks with the Cape Colony.
		add_technology_researched = international_trade
		# Rationalism: LMS missionary influence, same as PHL.
		add_technology_researched = rationalism
		# Bureaucracy: written laws and appointed officials under the Waterboer captaincy.
		add_technology_researched = tech_bureaucracy

		# National Supremacy: Griqua ethnic exclusivity in governance.
		activate_law = law_type:law_national_supremacy
		# Slavery Banned: LMS anti-slavery influence, same as PHL.
		activate_law = law_type:law_slavery_banned
		# Homesteading: Griqua land allocation system.
		activate_law = law_type:law_homesteading
		# State Religion: LMS Christianity central to governance.
		activate_law = law_type:law_state_religion
		# WBL lacks PHL's oligarchy, local police, and religious schools laws,
		# and has no institution investments. This reflects Waterboersland being smaller
		# and less institutionally developed than Philippolis under Adam Kok III.
	}
}
