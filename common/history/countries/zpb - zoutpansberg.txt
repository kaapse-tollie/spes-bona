# Zoutpansberg (ZPB) — northern Boer trek polity
# This is not a start-country. It exists so the late-1840s northern Transvaal
# split can create a recognizable Boer rival without falling back on generic
# breakaway logic.
COUNTRIES = {
	c:ZPB ?= {
		set_variable = sb_commandant_general_visible_var
		set_next_election_date = 1840.1.1

		# Same trekker knowledge base as the parent Transvaal polity, including
		# stronger frontier command culture than the raw tier-3 bundle provides.
		effect_starting_technology_tier_3_tech = yes
		effect_starting_politics_conservative = yes
		activate_law = law_type:law_sb_commandant_general_republic
		activate_law = law_type:law_oligarchy
		activate_law = law_type:law_national_supremacy
		activate_law = law_type:law_elected_bureaucrats
		activate_law = law_type:law_agrarianism
		activate_law = law_type:law_land_based_taxation
		activate_law = law_type:law_homesteading
		activate_law = law_type:law_national_militia
		activate_law = law_type:law_slave_trade
		activate_law = law_type:law_no_colonial_affairs
	}
}
