# Gcaleka Xhosa (XHG) — senior house of the Xhosa kingdom
# Historically: The Gcaleka were the paramount chieftaincy of the Xhosa nation, ruling east of
# the Kei River (the Transkei). Under Hintsa kaKhawuta (d. 1835) and then Sarhili (r. 1835-1892).
# Hintsa was killed by the British during the Sixth Frontier War (1835), making the Gcaleka
# deeply hostile to the Cape Colony. The Gcaleka later instigated the cattle-killing movement (1856-57).
# XHG is split from vanilla XHO (generic Xhosa) alongside XHR (Rharhabe) and XHT (Thembu).
# This split reflects the historically decentralised nature of Xhosa politics — the three
# chiefdoms often acted independently and sometimes fought each other.
# See also: xhr - rharhabe xhosa.txt, xht - thembu.txt (identical setup, different territories)
COUNTRIES = {
	c:XHG ?= {
		# Tier 6 tech: pre-industrial pastoral society, same baseline as other indigenous polities.
		effect_starting_technology_tier_6_tech = yes
		# Native conscription 4: reflects the Xhosa warrior tradition and militia-based defence.
		# Same conscription effect as ORL (Oorlam) — both could field armed forces relative to population.
		effect_native_conscription_4 = yes
		# Traditional politics: chieftaincy under the Great House lineage.
		effect_starting_politics_traditional = yes
		# Debt Slavery: captives from frontier conflicts and intra-Xhosa disputes were incorporated
		# as dependents. Standard for the region.
		activate_law = law_type:law_debt_slavery
	}
}
