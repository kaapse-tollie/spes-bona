	c:ORL = {

		# raids and tribute extraction. Known for his direct, forceful leadership.

			ruler = yes
			birth_date = 1790.1.1
			interest_group = ig_landowners
			ideology = ideology_moderate
			traits = {
				direct persistent
			}
		}
		# Jan Jonker Afrikaner (c.1820-1889) — Jonker's son and successor
