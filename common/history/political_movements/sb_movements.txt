POLITICAL_MOVEMENTS = {
	c:CAP ?= {
		create_political_movement = {
			type = movement_sb_eastern_cape_separatist
		}
		create_political_movement = {
			type = movement_sb_cape_liberal
		}

		# Vanilla startup history force-creates a generic cultural-majority movement
		# for almost every recognized country before our custom Cape JE logic takes
		# over. Keep it suppressed from day one so the player only sees the bespoke
		# Cape political struggle while the JE is active.
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_cultural_majority
			}
			set_suppression = yes
		}

		# Likewise, the generic Minority Rights Movement overlaps with the bespoke
		# Cape Liberal lane. Keep the vanilla movement suppressed from day one so
		# the player sees one coherent liberal reform bloc rather than two parallel
		# movements chasing the same politics.
		random_political_movement = {
			limit = {
				is_political_movement_type = movement_minority_rights
			}
			set_suppression = yes
		}
	}
}
