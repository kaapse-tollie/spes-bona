# Spes Bona - Flag Definitions
# CAP / ABY intentionally use a narrow, self-contained British-subject ladder:
#   - CAP-local colony flag while lesser subjects
#   - CAP-local blue ensign-style dominion flag while higher-tier subjects
#   - bespoke full-texture flags once fully independent
# This keeps the colonial transition readable without inheriting unrelated
# ideological variants or shared naming from other South African flag paths.
# XHG/XHR/XHT/ORL are decentralized — V3 auto-generates flags, but we define bases.

CAP = { # Cape Colony
	flag_definition = {
		coa = CAP
		subject_canton = CAP
		# Keep the independent Cape flag above vanilla's Southern Africa colonial
		# fallback (DEFAULT priority 100) if the colonial helper variable lingers.
		priority = 200
	}
	flag_definition = {
		coa = CAP_subject_colony
		priority = 320
		trigger = {
			coa_def_lesser_subject_trigger = yes
		}
	}
	flag_definition = {
		coa = CAP_subject_dominion
		priority = 310
		trigger = {
			coa_def_subject_trigger = yes
			NOT = { coa_def_lesser_subject_trigger = yes }
		}
	}
}
