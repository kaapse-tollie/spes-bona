# Spes Bona - Boer governance variants
# Keep these as true parented law variants so the republics remain inside the
# vanilla presidential-republic family for AI logic and scripted checks.

law_sb_commandant_general_republic = {
	group = lawgroup_governance_principles
	parent = law_presidential_republic

	icon = "gfx/interface/icons/law_icons/presidential_republic.dds"

	progressiveness = 75

	is_visible = {
		country_has_primary_culture = cu:boer
		has_variable = sb_commandant_general_visible_var
	}

	modifier = {
		country_legitimacy_headofstate_add = 10
		country_legitimacy_govt_total_votes_add = 10
		country_legitimacy_ideological_incoherence_mult = 0.2
		country_authority_add = 100
		country_farmers_pol_str_mult = 0.10
		country_officers_pol_str_mult = 0.10
	}
}

# Spes Bona compatibility override:
# After the Transvaal split, trekker frontier republics no longer satisfy the
# vanilla homeland/decentralized-state shape for frontier colonization, even
# though they should still be allowed to retain and use the law in this mod's
# Southern African setup.
REPLACE:law_frontier_colonization = {
	group = lawgroup_colonization

	can_enact = {
		OR = {
			AND = {
				country_is_colonial_or_company = yes
				any_decentralized_country = {
					is_adjacent_to_country = root
				}
			}
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
				any_neighbouring_state = {
					owner = {
						is_country_type = decentralized
					}
				}
			}
			AND = {
				country_has_primary_culture = cu:boer
				OR = {
					country_definition = cd:TRN
					country_definition = cd:ZPB
					country_definition = cd:LYD
					country_definition = cd:NAL
				}
			}
		}
	}

	icon = "gfx/interface/icons/law_icons/frontier_colonization.dds"

	progressiveness = 0

	unlocking_technologies = {
		colonization
	}

	modifier = {
		state_migration_pull_unincorporated_mult = 0.5
	}

	institution = institution_colonial_affairs

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_colonization_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}

	ai_will_do = {
		NOR = {
			has_law_or_variant = law_type:law_colonial_exploitation
			has_law_or_variant = law_type:law_colonial_resettlement
		}
	}

	ai_enact_weight_modifier = {
		value = 0
	}

	ai_impose_chance = {
		value = 0

		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_colonial_resettlement
					has_law = law_type:law_frontier_colonization
				}
				scope:target_country = {
					has_law_or_variant = law_type:law_no_colonial_affairs
				}
				power_bloc ?= { has_principle_group = principle_group:principle_group_colonial_offices }
			}
			add = 10
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= this
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}
