# Spes Bona - Xhosa frontier shell
#
# This is the first live scaffold for the later Frontier Wars lane. For now it
# simply opens once the mid-century frontier remains unresolved and keeps the
# conflict visible until the Xhosa resistance network is broken.

je_sb_xhosa_wars = {
	group = je_group_historical_content

	icon = "gfx/interface/icons/je_icons/sb_je_cape_politics.dds"

	possible = {
		OR = {
			country_definition = cd:CAP
			country_definition = cd:ABY
		}
		game_date >= 1846.1.1
		has_variable = sb_xhosa_wars_shell_opened_var
		any_scope_state = {
			state_region = s:STATE_EASTERN_CAPE
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					OR = {
						c:XHG ?= { is_country_alive = yes }
						c:XHR ?= { is_country_alive = yes }
						c:XHT ?= { is_country_alive = yes }
					}
				}
				desc = je_sb_xhosa_wars_status_live
			}
			desc = je_sb_xhosa_wars_status_quiet
		}
	}

	complete = {
		OR = {
			NOT = {
				OR = {
					c:XHG ?= { is_country_alive = yes }
					c:XHR ?= { is_country_alive = yes }
					c:XHT ?= { is_country_alive = yes }
				}
			}
			NOT = {
				any_scope_state = {
					state_region = s:STATE_EASTERN_CAPE
					has_state_trait = state_trait_sb_xhosa_resistance
				}
			}
		}
	}

	weight = 850
	should_be_pinned_by_default = yes
	transferable = no
}
