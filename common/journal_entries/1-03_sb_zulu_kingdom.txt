# Spes Bona - Zulu dynastic stability JE
# ==============================================================================
# This JE tracks whether the post-Shaka royal order survives:
#   - the Natal / Blood River crisis
#   - the coastal Port Natal / British intervention lane
#   - the later Dingane-era Swazi pressure release valve
# ==============================================================================

je_sb_zulu_kingdom = {
	group = je_group_historical_content

	icon = "gfx/interface/icons/invention_icons/dynamite.dds"

	scripted_progress_bar = sb_zulu_dynastic_stability_bar

	is_shown_in_lobby = {
		sb_is_zulu_dynasty_country = yes
		NOT = { has_variable = sb_zulu_dynasty_secured_var }
	}

	is_shown_when_inactive = {
		sb_is_zulu_dynasty_country = yes
		NOT = { has_variable = sb_zulu_dynasty_secured_var }
	}

	possible = {
		sb_is_zulu_dynasty_country = yes
		NOT = { has_variable = sb_zulu_dynasty_secured_var }
	}

	scripted_button = je_sb_zulu_enforce_dynastic_claims_button
	scripted_button = je_sb_zulu_relax_dynastic_claims_button

	immediate = {
		if = {
			limit = {
				NOT = { has_variable = sb_zulu_dynastic_stability_seeded_var }
			}
			set_variable = sb_zulu_dynastic_stability_seeded_var
			je:je_sb_zulu_kingdom ?= {
				set_bar_progress = {
					value = 50
					name = sb_zulu_dynastic_stability_bar
				}
			}
		}
		sb_zulu_update_dynastic_stability_modifiers = yes
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_sb_zulu_kingdom_status_secure
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" >= 75
					}
				}
			}
			triggered_desc = {
				desc = je_sb_zulu_kingdom_status_steady
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" >= 50
					}
				}
			}
			triggered_desc = {
				desc = je_sb_zulu_kingdom_status_crisis
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" <= 15
					}
				}
			}
			triggered_desc = {
				desc = je_sb_zulu_kingdom_status_precarious
				trigger = {
					scope:journal_entry = {
						"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" <= 35
					}
				}
			}
			desc = je_sb_zulu_kingdom_status
		}
	}

	complete = {
		custom_tooltip = {
			text = je_sb_zulu_kingdom_complete_dynasty
			scope:journal_entry = {
				"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" >= 100
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			sb_zulu_monthly_dynasty_housekeeping = yes
			if = {
				limit = {
					NOT = { has_variable = sb_zulu_dynastic_collapse_pending_var }
					scope:journal_entry = {
						"scripted_bar_progress(sb_zulu_dynastic_stability_bar)" <= 0
					}
				}
				set_variable = sb_zulu_dynastic_collapse_pending_var
				trigger_event = { id = sb_boer_zulu.090 days = 1 popup = yes }
			}
		}
	}

	on_complete = {
		trigger_event = { id = sb_boer_zulu.093 days = 1 popup = yes }
		remove_variable = sb_zulu_dynastic_collapse_pending_var
		sb_zulu_disable_dynastic_claims = yes
		sb_zulu_clear_dynastic_stability_modifiers = yes
	}

	on_invalid = {
		remove_variable = sb_zulu_dynastic_collapse_pending_var
		sb_zulu_clear_dynastic_stability_modifiers = yes
	}

	weight = 1000
	should_be_pinned_by_default = yes
}
