# Generic firearms-acquisition JE for southern African kingdoms that still
# rely heavily on melee-centric warfare at game start.
#
# Two progress lanes run in parallel:
#   - imports / treaty access to firearms
#   - domestic industry capable of producing them
# Whichever lane is ahead becomes the visible bar progress.

je_sb_adopt_firearms = {
	group = je_group_historical_content

	icon = "gfx/interface/icons/event_icons/event_military.dds"
	scripted_progress_bar = sb_firearms_adoption_bar

	is_shown_in_lobby = {
		hidden_trigger = {
			sb_is_firearms_acquisition_country = yes
			NOT = { has_variable = sb_firearms_modernized_var }
		}
	}

	is_shown_when_inactive = {
		hidden_trigger = {
			sb_is_firearms_acquisition_country = yes
			NOT = { has_variable = sb_firearms_modernized_var }
		}
	}

	possible = {
		hidden_trigger = {
			sb_is_firearms_acquisition_country = yes
			NOT = { has_variable = sb_firearms_modernized_var }
		}
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_sb_adopt_firearms_status_both
				trigger = {
					sb_has_active_firearms_import_treaty = yes
					sb_has_qualifying_firearms_industry = yes
				}
			}
			triggered_desc = {
				desc = je_sb_adopt_firearms_status_treaty
				trigger = {
					sb_has_active_firearms_import_treaty = yes
				}
			}
			triggered_desc = {
				desc = je_sb_adopt_firearms_status_industry
				trigger = {
					sb_has_qualifying_firearms_industry = yes
				}
			}
			desc = je_sb_adopt_firearms_status_no_progress
		}
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_treaty_progress_months_var }
			}
			if = {
				limit = {
					has_variable = sb_firearms_progress_months_var
				}
				set_variable = {
					name = sb_firearms_treaty_progress_months_var
					value = var:sb_firearms_progress_months_var
				}
			}
			else = {
				set_variable = {
					name = sb_firearms_treaty_progress_months_var
					value = 0
				}
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_industry_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_industry_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_progress_months_var }
			}
			set_variable = {
				name = sb_firearms_progress_months_var
				value = 0
			}
		}
		if = {
			limit = {
				NOT = { has_variable = sb_firearms_je_seeded_var }
			}
			sb_seed_initial_firearms_progress = yes
			set_variable = sb_firearms_je_seeded_var
		}
		sb_update_firearms_progress = yes
		sb_sync_iron_age_weapons_penalty = yes
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = { has_variable = sb_firearms_modernized_var }
					sb_has_active_firearms_import_treaty = yes
					var:sb_firearms_treaty_progress_months_var < 24
				}
				change_variable = {
					name = sb_firearms_treaty_progress_months_var
					add = 1
				}
			}
			if = {
				limit = {
					NOT = { has_variable = sb_firearms_modernized_var }
					sb_has_qualifying_firearms_industry = yes
					var:sb_firearms_industry_progress_months_var < 24
				}
				change_variable = {
					name = sb_firearms_industry_progress_months_var
					add = 1
				}
			}
			sb_update_firearms_progress = yes
			sb_sync_iron_age_weapons_penalty = yes
		}
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = je_sb_adopt_firearms_completion_treaty_tt
				var:sb_firearms_treaty_progress_months_var >= 24
			}
			custom_tooltip = {
				text = je_sb_adopt_firearms_completion_industry_tt
				var:sb_firearms_industry_progress_months_var >= 24
			}
		}
	}

	on_complete = {
		set_variable = sb_firearms_modernized_var
		sb_remove_iron_age_weapons_modifiers = yes
		remove_variable = sb_firearms_treaty_progress_months_var
		remove_variable = sb_firearms_industry_progress_months_var
		remove_variable = sb_firearms_progress_months_var
		trigger_event = { id = sb_firearms.001 }
	}

	weight = 1000
	should_be_pinned_by_default = yes
}
