						scope:target_country = {
							any_scope_state = {
								has_claim_by = ROOT
							}
						}
					}
					add = 300
				}
			}

		return_state = 3.0
	}

						has_claim_by = ROOT
					}

				add = 500
			}
		}
		return_state = {
			if = {
				limit = {
					scope:target_state = {
						has_claim_by = ROOT
					}
				}
				add = 750
			}
		}
	}

	}

	# to be far stronger than vanilla's broad "modernizer" weights.

			country_definition = cd:ZPB
		}
		has_law_or_variant = law_type:law_frontier_colonization
	}

	weight = {
		value = 1000
	}
}
