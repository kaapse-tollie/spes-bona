# Gcaleka Xhosa — The senior house of the Xhosa kingdom, based east of the Kei River.
# Led by Paramount Chief Hintsa (d. 1835) and then Sarhili during the game's timeframe.
# The Gcaleka were the "Great House" from which Xhosa paramount authority derived.
# country_type = decentralized: Xhosa polities lacked centralized European-style state structures.
# tier = principality: appropriate for a decentralized chieftaincy.
# Capital in Eastern Cape because the Gcaleka heartland lay in the Transkei region.
# Cross-ref: common/state_traits/sb_state_traits.txt — state_trait_sb_xhosa_resistance slows colonization here.
XHG = { # Gcaleka Xhosa
	color = { 88 80 163 }

	country_type = decentralized

	tier = principality

	cultures = { xhosa }
	capital = STATE_EASTERN_CAPE
}

# Rharhabe Xhosa — The junior house of the Xhosa, west of the Kei River.
# Bore the brunt of the Cape Frontier Wars (9 wars, 1779-1879) as they bordered the colony directly.
# Led by chiefs such as Maqoma and Sandile during the 1830s-1850s.
# Color is a lighter shade of the same purple hue as XHG to visually link them as Xhosa kin.
# Shares STATE_EASTERN_CAPE capital with XHG — V3 handles multi-tag states via decentralized mechanics.
XHR = { # Rharhabe Xhosa
	color = { 108 100 183 }

	country_type = decentralized

	tier = principality

	cultures = { xhosa }
	capital = STATE_EASTERN_CAPE
}

# Darkest purple of the three Xhosa tags to create a visual gradient on the map.
