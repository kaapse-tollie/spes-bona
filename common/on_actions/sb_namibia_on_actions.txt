sb_on_namibia_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				country_definition = cd:GER
				game_date >= 1870.1.1
				game_date <= 1875.12.31
				NOT = { has_variable = sb_nam_colonial_prestige_resolved_var }
				NOT = { has_law = law_type:law_frontier_colonization }
				NOT = { has_law = law_type:law_colonial_resettlement }
				NOT = { has_law = law_type:law_colonial_exploitation }
				OR = {
					ig:ig_landowners ?= { is_in_government = yes }
					ig:ig_armed_forces ?= { is_in_government = yes }
				}
			}
			trigger_event = { id = sb_nam.010 popup = yes }
		}

		if = {
			limit = {
				country_definition = cd:GER
				game_date >= 1883.1.1
				has_variable = sb_nam_angra_pequena_purchase_pending_var
				NOT = { has_variable = sb_nam_angra_pequena_purchased_var }
				OR = {
					has_law = law_type:law_colonial_resettlement
					has_law = law_type:law_colonial_exploitation
				}
				NOT = {
					any_scope_state = {
						state_region = s:STATE_NAMAQUALAND
						owner = root
						num_provinces >= 1
					}
				}
				p:xDEAA3C.state.owner = {
					is_country_type = decentralized
				}
				c:CAP ?= {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_NAMAQUALAND
							owner = prev
							num_provinces >= 2
						}
					}
				}
			}
			trigger_event = { id = sb_nam.011 popup = no }
		}

		if = {
				limit = {
					country_definition = cd:CAP
					game_date >= 1867.1.1
					game_date <= 1871.12.31
					NOT = { has_variable = sb_nam_rehoboth_branch_resolved_var }
					OR = {
						sb_nam_has_any_colonial_law = yes
					has_modifier = sb_cape_favour_settlers
				}
			}
			trigger_event = { id = sb_nam.020 popup = yes }
		}

		if = {
			limit = {
				p:xDEAA3C.state.owner = root
				NOT = { is_country_type = decentralized }
				NOT = { has_claim = s:STATE_NAMAQUALAND }
				c:CAP ?= {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_NAMAQUALAND
							owner = prev
							num_provinces >= 2
						}
					}
				}
			}
			s:STATE_NAMAQUALAND = {
				add_claim = root
			}
		}

		if = {
			limit = {
				NOT = { has_journal_entry = je_sb_consolidation_of_south_west_africa }
				sb_nam_can_run_consolidation_je = yes
			}
			sb_nam_add_consolidation_je_and_seed = yes
		}

		if = {
			limit = {
				OR = {
					has_variable = sb_nam_final_harshness_10_var
					has_variable = sb_nam_final_harshness_7_var
					has_variable = sb_nam_final_harshness_5_var
				}
				NOT = { has_variable = sb_nam_forced_camps_fired_var }
				any_political_movement = {
					is_political_movement_type = movement_minority_rights
					political_movement_radicalism >= 0.30
					political_movement_radicalism < 0.75
				}
				OR = {
					has_law = law_type:law_colonial_exploitation
					has_law = law_type:law_national_supremacy
					has_law = law_type:law_ethnostate
				}
			}
			trigger_event = { id = sb_nam.200 popup = yes }
		}

		if = {
			limit = {
				OR = {
					has_variable = sb_nam_final_harshness_10_var
					has_variable = sb_nam_final_harshness_7_var
				}
				NOT = { has_variable = sb_nam_extermination_fired_var }
				any_political_movement = {
					is_political_movement_type = movement_minority_rights
					political_movement_radicalism >= 0.75
				}
				OR = {
					has_law = law_type:law_colonial_exploitation
					has_law = law_type:law_national_supremacy
					has_law = law_type:law_ethnostate
				}
			}
			trigger_event = { id = sb_nam.201 popup = yes }
		}
	}
}
