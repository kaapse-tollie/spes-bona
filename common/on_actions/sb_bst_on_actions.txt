# Spes Bona - BST frontier and Cape transfer hooks

on_monthly_pulse_country = {
	on_actions = { sb_on_bst_monthly_pulse_country }
}

on_wargoal_enforced = {
	on_actions = { sb_on_bst_wargoal_enforced }
}

on_war_end = {
	on_actions = { sb_on_bst_war_end }
}

on_diplo_play_back_down = {
	on_actions = { sb_on_bst_diplo_play_back_down }
}

sb_on_bst_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				country_definition = cd:ORA
				has_journal_entry = je_sb_bst_ora_frontier
				NOT = { has_variable = sb_bst_frontier_claim_seeded_var }
				c:BST ?= {
					sb_bst_controls_only_drakensberg = yes
				}
			}
			set_variable = sb_bst_frontier_claim_seeded_var
			s:STATE_DRAKENSBERG = {
				add_claim = c:ORA
			}
		}
		if = {
			limit = {
				country_definition = cd:ORA
				has_journal_entry = je_sb_bst_ora_frontier
				has_variable = sb_bst_frontier_claim_seeded_var
				NOT = {
					c:BST ?= {
						sb_bst_controls_only_drakensberg = yes
					}
				}
			}
			remove_variable = sb_bst_frontier_claim_seeded_var
			s:STATE_DRAKENSBERG = {
				remove_claim = c:ORA
			}
		}

		if = {
			limit = {
				country_definition = cd:BST
				has_journal_entry = je_sb_bst_ora_frontier
				sb_bst_controls_only_drakensberg = yes
				NOT = { has_variable = sb_bst_into_the_mountains_var }
			}
			trigger_event = { id = sb_bst_frontier.020 days = 1 popup = yes }
		}

		if = {
			limit = {
				country_definition = cd:BST
				is_ai = yes
				is_country_alive = yes
				is_subject = no
				has_journal_entry = je_sb_bst_ora_frontier
				sb_bst_controls_only_drakensberg = yes
				sb_has_adjacent_british_protection_route = yes
				NOT = { has_variable = sb_bst_gbr_protection_request_resolved_var }
				game_date >= 1868.5.1
				c:GBR ?= {
					is_country_alive = yes
				}
			}
			set_variable = sb_bst_gbr_protection_request_resolved_var
			c:GBR ?= {
				trigger_event = { id = sb_bst_frontier.100 popup = yes }
			}
		}

		if = {
			limit = {
				country_definition = cd:BST
				is_country_alive = yes
				is_subject = no
				owns_entire_state_region = STATE_VRYSTAAT
				NOT = { has_journal_entry = je_sb_sotho_tswana_nation }
				NOT = { has_variable = sb_sotho_tswana_nation_completed_var }
			}
			add_journal_entry = {
				type = je_sb_sotho_tswana_nation
			}
		}

		if = {
			limit = {
				country_definition = cd:ORA
				NOT = { has_variable = sb_ora_annexed_bst_var }
				NOT = { has_variable = sb_settled_drakensberg_var }
				NOT = {
					c:BST ?= {
						is_country_alive = yes
					}
				}
				owns_entire_state_region = STATE_DRAKENSBERG
			}
			set_variable = sb_ora_annexed_bst_var
		}

		if = {
			limit = {
				country_definition = cd:ORA
				has_variable = sb_ora_annexed_bst_var
				NOT = { has_journal_entry = je_sb_settle_drakensberg }
				NOT = { has_variable = sb_settled_drakensberg_var }
				NOT = {
					c:BST ?= {
						is_country_alive = yes
					}
				}
			}
			add_journal_entry = {
				type = je_sb_settle_drakensberg
			}
		}

		if = {
			limit = {
				country_definition = cd:GBR
				c:BST ?= {
					has_variable = sb_bst_cap_transfer_pending_var
				}
				OR = {
					NOT = { c:BST ?= { is_country_alive = yes } }
					NOT = { c:BST ?= { is_subject_of = c:GBR } }
					NOT = { c:BST ?= { is_subject_type = subject_type_protectorate } }
					NOT = {
						c:CAP ?= {
							is_country_alive = yes
							is_subject_of = c:GBR
							OR = {
								is_subject_type = subject_type_dominion
								is_subject_type = subject_type_sb_dominion
								is_subject_type = subject_type_sb_responsible_colony
								is_subject_type = subject_type_sb_responsible_colony_monarchy
							}
							sb_cap_controls_bst_frontier = yes
						}
					}
				}
			}
			c:BST ?= {
				remove_variable = sb_bst_cap_transfer_pending_var
			}
		}

		if = {
			limit = {
				country_definition = cd:GBR
				c:BST ?= {
					is_country_alive = yes
					is_subject_of = c:GBR
					is_subject_type = subject_type_protectorate
					NOT = { has_variable = sb_bst_cap_transfer_resolved_var }
					NOT = { has_variable = sb_bst_cap_transfer_pending_var }
				}
				c:CAP ?= {
					is_country_alive = yes
					is_subject_of = c:GBR
					OR = {
						is_subject_type = subject_type_dominion
						is_subject_type = subject_type_sb_dominion
						is_subject_type = subject_type_sb_responsible_colony
						is_subject_type = subject_type_sb_responsible_colony_monarchy
					}
					sb_cap_controls_bst_frontier = yes
				}
			}
			c:BST ?= {
				set_variable = sb_bst_cap_transfer_pending_var
			}
			random_list = {
				1 = { trigger_event = { id = sb_bst_frontier.200 days = 730 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.200 days = 1095 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.200 days = 1460 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.200 days = 1825 popup = yes } }
			}
		}

		if = {
			limit = {
				country_definition = cd:CAP
				c:BST ?= {
					has_variable = sb_bst_cap_disarmament_pending_var
				}
				OR = {
					is_at_war = yes
					NOT = { aggressive_diplomatic_plays_permitted = yes }
					NOT = { sb_cap_has_bst_subject_for_disarmament = yes }
				}
			}
			c:BST ?= {
				remove_variable = sb_bst_cap_disarmament_pending_var
			}
		}

		if = {
			limit = {
				country_definition = cd:CAP
				is_at_war = no
				aggressive_diplomatic_plays_permitted = yes
				sb_cap_has_bst_subject_for_disarmament = yes
				NOT = { has_variable = sb_bst_cap_annexation_crisis_var }
				c:BST ?= {
					NOT = { has_variable = sb_bst_cap_annexation_crisis_var }
					NOT = { has_variable = sb_bst_cap_disarmament_pending_var }
					NOT = { has_variable = sb_bst_cap_disarmament_resolved_var }
				}
			}
			c:BST ?= {
				set_variable = sb_bst_cap_disarmament_pending_var
			}
			random_list = {
				1 = { trigger_event = { id = sb_bst_frontier.220 days = 365 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.220 days = 730 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.220 days = 1095 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.220 days = 1460 popup = yes } }
				1 = { trigger_event = { id = sb_bst_frontier.220 days = 1825 popup = yes } }
			}
		}
	}
}

sb_on_bst_wargoal_enforced = {
	effect = {
		if = {
			limit = {
				country_definition = cd:ORA
				scope:target ?= {
					country_definition = cd:BST
				}
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_annex_war
				}
				NOT = { has_variable = sb_settled_drakensberg_var }
			}
			set_variable = sb_ora_annexed_bst_var
		}

		if = {
			limit = {
				country_definition = cd:CAP
				scope:target ?= {
					country_definition = cd:BST
				}
				scope:diplomatic_play ?= {
					OR = {
						is_diplomatic_play_type = dp_annex_subject
						is_diplomatic_play_type = dp_sb_basotho_gun_war
					}
				}
				has_variable = sb_bst_cap_annexation_crisis_var
			}
			remove_variable = sb_bst_cap_annexation_crisis_var
		}
	}
}

sb_on_bst_war_end = {
	effect = {
		if = {
			limit = {
				scope:actor ?= {
					country_definition = cd:CAP
					has_variable = sb_bst_cap_annexation_crisis_var
				}
				scope:target ?= {
					country_definition = cd:BST
					has_variable = sb_bst_cap_annexation_crisis_var
					is_country_alive = yes
				}
			}
			scope:target = {
				sb_bst_restore_gbr_protectorate = yes
			}
		}
		if = {
			limit = {
				scope:actor ?= {
					country_definition = cd:BST
					has_variable = sb_bst_cap_annexation_crisis_var
					is_country_alive = yes
				}
				scope:target ?= {
					country_definition = cd:CAP
					has_variable = sb_bst_cap_annexation_crisis_var
				}
			}
			scope:actor = {
				sb_bst_restore_gbr_protectorate = yes
			}
		}
	}
}

sb_on_bst_diplo_play_back_down = {
	effect = {
		if = {
			limit = {
				root = {
					is_diplomatic_play_type = dp_annex_war
				}
				scope:actor ?= {
					country_definition = cd:ORA
					NOT = { has_variable = sb_settled_drakensberg_var }
				}
				scope:target ?= {
					country_definition = cd:BST
				}
			}
			scope:actor = {
				set_variable = sb_ora_annexed_bst_var
			}
		}

		if = {
			limit = {
				root = {
					OR = {
						is_diplomatic_play_type = dp_annex_subject
						is_diplomatic_play_type = dp_sb_basotho_gun_war
					}
				}
				scope:actor ?= {
					country_definition = cd:CAP
					has_variable = sb_bst_cap_annexation_crisis_var
				}
				scope:target ?= {
					country_definition = cd:BST
					has_variable = sb_bst_cap_annexation_crisis_var
					is_country_alive = yes
				}
			}
			scope:target = {
				sb_bst_restore_gbr_protectorate = yes
			}
		}
		if = {
			limit = {
				root = {
					OR = {
						is_diplomatic_play_type = dp_annex_subject
						is_diplomatic_play_type = dp_sb_basotho_gun_war
					}
				}
				scope:actor ?= {
					country_definition = cd:BST
					has_variable = sb_bst_cap_annexation_crisis_var
					is_country_alive = yes
				}
				scope:target ?= {
					country_definition = cd:CAP
					has_variable = sb_bst_cap_annexation_crisis_var
				}
			}
			scope:actor = {
				sb_bst_restore_gbr_protectorate = yes
			}
		}
	}
}
