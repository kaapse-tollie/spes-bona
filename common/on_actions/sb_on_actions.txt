# ============================================================================
# Spes Bona — On Actions (Game Start Effects)
# ============================================================================
# Runs once at game start to fix homelands and set initial tensions.
#
# Why on_actions? Two V3 limitations force this approach:
#   1. remove_homeland is an effect — it cannot be used in state history files
#      (only add_homeland works there). So removals must happen here.
#   2. change_tension is an effect — it cannot be used in diplomacy history
#      files. Relations (set_relations) go in sb_relations.txt, but tensions
#      must be set here.
#
# Design philosophy: Tensions + negative relations push the AI toward declaring
# conventional V3 wars. Wars are NOT scripted via events — the AI decides based
# on these starting conditions. High tension (80-100) = war likely in 1-3 years.
# Moderate tension (40-60) = war possible but not immediate.
#
# Cross-references:
#   - Relations:         common/history/diplomacy/sb_relations.txt
#   - Homeland additions: common/history/states/00_states.txt
#   - State traits:      common/history/global/sb_state_traits.txt
# ============================================================================

sb_on_game_started = {
	effect = {
		# The vanilla Highveld JE is deliberately disabled by Spes Bona. Without a
		# live reference path, two of its endpoint flavor events show up as
		# orphaned in the startup log. Keep a dead-code reference under an actual
		# country scope so the event manager still sees them as wired while never
		# firing them in practice.
		c:CAP ?= {
			if = {
				limit = { always = no }
				trigger_event = { id = struggle_for_the_highveld.6 days = 1 popup = no }
				trigger_event = { id = struggle_for_the_highveld.7 days = 1 popup = no }
				trigger_event = { id = sb_cape.104 days = 1 popup = no }
				trigger_event = { id = sb_boer_zulu.056 days = 1 popup = no }
				trigger_event = { id = sb_boer_zulu.057 days = 1 popup = no }
				trigger_event = { id = sb_boer_zulu.071 days = 1 popup = no }
				trigger_event = { id = sb_trek.004 days = 1 popup = no }
			}
		}

		# ============================================================
		# Homeland corrections — enforcing the 50-100yr presence rule
		# ============================================================
		# Vanilla V3 assigns homelands too liberally in Southern Africa.
		# Many cultures are given homeland status in states where they had
		# little or no presence by 1836. These removals correct that.
		#
		# Counterpart additions are in common/history/states/00_states.txt.

		# Western Cape: remove sotho
		# Sotho peoples are indigenous to the Highveld (Lesotho/Free State),
		# not the Western Cape. Vanilla incorrectly assigns them here.
		s:STATE_CAPE_COLONY = {
			remove_homeland = cu:sotho
		}
		# Northern Cape: remove sotho
		# Same reasoning — Sotho presence here was minimal and recent.
		# The Northern Cape was historically Khoisan and later Griqua territory.
		s:STATE_NORTHERN_CAPE = {
			remove_homeland = cu:sotho
		}
		# Zululand (KZN): remove sotho
		# Zululand was Zulu/Nguni territory. Sotho homeland is Lesotho,
		# not KwaZulu-Natal. Vanilla conflates "Bantu" groups incorrectly.
		s:STATE_ZULULAND = {
			remove_homeland = cu:sotho
		}
		# Free State: remove griqua, nguni, boer
		# - griqua: Philippolis settlement only ~1826 (Adam Kok II). Too recent.
		# - nguni: Mfecane refugees (~1820s). No established homeland by 1836.
		# - boer: Great Trek hadn't happened yet (1836-38). No Boer settlement.
		# Correct homelands: sotho, tswana (long indigenous presence)
		s:STATE_VRYSTAAT = {
			remove_homeland = cu:griqua
			remove_homeland = cu:nguni
			remove_homeland = cu:boer
		}
		# Transvaal: remove zulu, boer
		# - zulu: Zulu homeland is KZN, not the Transvaal. The Ndebele (a Zulu
		#   offshoot) were here, but "zulu" culture represents the Zulu kingdom.
		# - boer: Great Trek hadn't reached the Transvaal by 1836.
		# Correct homelands: tswana, sotho, venda (long indigenous presence)
		s:STATE_TRANSVAAL = {
			remove_homeland = cu:zulu
			remove_homeland = cu:boer
		}
		s:STATE_EAST_TRANSVAAL = {
			remove_homeland = cu:zulu
			remove_homeland = cu:boer
		}
		s:STATE_NORTHERN_TRANSVAAL = {
			remove_homeland = cu:zulu
			remove_homeland = cu:boer
		}
		# Namaqualand: remove tswana
		# Tswana are indigenous to Botswana and the Northern Cape interior,
		# not the arid Namaqualand coast. This region was Nama/Khoisan territory.
		s:STATE_NAMAQUALAND = {
			remove_homeland = cu:tswana
		}
		# Botswana: remove herero, nguni
		# - herero: Herero homeland is Hereroland (Namibia), not Botswana.
		# - nguni: No established Nguni presence in Botswana by 1836.
		# Correct homelands: tswana (dominant), khoisan (indigenous San peoples)
		s:STATE_BOTSWANA = {
			remove_homeland = cu:herero
			remove_homeland = cu:nguni
		}

		# Southern African English/Boer agrarians should use the settler-colonialist
		# rural ideology rather than vanilla isolationism. This is no longer tied to
		# the Cape JE and applies to any qualifying stateholder in the region.
		every_country = {
			limit = {
				OR = {
					country_has_primary_culture = cu:british
					country_has_primary_culture = cu:boer
				}
				any_scope_state = {
					OR = {
						state_region = s:STATE_CAPE_COLONY
						state_region = s:STATE_EASTERN_CAPE
						state_region = s:STATE_NORTHERN_CAPE
						state_region = s:STATE_ZULULAND
						state_region = s:STATE_TRANSVAAL
						state_region = s:STATE_EAST_TRANSVAAL
						state_region = s:STATE_NORTHERN_TRANSVAAL
						state_region = s:STATE_VRYSTAAT
						state_region = s:STATE_NAMAQUALAND
						state_region = s:STATE_HEREROLAND
						state_region = s:STATE_BOTSWANA
						state_region = s:STATE_LOURENCO_MARQUES
					}
				}
			}
				ig:ig_rural_folk ?= {
					if = {
						limit = { has_ideology = ideology:ideology_isolationist }
						remove_ideology = ideology_isolationist
					}
					if = {
						limit = { NOT = { has_ideology = ideology:ideology_sb_settler_colonialist } }
						add_ideology = ideology_sb_settler_colonialist
					}
				}
		}

		# Force the trekker republics into their intended opening coalitions after
		# startup character generation has completed. ORA's V2 player-facing start
		# is anchored on Potgieter, Retief, and Maritz, so keep the opening
		# government tied to those named figures rather than to generated filler.
		c:ORA ?= {
			every_interest_group = {
				remove_ruling_interest_group = yes
			}
			ig:ig_armed_forces ?= {
				add_ruling_interest_group = yes
			}
			ig:ig_landowners ?= {
				add_ruling_interest_group = yes
			}
			ig:ig_petty_bourgeoisie ?= {
				add_ruling_interest_group = yes
			}
		}
		c:TRN ?= {
			every_interest_group = {
				remove_ruling_interest_group = yes
			}
			ig:ig_armed_forces ?= {
				add_ruling_interest_group = yes
			}
			ig:ig_landowners ?= {
				add_ruling_interest_group = yes
			}
			ig:ig_rural_folk ?= {
				add_ruling_interest_group = yes
			}
			ig:ig_devout ?= {
				add_ruling_interest_group = yes
			}
		}

		# ============================================================
		# Initial tensions — drive AI behavior in the first decades
		# ============================================================
		# Tension is a hidden value (0-100) that affects AI willingness
		# to start diplomatic plays. Combined with negative relations
		# (sb_relations.txt), high tension makes early wars likely.

		# --- Cape-Xhosa Frontier ---
		# The 6th Frontier War (1834-36) had just ended. Tensions remained
		# extremely high, especially with the Rharhabe who lost territory.
		# These values make the 7th Frontier War (historically 1846) plausible
		# within the first 5-10 years of gameplay.
			c:CAP ?= {
				if = {
					limit = {
						has_journal_entry = je_sb_cape_politics
				}
				if = {
					limit = {
						NOT = { has_variable = sb_cape_balance_current }
					}
					set_variable = { name = sb_cape_balance_current value = 0 }
				}
				if = {
					limit = {
						NOT = { has_variable = sb_cape_tensions_initialized_var }
						NOT = { has_variable = sb_cape_tensions_event_pending_var }
					}
					set_variable = sb_cape_tensions_event_pending_var
					trigger_event = { id = sb_cape.001 days = 7 popup = yes }
				}
				every_political_lobby = {
					limit = {
						is_political_lobby_type = lobby_pro_overlord
						target = c:GBR
					}
					remove_variable = lobby_custom_name_var_15
					remove_variable = lobby_custom_name_var_16
					remove_variable = lobby_custom_name_var_17
					remove_variable = lobby_custom_name_var_18
					remove_variable = lobby_custom_name_var_19
					remove_variable = lobby_custom_name_var_20
					set_variable = lobby_custom_name_var_15
				}
				every_political_lobby = {
					limit = {
						is_political_lobby_type = lobby_anti_overlord
						target = c:GBR
					}
					remove_variable = lobby_custom_name_var_7
					remove_variable = lobby_custom_name_var_8
					remove_variable = lobby_custom_name_var_9
					remove_variable = lobby_custom_name_var_10
					remove_variable = lobby_custom_name_var_11
					remove_variable = lobby_custom_name_var_12
					remove_variable = lobby_custom_name_var_13
					remove_variable = lobby_custom_name_var_14
						set_variable = lobby_custom_name_var_14
					}
				}
				if = {
					limit = {
						NOT = { has_truce_with = c:XHT }
					}
					create_bidirectional_truce = {
						country = c:XHT
						months = 48
					}
				}
				change_tension = {
					country = c:XHR
					value = 100
			}
			change_tension = {
				country = c:XHG
				value = 100
			}
			change_tension = {
				country = c:XHT
				value = 100
			}
		}

		# --- Voortrekker-Ndebele Conflict ---
		# In the v2 setup the inland trek starts as one Voortrekker polity under
		# ORA, so the opening Highveld war is now MTB against ORA directly.
		c:ORA ?= {
			change_tension = {
				country = c:MTB
				value = 90
			}
			if = {
				limit = {
					is_at_war = no
					aggressive_diplomatic_plays_permitted = yes
					c:MTB ?= {
						NOT = { has_variable = sb_ndebele_opening_play_fired }
						is_country_alive = yes
						is_at_war = no
						aggressive_diplomatic_plays_permitted = yes
					}
				}
				# Native uprisings are vanilla plays whose initiator must be the
				# decentralized country itself. Fire the opening event from MTB,
				# not TRN, or the play will either fail to start or present the
				# default colonization-rights article on the wrong side.
				c:MTB ?= {
					trigger_event = { id = sb_trek.001 days = 90 }
				}
			}
		}
		# Ndebele reciprocal tension — Mzilikazi now reads the inland
		# Voortrekker camps as a single advancing frontier under ORA.
		c:MTB ?= {
			change_tension = {
				country = c:ORA
				value = 90
			}
		}

		# --- Orange Free State Frontier ---
		# ORA-PHL tension 50: Griqua-Boer land competition around Philippolis.
		# Adam Kok III's Griqua territory was being encroached upon.
		# Historically led to the 1845 Kok-Maitland Treaty and later conflict.
		# Tension 50 makes war possible but not immediate — correct for the
		# slower-burning Griqua-Boer rivalry.
		c:ORA ?= {
			change_tension = {
				country = c:PHL
				value = 50
			}
			# ORA-BST tension 15: Boer-Basotho friction over the Caledon River valley
			# should exist at start, but it should not immediately collapse into a
			# maximal annexation war before the historical escalation of the 1850s.
			change_tension = {
				country = c:BST
				value = 15
			}
		}

	}
}

sb_on_game_started_after_lobby = {
	effect = {
		# Startup treaties are now history-authored in the local
		# 00_historical_treaties override. Do not withdraw/recreate them here:
		# doing so creates visible treaty banners and "broken obligation" scars.

		# The Nguni JEs now auto-instance reliably from their JE definitions.
		# Retaining the old after-lobby fallback only creates duplicate-startup
		# warnings, so it stays disabled unless a future regression reappears.

		c:MTB ?= {
			if = {
				limit = { has_modifier = sb_native_conscription_MTB_player }
				remove_modifier = sb_native_conscription_MTB_player
			}
			if = {
				limit = { has_modifier = sb_native_conscription_MTB }
				remove_modifier = sb_native_conscription_MTB
			}
			if = {
				limit = {
					c:ORA ?= {
						is_country_alive = yes
						is_player = yes
					}
				}
				add_modifier = {
					name = sb_native_conscription_MTB_player
					months = -1
				}
			}
		}
		c:GBR ?= {
			if = {
				limit = { has_modifier = repudiated_obligation }
				remove_modifier = repudiated_obligation
			}
		}
	}
}

sb_on_spes_bona_colony_created = {
	effect = {
		# V2 Potchefstroom split: once ORA establishes a real foothold in the west
		# Transvaal core, surface the delayed TRN republic split immediately
		# instead of waiting on the old Blood River/date gate.
		if = {
			limit = {
				state_region = s:STATE_TRANSVAAL
				owner = {
					country_definition = cd:ORA
					is_country_alive = yes
					NOT = { has_variable = sb_trn_v2_spawned_var }
					NOT = { has_variable = sb_trn_v2_spawn_pending_var }
					NOT = { exists = c:TRN }
					# Inline the foothold check on the action path too so the split can
					# fire as soon as a real West Transvaal colony is created, without
					# depending on a separate scripted trigger resolving during boot.
					OR = {
						p:xD0A010.state.owner = this
						p:x8BDC4D.state.owner = this
						p:xA22DC2.state.owner = this
						p:xD02190.state.owner = this
						p:x6C1786.state.owner = this
						p:x9A137B.state.owner = this
					}
				}
			}
			owner = {
				set_variable = {
					name = sb_trn_v2_spawn_pending_var
					days = 21
				}
				if = {
					limit = {
						NOT = {
							c:NAL ?= {
								is_country_alive = yes
								is_player = yes
							}
						}
					}
					trigger_event = { id = sb_trek.006 days = 1 popup = yes }
				}
			}
			if = {
				limit = {
					c:NAL ?= {
						is_country_alive = yes
						is_player = yes
					}
				}
				c:NAL ?= {
					trigger_event = { id = sb_trek.006 days = 1 popup = yes }
				}
			}
		}
	}
}

sb_on_cape_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				OR = {
					country_has_primary_culture = cu:british
					country_has_primary_culture = cu:boer
				}
				any_scope_state = {
					OR = {
						state_region = s:STATE_CAPE_COLONY
						state_region = s:STATE_EASTERN_CAPE
						state_region = s:STATE_NORTHERN_CAPE
						state_region = s:STATE_ZULULAND
						state_region = s:STATE_TRANSVAAL
						state_region = s:STATE_EAST_TRANSVAAL
						state_region = s:STATE_NORTHERN_TRANSVAAL
						state_region = s:STATE_VRYSTAAT
						state_region = s:STATE_NAMAQUALAND
						state_region = s:STATE_HEREROLAND
						state_region = s:STATE_BOTSWANA
						state_region = s:STATE_LOURENCO_MARQUES
					}
				}
			}
				ig:ig_rural_folk ?= {
					if = {
						limit = { has_ideology = ideology:ideology_isolationist }
						remove_ideology = ideology_isolationist
					}
					if = {
						limit = { NOT = { has_ideology = ideology:ideology_sb_settler_colonialist } }
						add_ideology = ideology_sb_settler_colonialist
					}
				}
		}

		# Eastern Cape resistance is owner-agnostic. Any country that currently
		# holds the state should remove the trait once the Xhosa polities are gone,
		# and while the trait exists the three Xhosa tags should keep their
		# differentiated tension-decay modifiers.
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_EASTERN_CAPE
					has_state_trait = state_trait_sb_xhosa_resistance
				}
			}
			if = {
				limit = {
					NOT = { c:XHG ?= { is_country_alive = yes } }
					NOT = { c:XHR ?= { is_country_alive = yes } }
					NOT = { c:XHT ?= { is_country_alive = yes } }
				}
				s:STATE_EASTERN_CAPE = {
					remove_state_trait = state_trait_sb_xhosa_resistance
				}
			}
		}

		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_EASTERN_CAPE
					has_state_trait = state_trait_sb_xhosa_resistance
				}
			}
			c:XHG ?= {
				if = {
					limit = {
						NOT = { has_modifier = sb_xhosa_frontier_pressure_xhg }
					}
					add_modifier = {
						name = sb_xhosa_frontier_pressure_xhg
					}
				}
			}
			c:XHR ?= {
				if = {
					limit = {
						NOT = { has_modifier = sb_xhosa_frontier_pressure_xhr }
					}
					add_modifier = {
						name = sb_xhosa_frontier_pressure_xhr
					}
				}
			}
			c:XHT ?= {
				if = {
					limit = {
						NOT = { has_modifier = sb_xhosa_frontier_pressure_xht }
					}
					add_modifier = {
						name = sb_xhosa_frontier_pressure_xht
					}
				}
			}
		}
		else = {
			c:XHG ?= {
				if = {
					limit = { has_modifier = sb_xhosa_frontier_pressure_xhg }
					remove_modifier = sb_xhosa_frontier_pressure_xhg
				}
			}
			c:XHR ?= {
				if = {
					limit = { has_modifier = sb_xhosa_frontier_pressure_xhr }
					remove_modifier = sb_xhosa_frontier_pressure_xhr
				}
			}
			c:XHT ?= {
				if = {
					limit = { has_modifier = sb_xhosa_frontier_pressure_xht }
					remove_modifier = sb_xhosa_frontier_pressure_xht
				}
			}
			}

			if = {
				limit = {
					country_definition = cd:CAP
					has_journal_entry = je_sb_cape_politics
				}

				if = {
					limit = {
						NOT = { has_variable = sb_cape_balance_current }
					}
				set_variable = { name = sb_cape_balance_current value = 0 }
			}

			je:je_sb_cape_politics = {
				if = {
					limit = {
						has_variable = sb_cape_tensions_initialized_var
						NOT = { has_modifier = sb_cape_political_tensions }
					}
					add_modifier = {
						name = sb_cape_political_tensions
					}
				}
			}

			set_variable = {
				name = sb_cape_balance_current
				value = "je:je_sb_cape_politics.scripted_bar_progress(sb_cape_balance_bar)"
			}

			# Keep the JE-side "ascendant" modifiers attached to the journal entry
			# itself so the player sees them in the JE UI rather than as hidden
			# country state.
			je:je_sb_cape_politics = {
				if = {
					limit = { has_modifier = sb_cape_liberal_ascendant }
					remove_modifier = sb_cape_liberal_ascendant
				}
				if = {
					limit = { has_modifier = sb_cape_settler_ascendant }
					remove_modifier = sb_cape_settler_ascendant
				}
				if = {
					limit = {
						root.var:sb_cape_balance_current >= 50
					}
					add_modifier = { name = sb_cape_liberal_ascendant }
				}
				else_if = {
					limit = {
						root.var:sb_cape_balance_current <= -50
					}
					add_modifier = { name = sb_cape_settler_ascendant }
				}
			}

			# Once either side becomes dominant, surface a one-off demand for the
			# matching law instead of leaving the pressure hidden in the bar drift.
			if = {
				limit = {
					var:sb_cape_balance_current <= -50
					NOT = {
						OR = {
							has_law = law_type:law_cultural_exclusion
							has_law_or_variant = law_type:law_multicultural
						}
					}
					NOT = {
						OR = {
							is_enacting_law = law_type:law_cultural_exclusion
							is_enacting_law = law_type:law_multicultural

				}
			}

				NOT = { has_variable = sb_trn_v2_spawn_pending_var }
				NOT = { exists = c:TRN }

			}

			}

				limit = {
					NOT = {
						c:NAL ?= {
							is_country_alive = yes
							is_player = yes
						}
					}
				}

				limit = {
					c:NAL ?= {
						is_country_alive = yes
						is_player = yes
					}
				}
				c:NAL ?= {
					trigger_event = { id = sb_trek.006 days = 1 popup = yes }
				}
			}
		}

		# Legacy compatibility for pre-V2 saves where TRN could still appear in the
		# Natal crisis. Fresh V2 starts should never set this up, but keeping the
		# hook avoids stranding those older branches.

			limit = {
				country_definition = cd:TRN
				is_ai = yes
				has_variable = sb_trn_joined_natal_war_var
				is_at_war = yes
			}

				has_variable = sb_natal_war_active_var
				is_at_war = yes
			}
			sb_ensure_ora_ai_natal_front_commitment = yes
		}

			limit = {
				country_definition = cd:ORA

				NOT = { has_variable = sb_natal_path_resolved_var }
				NOT = { has_variable = sb_natal_diplomacy_started_var }

		# time to breathe before London moves.

			limit = {
				country_definition = cd:NAL
				country_has_primary_culture = cu:boer
				NOT = { has_journal_entry = je_sb_great_trek }
				NOT = { has_variable = sb_early_republic_var }
				NOT = { has_variable = sb_natalia_trek_normalized_var }
			}

		}

				NOT = { has_variable = sb_british_natal_ultimatum_var }
				NOT = { has_variable = sb_british_natal_ultimatum_pending_var }
				c:NAL ?= {

				1 = { trigger_event = { id = sb_boer_zulu.100 days = 270 popup = yes } }
				1 = { trigger_event = { id = sb_boer_zulu.100 days = 360 popup = yes } }
				1 = { trigger_event = { id = sb_boer_zulu.100 days = 450 popup = yes } }
				1 = { trigger_event = { id = sb_boer_zulu.100 days = 540 popup = yes } }
				1 = { trigger_event = { id = sb_boer_zulu.100 days = 630 popup = yes } }
				1 = { trigger_event = { id = sb_boer_zulu.100 days = 720 popup = yes } }
			}
		}

		# Catch-up path so ongoing saves also receive the Lydenburg reunion JE if

			if = {

					country_definition = cd:TRN
					NOT = { has_journal_entry = je_sb_transvaal_unity }

			if = {
				limit = {

					is_ai = yes

					c:LYD ?= {
						is_country_alive = yes
						relations:root < relations_threshold:amicable
					}
				}
				set_variable = {
					name = sb_trn_lyd_relations_nudge_cooldown_var
					months = 6
				}
				change_relations = {
					country = c:LYD
					value = 10
				}
			}

			if = {
				limit = {
				country_definition = cd:TRN
				has_modifier = sb_trek_frontier_republic
				NOT = { has_journal_entry = je_sb_pacifying_eastern_transvaal }
				s:STATE_EAST_TRANSVAAL = {
					NOT = { is_homeland = cu:boer }
				}
			}
			add_journal_entry = { type = je_sb_pacifying_eastern_transvaal }
		}

		# SWZ can only seek British shelter after the Boer pressure event is
		# resolved and Natal is already inside the British orbit.
		if = {
			limit = {
				country_definition = cd:SWZ
				is_subject = no
				has_variable = sb_swazi_boer_pressure_resolved_var
				NOT = { has_variable = sb_swazi_british_accommodation_var }
				NOT = { has_variable = sb_swazi_stand_apart_var }
				NOT = { has_variable = sb_swazi_british_accommodation_pending_var }

				name = sb_swazi_british_accommodation_pending_var

			}

# War-goal hooks for the Phase 1 frontier wars.
sb_on_spes_bona_wargoal_enforced = {
	effect = {
		if = {
			limit = {
				country_definition = cd:ZUL
				NOT = { has_variable = sb_zulu_generic_war_resolution_recent_var }
				NOT = { has_variable = sb_natal_war_active_var }
				NOT = { has_variable = sb_port_natal_british_war_var }
				NOT = { has_variable = sb_zulu_swazi_war_active_var }
			}
			sb_zulu_apply_generic_war_victory = yes
		}

		if = {
			limit = {
				scope:target ?= {
					country_definition = cd:ZUL
					NOT = { has_variable = sb_zulu_generic_war_resolution_recent_var }
					NOT = { has_variable = sb_natal_war_active_var }
					NOT = { has_variable = sb_port_natal_british_war_var }
					NOT = { has_variable = sb_zulu_swazi_war_active_var }
				}
			}
			scope:target = {
				sb_zulu_apply_generic_war_defeat = yes
			}
		}

		if = {
			limit = {
				country_definition = cd:ZUL
				has_variable = sb_zulu_swazi_war_active_var
				scope:target ?= {
					country_definition = cd:SWZ
					has_variable = sb_zulu_swazi_war_active_var
				}
			}
			remove_variable = sb_zulu_swazi_war_active_var
			scope:target = {
				remove_variable = sb_zulu_swazi_war_active_var
			}
			sb_zulu_apply_swazi_campaign_victory = yes
		}

		if = {
			limit = {
				country_definition = cd:SWZ
				has_variable = sb_zulu_swazi_war_active_var
				scope:target ?= {
					country_definition = cd:ZUL
					has_variable = sb_zulu_swazi_war_active_var
				}
			}
			remove_variable = sb_zulu_swazi_war_active_var
			scope:target = {
				remove_variable = sb_zulu_swazi_war_active_var
				sb_zulu_apply_swazi_campaign_defeat = yes
			}
		}

		# ORA wins the Natal crisis war.
		if = {
			limit = {
				country_definition = cd:ORA
				scope:target ?= {
					country_definition = cd:ZUL
					has_variable = sb_natal_war_active_var
				}
				has_variable = sb_natal_war_active_var
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_sb_natal_crisis
				}
			}
			remove_variable = sb_natal_war_active_var
			scope:target = {
				remove_variable = sb_natal_war_active_var
			}
			trigger_event = { id = sb_boer_zulu.070 days = 3 popup = yes }
		}

		# ZUL wins the Natal crisis war.
		if = {
			limit = {
				country_definition = cd:ZUL
				scope:target ?= {
					country_definition = cd:ORA
					has_variable = sb_natal_war_active_var
				}
				has_variable = sb_natal_war_active_var
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_sb_natal_crisis
				}
			}
			remove_variable = sb_natal_war_active_var
			scope:target = {
				remove_variable = sb_natal_war_active_var
			}
			trigger_event = { id = sb_boer_zulu.080 days = 3 popup = yes }
		}

		# Britain annexes ZUL and reorganizes the coast under a Natal colony shell.
		if = {
			limit = {
				country_definition = cd:GBR
				scope:target ?= {
					country_definition = cd:ZUL
				}
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_annex_war
				}
			}
			scope:target = {
				if = {
					limit = { has_variable = sb_port_natal_british_war_var }
					remove_variable = sb_port_natal_british_war_var
				}
				if = {
					limit = { has_variable = sb_port_natal_raid_var }
					remove_variable = sb_port_natal_raid_var
				}
			}
			trigger_event = { id = sb_boer_zulu.091 days = 7 popup = yes }
		}

		# A surviving victory over the British should feel exceptional.
		if = {
			limit = {
				country_definition = cd:ZUL
				scope:target ?= {
					country_definition = cd:GBR
				}
				has_variable = sb_port_natal_british_war_var
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_annex_war
				}
			}
			remove_variable = sb_port_natal_british_war_var
			if = {
				limit = { has_variable = sb_port_natal_raid_var }
				remove_variable = sb_port_natal_raid_var
			}
			trigger_event = { id = sb_boer_zulu.092 days = 7 popup = yes }
		}

		# ORA wins the opening Ndebele war and the inland trek consolidates.
		if = {
			limit = {
				country_definition = cd:ORA
				scope:target ?= {
					country_definition = cd:MTB
				}
				scope:diplomatic_play ?= {
					is_diplomatic_play_type = dp_native_uprising
				}
			}
			trigger_event = { id = sb_trek.002 days = 3 popup = yes }
		}
	}
}
