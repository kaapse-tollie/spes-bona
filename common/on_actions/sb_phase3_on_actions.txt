# Spes Bona - Phase 3 bootstrap hooks
#
# Phase 3 begins as a late-century routing layer rather than as a single event.
# These hooks do three jobs:
#   1. detect when the save crosses into the 1867+ mineral-revolution window
#   2. snapshot the Phase 1-2 branch outcomes once, so later content can read
#      a stable late-century branch interface instead of live-reacting forever
#   3. hand the first visible Wave 1 content to CAP when possible, or GBR as
#      the fallback imperial host when the Cape no longer exists independently

on_game_started_after_lobby = {
	on_actions = { sb_on_phase3_after_lobby }
}

on_monthly_pulse_country = {
	on_actions = { sb_on_phase3_monthly_pulse_country }
}

sb_on_phase3_after_lobby = {
	effect = {
		c:GBR ?= {
			if = {
				limit = {
					sb_can_start_phase3_wave1 = yes
					NOT = { has_variable = sb_phase3_initialized_var }
					NOT = { has_variable = sb_phase3_bootstrap_pending_var }
				}
				set_variable = sb_phase3_bootstrap_pending_var
				trigger_event = { id = sb_phase3.001 days = 1 popup = no }
			}
		}
	}
}

sb_on_phase3_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				sb_is_phase3_coordinator_country = yes
				sb_can_start_phase3_wave1 = yes
				NOT = { has_variable = sb_phase3_initialized_var }
				NOT = { has_variable = sb_phase3_bootstrap_pending_var }
			}
			set_variable = sb_phase3_bootstrap_pending_var
			trigger_event = { id = sb_phase3.001 days = 1 popup = no }
		}
	}
}
