# Re-assert the general role for Boer commandant-generals after startup and
# on a light quarterly cadence during play. This preserves the intended
# ruler-commander behavior without fighting vanilla retirement logic every month.

on_game_started_after_lobby = {
	on_actions = { sb_on_boer_commandant_started_after_lobby }
}

on_monthly_pulse_country = {
	on_actions = { sb_on_boer_commandant_monthly_pulse_country }
}

sb_on_boer_commandant_started_after_lobby = {
	effect = {
		every_country = {
			limit = {
				has_law = law_type:law_sb_commandant_general_republic
			}
			sb_refresh_boer_commandant_general_government = yes
			sb_ensure_commandant_general_ruler_command = yes
			sb_ensure_commandant_general_ruler_landowner_leadership = yes
		}
	}
}

sb_on_boer_commandant_monthly_pulse_country = {
	effect = {
		if = {
			limit = {
				has_law = law_type:law_sb_commandant_general_republic
				NOT = { has_variable = sb_commandant_quarterly_check_var }
			}
			set_variable = {
				name = sb_commandant_quarterly_check_var
				days = 95
			}
			sb_ensure_commandant_general_ruler_command = yes
		}
	}
}
