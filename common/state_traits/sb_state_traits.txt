# Spes Bona - State Traits
# Cross-ref: common/history/states/ for where these are assigned at game start.

# Xhosa Resistance — Represents active Xhosa resistance to colonial encroachment
# in the Eastern Cape. This should hinder any colonizing power, not just CAP.
# REMOVAL: The trait is stripped once the Xhosa frontier polities (XHG, XHR, XHT)
# are all gone, regardless of whether the state ends up under CAP, ABY, or some
# unrelated outside power.
# Current Phase 1 target: slow down, but do not freeze, colonization into the
# frontier while the Xhosa polities still exist.
state_trait_sb_xhosa_resistance = {
	icon = "gfx/interface/icons/state_trait_icons/river.dds"

	modifier = {
		state_colony_growth_speed_mult = -0.25
		state_mortality_mult = 0.02
		building_mobilization_cost_mult = 0.05
	}
}

# Thaba Bosiu Mountain Fortress — Defensive terrain advantage for Basutoland (modern Lesotho).
# Thaba Bosiu was the fortified mountain capital of King Moshoeshoe I, founder of the Basotho nation.
# It successfully repelled attacks by Boer commandos (1852, 1858) and British forces (1852).
# This remains the terrain-side Basotho defensive flavor. Country-wide BST
# defense was intentionally removed in favor of Moshoeshoe's commander setup.
state_trait_sb_mountain_fortress = {
	icon = "gfx/interface/icons/state_trait_icons/mountain.dds"

	modifier = {
		unit_defense_mult = 0.25
		building_mobilization_cost_mult = -0.15
	}
}
