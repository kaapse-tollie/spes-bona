# Spes Bona country-color overrides for South African colonial transitions.
# Use dynamic country map colors rather than subject-type overrides so CAP/ABY
# can keep bespoke dominion/independence colors without globally changing how
# vanilla colonies and dominions behave.

cap_dominion_color = {
	color = "sb_cap_dominion_red"

	possible = {
		exists = c:CAP
		c:CAP ?= THIS
		has_variable = sb_cap_dominion_map_color_var
	}
}

cap_independent_color = {
	color = "sb_cap_independent_blue"

	possible = {
		exists = c:CAP
		c:CAP ?= THIS
		is_subject = no
	}
}

aby_dominion_color = {
	color = "sb_aby_dominion_red"

	possible = {
		exists = c:ABY
		c:ABY ?= THIS
		has_variable = sb_aby_dominion_map_color_var
	}
}

aby_independent_color = {
	color = "sb_aby_independent_orange"

	possible = {
		exists = c:ABY
		c:ABY ?= THIS
		is_subject = no
	}
}
