# Spes Bona - Dynamic Country Names
# V3 dynamic names override the base country tag name when trigger conditions are met.
# Higher priority values take precedence when multiple names match.
# Cross-ref: localization/ for the actual display strings (dyn_c_sb_* keys).

# TRN: Potchefstroom -> South African Republic
# REPLACE: overwrites any vanilla dynamic names for TRN.
# Historically, the Boers initially called their polity after Potchefstroom (est. 1838),
# the first major Voortrekker settlement in the Transvaal. It was renamed the
# South African Republic (Zuid-Afrikaansche Republiek / ZAR) upon consolidation under
# M.W. Pretorius in 1852 (Sand River Convention).
REPLACE:TRN = {
	# Potchefstroom — early small Boer republic name.
	# priority = 0: lowest priority, only used when sb_early_republic_var is set.
	# This variable is set at game start and cleared by the trek/consolidation JE.
	dynamic_country_name = {
		name = dyn_c_sb_potchefstroom
		adjective = dyn_c_sb_potchefstroom_adj

		is_main_tag_only = yes
		priority = 0

		trigger = {
			scope:actor ?= {
				has_variable = sb_early_republic_var
			}
		}
	}

	# South African Republic — after JE completion / political consolidation.
	# priority = 1: takes over once the early republic variable is cleared.
	# Represents the mature ZAR state recognized by Britain at the Sand River Convention.
	dynamic_country_name = {
		name = dyn_c_sb_south_african_republic
		adjective = dyn_c_sb_south_african_republic_adj

		is_main_tag_only = yes
		priority = 1

		trigger = {
			scope:actor ?= {
				NOT = { has_variable = sb_early_republic_var }
			}
		}
	}
}
